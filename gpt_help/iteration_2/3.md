# 1) `Code in JavaScript12`

## Что исправляет

* корректно читает `rawdata`
* enrich `self_assessments.full_name`
* возвращает **n8n items**, а не plain array

## Вставить вместо текущего кода

```js
const raw = $input.first().json.rawdata || [];

const employeeMap = new Map();
let modifiedCount = 0;

// Собираем employee_id -> full_name
for (const block of raw) {
  if (Array.isArray(block.employees)) {
    for (const emp of block.employees) {
      if (!emp?.employee_id) continue;
      employeeMap.set(String(emp.employee_id), emp.full_name || '');
    }
  }
}

// Обогащаем self_assessments
for (const block of raw) {
  if (Array.isArray(block.self_assessments)) {
    for (const assessment of block.self_assessments) {
      const employeeId = assessment?.employee_id;
      if (!employeeId) continue;

      const fullName = employeeMap.get(String(employeeId));
      if (fullName) {
        assessment.full_name = fullName;
        modifiedCount++;
      }
    }
  }
}

console.log(`Employees found: ${employeeMap.size}`);
console.log(`Self assessments enriched: ${modifiedCount}`);

return raw.map(block => ({ json: block }));
```

## Возможные ошибки

* `rawdata` пустой → нода вернёт пустой массив
* `employee_id` в employees и self_assessments разного типа → поэтому я привожу к `String(...)`

const inputItems = $input.all().map(i => i.json);

function normalizeRoleMatrix(roleMatrix) {
  if (!roleMatrix) return null;

  const competencyGroups =
    roleMatrix.compentecy_groups ||
    roleMatrix.competency_groups ||
    [];

  const competencies = competencyGroups.flatMap((group, groupIndex) =>
    (group.competencies || []).map((comp, compIndex) => ({
      competency_id:
        comp.competency_id ||
        comp.id ||
        `${roleMatrix.role || 'role'}__${groupIndex}__${compIndex}`,
      competency_name: comp.competency_name || comp.name || '',
      name: comp.name || comp.competency_name || '',
      description: comp.description || '',
      type: comp.type || '',
      category: comp.category || group.group_name || '',
      group_name: group.group_name || '',
      levels: comp.levels || [],
      priority_weight:
        comp.priority_weight ??
        group.priority_weight ??
        0.5
    }))
  );

  return {
    ...roleMatrix,
    competency_groups: competencyGroups,
    competencies
  };
}

const employees = [];
const matrices = [];
const learningAssets = [];
const selfAssessments = [];
const employeeIssues = [];
const dataQualityIssues = [];

for (const block of inputItems) {
  if (Array.isArray(block.employees)) employees.push(...block.employees);
  if (Array.isArray(block.matrices)) matrices.push(...block.matrices.map(normalizeRoleMatrix));
  if (Array.isArray(block.learning_assets)) learningAssets.push(...block.learning_assets);
  if (Array.isArray(block.self_assessments)) selfAssessments.push(...block.self_assessments);
  if (Array.isArray(block.employee_issues)) employeeIssues.push(...block.employee_issues);
  if (Array.isArray(block.data_quality_issues)) dataQualityIssues.push(...block.data_quality_issues);
}

const results = [];

for (const employee of employees) {
  const selfAssessment =
    selfAssessments.find(sa => String(sa.employee_id) === String(employee.employee_id)) || null;

  const roleMatrix =
    matrices.find(m =>
      String(m.role || '').toLowerCase() === String(employee.role || '').toLowerCase()
    ) || null;

  const employeeDataQualityIssues = [
    ...employeeIssues.filter(i => String(i.employee_id) === String(employee.employee_id)),
    ...dataQualityIssues.filter(i => String(i.employee_id) === String(employee.employee_id))
  ];

  results.push({
    json: {
      employee,
      self_assessment: selfAssessment,
      role_matrix: roleMatrix,
      learning_assets_pool: learningAssets,
      employee_data_quality_issues: employeeDataQualityIssues,
      config: {
        max_competencies_for_llm: 12,
        max_competencies_per_quarter: 4,
        plan_duration_months: 6,
        quarters: 2,
        assessment_date: new Date().toISOString().slice(0, 10),
        review_period_start: '',
        review_period_end: '',

}

# 3) Новая нода `Build Competency Items`

## Зачем нужна

Сейчас `Score Competencies` ожидает `competency_items`, но выше они не формируются. Это один из главных разрывов контракта.

## Создайте новую Code node

Название:
`Build Competency Items`

## Поставить между

`Build Employee Context` / `Build Employee Context1` → `Score Competencies`

## Код

```js
const roleMatrix = $json.role_matrix || {};
const selfAssessment = $json.self_assessment || {};
const competencies = roleMatrix.competencies || [];

const assessments = Array.isArray(selfAssessment.assessments)
  ? selfAssessment.assessments
  : [];

const latestAssessment =
  assessments.length > 0 ? assessments[assessments.length - 1] : null;

const saItems = Array.isArray(latestAssessment?.items)
  ? latestAssessment.items
  : [];

const saMap = new Map(
  saItems.map(item => [String(item.competency_id || item.competency_name || ''), item])
);

function detectTargetLevel(comp) {
  const levels = Array.isArray(comp.levels) ? comp.levels : [];
  if (!levels.length) return null;

  const numericLevels = levels
    .map(l => Number(l.level))
    .filter(v => !Number.isNaN(v));

  if (!numericLevels.length) return null;

  return Math.max(...numericLevels) >= 3 ? 3 : Math.max(...numericLevels);
}

const competency_items = competencies.map(comp => {
  const key = String(comp.competency_id || comp.competency_name || '');
  const sa = saMap.get(key) || null;

  const selectedLevel =
    sa?.self_selected_level ??
    sa?.selected_level ??
    sa?.level ??
    null;

  return {
    competency: comp,
    competency_id: comp.competency_id,
    competency_name: comp.competency_name || comp.name || '',
    target_level: detectTargetLevel(comp),
    self_assessment_item: {
      competency_id: comp.competency_id,
      selected_level: selectedLevel,
      comment: sa?.comment || ''
    },
    jira_evidence: []
  };
});

return [{
  json: {
    ...$json,
    competency_items
  }
}];
```

## Возможные ошибки

* если в self-assessment нет `items`, всё равно будет валидный массив competency_items
* если нет role matrix, массив будет пустым — это ок, но тогда лучше отдельно логировать проблему

# 4) `Score Competencies`

## Что рекомендую

Оставить ноду, но заменить на более устойчивый скоринг.

## Код

```js
const competency_items = $json.competency_items || [];
const config = $json.config || {};

function toNumberOrNull(v) {
  if (v === null || v === undefined || v === '') return null;
  const n = Number(v);
  return Number.isNaN(n) ? null : n;
}

function scoreItem(item) {
  const targetLevel = toNumberOrNull(item.target_level);
  const selectedLevel = toNumberOrNull(item.self_assessment_item?.selected_level);
  const priorityWeight = Number(item.competency?.priority_weight ?? config.default_competency_priority_weight ?? 0.5);

  let gapScore = 0;
  if (targetLevel !== null && selectedLevel !== null) {
    gapScore = Math.max(targetLevel - selectedLevel, 0);
  } else if (targetLevel !== null && selectedLevel === null) {
    gapScore = 1.5;
  }

  const hasComment = item.self_assessment_item?.comment ? 0.2 : 0;
  const hasEvidence = Array.isArray(item.jira_evidence) && item.jira_evidence.length ? -0.3 : 0;

  return Number((gapScore * 2 + priorityWeight + hasComment + hasEvidence).toFixed(3));
}

const competency_items_scored = competency_items.map(item => ({
  ...item,
  candidate_score: scoreItem(item)
}));

return [{
  json: {
    ...$json,
    competency_items_scored
  }
}];
```

# 5) `Basic LLM Chain1`

## Проблема

Сейчас промпт, скорее всего, использует `{{$json.payload}}`, но в `Code in JavaScript3` payload не создаётся.

## Исправление

В system/user prompt вместо:

```txt
{{ JSON.stringify($json.payload) }}
```

используйте:

```txt
{{ JSON.stringify($json) }}
```

Ты анализируешь одну компетенцию сотрудника.

{
  "competency_id": "string",
  "competency_name": "string",
  "self_level": 0,
  "estimated_level": 0,
  "confidence": "low|medium|high",

* видно весь item целиком
* меньше шансов сломать контракт между нодами

const competency_results = Array.isArray($json.competency_results)
  ? $json.competency_results
  : [];

const employee_summary = $json.employee_summary || {};

const sortedByPriority = [...competency_results].sort((a, b) => {
  const weight = { high: 3, medium: 2, low: 1 };
  return (weight[b.priority] || 0) - (weight[a.priority] || 0);

const top_priority_competencies = sortedByPriority.slice(0, 6);

const validResultsCount = competency_results.filter(r => r && r.competency_id).length;
const analysis_quality =
  validResultsCount >= 6 ? 'high'

  : 'low';

if (!employee_summary || !employee_summary.overall_summary) flags.push('missing_employee_summary');

return [{
  json: {
    ...$json,
    competency_analysis: {
      competency_results,
      employee_summary,
      top_priority_competencies,
      analysis_quality,
      flags
    }
  }
}];

function bulletList(items) {
  if (!Array.isArray(items) || !items.length) return '- Нет данных';
  return items.map(x => `- ${x}`).join('\n');
}

function safe(v) {
  return v === null || v === undefined || v === '' ? '—' : String(v);
}

function renderPlanQuarter(items, quarterTitle) {
  if (!Array.isArray(items) || !items.length) {
    return `### ${quarterTitle}\n\nНет запланированных компетенций.\n`;
  }

  let out = `### ${quarterTitle}\n\n`;
  items.forEach((item, idx) => {
    out += `#### ${idx + 1}. ${safe(item.competency_name)} \`${safe(item.competency_id)}\`\n\n`;
    out += `**Почему выбрано:** ${safe(item.reason)}\n\n`;
    out += `**Целевой переход:** ${safe(item.target_from)} → ${safe(item.target_to)}\n\n`;

    out += `**Действия:**\n`;
    if (Array.isArray(item.actions) && item.actions.length) {
      item.actions.forEach(a => {
        out += `- [${safe(a.type)}] **${safe(a.title)}** — ${safe(a.description)}\n`;
      });
    } else {
      out += `- Нет действий\n`;
    }
    out += `\n`;

    out += `**Критерии прогресса:**\n${bulletList(item.success_criteria)}\n\n`;
    out += `**Поддержка руководителя:**\n${bulletList(item.manager_support)}\n\n`;
  });

  return out;
}

function renderCompetencyTable(rows) {
  if (!Array.isArray(rows) || !rows.length) return 'Нет данных по компетенциям.\n';

  let out = '| Competency | Self | Estimated | Confidence | Priority | Focus |\n';
  out += '|---|---:|---:|---|---|---|\n';

  for (const r of rows) {
    out += `| ${safe(r.competency_id)} | ${safe(r.self_level)} | ${safe(r.estimated_level)} | ${safe(r.confidence)} | ${safe(r.priority)} | ${safe(r.development_focus)} |\n`;
  }

  return out;
}
