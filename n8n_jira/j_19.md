# 20. app/infrastructure/clients/n8n_client.py

```python
import httpx
from app.config import settings
from app.domain.exceptions import ExternalServiceError


class N8nClient:
    async def trigger_assessment(self, payload: dict) -> dict:
        headers = {
            "Authorization": f"Bearer {settings.n8n_api_token}",
            "Content-Type": "application/json",
        }

        async with httpx.AsyncClient(timeout=30.0) as client:
            response = await client.post(
                settings.n8n_webhook_url,
                json=payload,
                headers=headers,
            )

        if response.status_code >= 400:
            raise ExternalServiceError(
                f"n8n workflow trigger failed: {response.status_code} {response.text}"
            )

        return response.json() if response.content else {}
```

## app/domain/repositories/assessment_repository.py

```python
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.infrastructure.db.models.assessment_job import AssessmentJobOrm
from app.infrastructure.db.models.latest_assessment_snapshot import LatestAssessmentSnapshotOrm


class AssessmentRepository:
    def __init__(self, session: AsyncSession) -> None:
        self.session = session

    async def create_job(
        self,
        employee_id: str,
        lookback_days: int,
        force_refresh: bool,
    ) -> AssessmentJobOrm:
        job = AssessmentJobOrm(
            employee_id=employee_id,
            lookback_days=lookback_days,
            force_refresh=force_refresh,
            status="queued",
        )
        self.session.add(job)
        await self.session.commit()
        await self.session.refresh(job)
        return job

    async def get_job(self, job_id: str) -> AssessmentJobOrm | None:
        result = await self.session.execute(
            select(AssessmentJobOrm).where(AssessmentJobOrm.job_id == job_id)
        )
        return result.scalar_one_or_none()

    async def get_latest_snapshot(self, employee_id: str) -> LatestAssessmentSnapshotOrm | None:
        result = await self.session.execute(
            select(LatestAssessmentSnapshotOrm).where(
                LatestAssessmentSnapshotOrm.employee_id == employee_id
            )
        )
        return result.scalar_one_or_none()
```

## app/domain/services/assessment_service.py

```python
from app.domain.exceptions import NotFoundError
from app.domain.repositories.assessment_repository import AssessmentRepository
from app.domain.schemas.job import AssessmentJobDto
from app.domain.schemas.run_assessment import (
    RunAssessmentRequestDto,
    RunAssessmentResponseDto,
)
from app.domain.schemas.snapshot import SnapshotEnvelopeDto
from app.domain.services.workflow_service import WorkflowService
from app.infrastructure.mappers.job_mapper import map_job_to_dto
from app.infrastructure.mappers.snapshot_mapper import map_snapshot_payload_to_envelope


class AssessmentService:
    def __init__(
        self,
        assessment_repository: AssessmentRepository,
        workflow_service: WorkflowService,
    ) -> None:
        self.assessment_repository = assessment_repository
        self.workflow_service = workflow_service

    async def run_assessment(
        self,
        request: RunAssessmentRequestDto,
    ) -> RunAssessmentResponseDto:
        job = await self.assessment_repository.create_job(
            employee_id=request.employee_id,
            lookback_days=request.lookback_days,
            force_refresh=request.force_refresh,
        )

        await self.workflow_service.trigger_assessment_workflow(
            {
                "job_id": str(job.job_id),
                "employee_id": request.employee_id,
                "lookback_days": request.lookback_days,
                "force_refresh": request.force_refresh,
                "features": request.features.model_dump(),
            }
        )

        return RunAssessmentResponseDto(
            job_id=job.job_id,
            status=job.status,
        )

    async def get_job(self, job_id: str) -> AssessmentJobDto:
        job = await self.assessment_repository.get_job(job_id)
        if job is None:
            raise NotFoundError(f"Job not found: {job_id}")
        return map_job_to_dto(job)

    async def get_latest_snapshot(self, employee_id: str) -> SnapshotEnvelopeDto:
        row = await self.assessment_repository.get_latest_snapshot(employee_id)
        if row is None:
            raise NotFoundError(f"Latest snapshot not found for employee_id={employee_id}")
        return map_snapshot_payload_to_envelope(row.payload)
```
