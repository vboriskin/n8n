Вот вторая чистая **Code node для n8n**: она собирает результаты анализа по компетенциям в payload для summary LLM.

Предположение по входу:

* на вход этой ноды приходят **результаты LLM по отдельным компетенциям**
* каждый item уже выглядит примерно так:

```json
{
  "competency_id": "api_testing",
  "self_level": 2,
  "estimated_level": 2,
  "confidence": "medium",
  "assessment": "...",
  "evidence_for": ["..."],
  "evidence_missing": ["..."],
  "risk_flags": [],
  "priority": "high",
  "development_focus": "..."
}
```

# n8n Code node: `Build Employee Summary Payload`

Режим:

* **Run Once for All Items**

```javascript
// n8n Code node
// Mode: Run Once for All Items

function asArray(v) {
  return Array.isArray(v) ? v : [];
}

function toNumber(v, fallback = null) {
  if (v === null || v === undefined || v === '') return fallback;
  const n = Number(v);
  return Number.isFinite(n) ? n : fallback;
}

function normalizeResult(raw) {
  let item = raw;

  // если LLM вернул JSON строкой
  if (typeof item === 'string') {
    item = JSON.parse(item);
  }

  // если ответ лежит в output/text/result
  if (item && typeof item.output === 'string') {
    item = JSON.parse(item.output);
  } else if (item && typeof item.text === 'string') {
    item = JSON.parse(item.text);
  } else if (item && typeof item.result === 'string') {
    item = JSON.parse(item.result);
  }

  return item;
}

function safeArray(v) {
  return Array.isArray(v) ? v : [];
}

function alignmentType(selfLevel, estimatedLevel) {
  if (selfLevel === null || estimatedLevel === null) return 'unknown';
  if (selfLevel > estimatedLevel) return 'overestimated';
  if (selfLevel < estimatedLevel) return 'underestimated';
  return 'aligned';
}

function priorityScore(result) {
  let score = 0;

  if (result.priority === 'high') score += 30;
  else if (result.priority === 'medium') score += 20;
  else score += 10;

  if (result.confidence === 'high') score += 15;
  else if (result.confidence === 'medium') score += 8;
  else score += 2;

  const selfLevel = toNumber(result.self_level, null);
  const estimatedLevel = toNumber(result.estimated_level, null);

  if (selfLevel !== null && estimatedLevel !== null) {
    score += Math.abs(selfLevel - estimatedLevel) * 5;
  }

  score += safeArray(result.risk_flags).length * 4;
  score += safeArray(result.evidence_missing).length * 2;
  score += safeArray(result.evidence_for).length * 1;

  return score;
}

const competencyResults = items.map(i => normalizeResult(i.json));

const validResults = competencyResults.filter(r =>
  r &&
  typeof r === 'object' &&
  r.competency_id
);

const overestimated = [];
const underestimated = [];
const aligned = [];

for (const r of validResults) {
  const selfLevel = toNumber(r.self_level, null);
  const estimatedLevel = toNumber(r.estimated_level, null);
  const type = alignmentType(selfLevel, estimatedLevel);

  if (type === 'overestimated') overestimated.push(r.competency_id);
  if (type === 'underestimated') underestimated.push(r.competency_id);
  if (type === 'aligned') aligned.push(r.competency_id);
}

const scored = validResults.map(r => ({
  competency_id: r.competency_id,
  score: priorityScore(r),
}));

scored.sort((a, b) => b.score - a.score);

const topPrioritiesNext6m = scored.slice(0, 6).map(x => x.competency_id);

const lowConfidenceCount = validResults.filter(r => r.confidence === 'low').length;
const highConfidenceCount = validResults.filter(r => r.confidence === 'high').length;

let summaryConfidence = 'medium';
if (validResults.length > 0) {
  if (lowConfidenceCount / validResults.length > 0.5) {
    summaryConfidence = 'low';
  } else if (highConfidenceCount / validResults.length >= 0.6) {
    summaryConfidence = 'high';
  }
}

const strengthsCandidates = validResults
  .filter(r =>
    (r.priority === 'high' || r.priority === 'medium') &&
    (r.confidence === 'high' || r.confidence === 'medium') &&
    safeArray(r.evidence_for).length > 0 &&
    toNumber(r.estimated_level, 0) >= toNumber(r.self_level, 0)
  )
  .slice(0, 8)
  .map(r => ({
    competency_id: r.competency_id,
    estimated_level: r.estimated_level,
    evidence_for: safeArray(r.evidence_for),
    assessment: r.assessment ?? '',
  }));

const growthAreasCandidates = validResults
  .filter(r =>
    safeArray(r.evidence_missing).length > 0 ||
    safeArray(r.risk_flags).length > 0 ||
    toNumber(r.estimated_level, 0) < toNumber(r.self_level, 0)
  )
  .slice(0, 8)
  .map(r => ({
    competency_id: r.competency_id,
    estimated_level: r.estimated_level,
    evidence_missing: safeArray(r.evidence_missing),
    risk_flags: safeArray(r.risk_flags),
    development_focus: r.development_focus ?? '',
  }));

return [
  {
    json: {
      employee_summary_payload: {
        competency_results: validResults,
        summary_inputs: {
          strengths_candidates: strengthsCandidates,
          growth_areas_candidates: growthAreasCandidates,
          alignment: {
            overestimated,
            underestimated,
            aligned,
          },
          top_priorities_next_6m_candidates: topPrioritiesNext6m,
          confidence_hint: summaryConfidence,
        }
      }
    }
  }
];
```

# Какой prompt лучше для summary LLM

## User prompt

```text
Верни строго JSON по схеме:
{
  "overall_summary": "string",
  "strengths": ["string"],
  "growth_areas": ["string"],
  "alignment": {
    "overestimated": ["string"],
    "underestimated": ["string"],
    "aligned": ["string"]
  },
  "top_priorities_next_6m": ["string"],
  "confidence": "low|medium|high"
}

Данные:
{{ JSON.stringify($json.employee_summary_payload) }}
```

## System prompt

```text
Ты анализируешь инженерный профиль сотрудника по результатам анализа компетенций.
Опирайся только на переданные данные.
Если evidence слабое или неполное, отмечай это явно.
Не путай отсутствие данных с отсутствием компетенции.
В alignment и top_priorities_next_6m возвращай competency_id.
Верни строго JSON без markdown и без пояснений.
```

# И сразу следующая нода: финальная сборка WF30

Если хочешь не дробить, вот еще и **финальная Code node**.

Она ожидает, что после Merge к ней попадут:

* несколько items с competency_results
* один item с employee_summary

## `Build WF30 Output`

Режим:

* **Run Once for All Items**

```javascript
// n8n Code node
// Mode: Run Once for All Items

function parseMaybeJson(v) {
  if (typeof v === 'string') {
    try {
      return JSON.parse(v);
    } catch {
      return v;
    }
  }
  return v;
}

function normalize(raw) {
  let item = raw;

  if (typeof item === 'string') {
    item = parseMaybeJson(item);
  }

  if (item && typeof item.output === 'string') {
    item = parseMaybeJson(item.output);
  } else if (item && typeof item.text === 'string') {
    item = parseMaybeJson(item.text);
  } else if (item && typeof item.result === 'string') {
    item = parseMaybeJson(item.result);
  }

  return item;
}

const normalized = items.map(i => normalize(i.json));

const competencyResults = [];
let employeeSummary = null;

for (const item of normalized) {
  if (
    item &&
    typeof item === 'object' &&
    item.competency_id &&
    Object.prototype.hasOwnProperty.call(item, 'estimated_level')
  ) {
    competencyResults.push(item);
    continue;
  }

  if (
    item &&
    typeof item === 'object' &&
    Object.prototype.hasOwnProperty.call(item, 'overall_summary') &&
    Object.prototype.hasOwnProperty.call(item, 'strengths')
  ) {
    employeeSummary = item;
  }
}

const lowConfidenceCount = competencyResults.filter(r => r.confidence === 'low').length;
const highConfidenceCount = competencyResults.filter(r => r.confidence === 'high').length;

let analysisQuality = 'medium';
if (competencyResults.length > 0) {
  if (lowConfidenceCount / competencyResults.length > 0.5) {
    analysisQuality = 'low';
  } else if (highConfidenceCount / competencyResults.length >= 0.6) {
    analysisQuality = 'high';
  }
}

const flags = [];
if (!employeeSummary) flags.push('employee_summary_missing');
if (competencyResults.length === 0) flags.push('competency_results_missing');
if (lowConfidenceCount > 0) flags.push('some_competencies_low_confidence');

const topPriorityCompetencies =
  employeeSummary?.top_priorities_next_6m?.length
    ? employeeSummary.top_priorities_next_6m
    : competencyResults
        .filter(r => r.priority === 'high')
        .slice(0, 6)
        .map(r => r.competency_id);

return [
  {
    json: {
      competency_analysis: {
        competency_results: competencyResults,
        employee_summary: employeeSummary || {
          overall_summary: '',
          strengths: [],
          growth_areas: [],
          alignment: {
            overestimated: [],
            underestimated: [],
            aligned: []
          },
          top_priorities_next_6m: [],
          confidence: 'low'
        },
        top_priority_competencies: topPriorityCompetencies,
        analysis_quality: analysisQuality,
        flags
      }
    }
  }
];
```

# Полная цепочка получается такая

1. `Build Competency Payloads`
2. `Pick Top Competencies`
3. `LLM Analyze One Competency`
4. `Normalize Competency Result`
5. `Build Employee Summary Payload`
6. `LLM Build Employee Summary`
7. `Build WF30 Output`
