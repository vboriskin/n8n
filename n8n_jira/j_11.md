# 🔻 ДОБАВЛЯЕМ НОВЫЙ БЛОК: FOUNDATION (после Normalize Jira Evidence)

## 13.1 `Code / Build Evidence Trace`

```javascript
const issues = $json.issues || [];

const trace = [];

for (const issue of issues) {
  const key = issue.issue_key || issue.key;

  if (!key) continue;

  // статус
  if (issue.fields?.status?.name) {
    trace.push({
      issue_key: key,
      signal_type: "status",
      value: issue.fields.status.name,
      weight: 0.5
    });
  }

  // worklog
  const worklogs = issue.fields?.worklog?.worklogs || [];
  for (const wl of worklogs) {
    trace.push({
      issue_key: key,
      signal_type: "worklog",
      value: wl.timeSpent || null,
      weight: 0.3
    });
  }

  // comments
  const comments = issue.fields?.comment?.comments || [];
  for (const c of comments) {
    trace.push({
      issue_key: key,
      signal_type: "comment",
      value: c.body?.slice(0, 200) || null,
      weight: 0.4
    });
  }
}

return [{ json: { evidence_trace: trace } }];
```

## 13.2 `Code / Build Coverage Map`

```javascript
const trace = $json.evidence_trace || [];

const map = {};

for (const t of trace) {
  const key = t.signal_type;

  if (!map[key]) {
    map[key] = {
      count: 0
    };
  }

  map[key].count++;
}

const coverage = Object.entries(map).map(([k, v]) => ({
  signal_type: k,
  coverage: v.count > 10 ? "high" : v.count > 3 ? "medium" : "low",
  count: v.count
}));

return [{ json: { coverage_map: coverage } }];
```

## 13.3 `Code / Compute Confidence Signals`

```javascript
const issues = $json.issues || [];
const coverage = $json.coverage_map || [];

const signals = {
  issue_count: issues.length,
  has_dev_signal: coverage.some(c => c.signal_type === "comment"),
  has_worklog: coverage.some(c => c.signal_type === "worklog")
};

return [{ json: { confidence_signals: signals } }];
```
