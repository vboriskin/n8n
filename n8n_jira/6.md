Вот чистая **первая Code node для n8n** под тот контракт, который мы зафиксировали.

Она делает ровно это:

* берет `employee`
* берет `competency_catalog.competencies`
* берет `assessment_input.self_assessments`
* берет `evidence_input.jira_competency_evidence`
* джойнит всё по `competency_id`
* считает `priority_score`
* выставляет `priority`
* собирает `payload` для LLM по каждой компетенции

Работает без эвристик и без парсинга “по словам”. Это соответствует тому, что у тебя матрица уже может быть нормализована в явный каталог компетенций, вместо использования сырой вложенной структуры исходного JSON

# n8n Code node: `Build Competency Payloads`

Режим:

* **Run Once for All Items**

```javascript
// n8n Code node
// Mode: Run Once for All Items

function asArray(v) {
  return Array.isArray(v) ? v : [];
}

function toNumber(v, fallback = null) {
  if (v === null || v === undefined || v === '') return fallback;
  const n = Number(v);
  return Number.isFinite(n) ? n : fallback;
}

function clamp(num, min, max) {
  return Math.max(min, Math.min(max, num));
}

function priorityFromScore(score) {
  if (score >= 60) return 'high';
  if (score >= 35) return 'medium';
  return 'low';
}

function getTargetLevel(competency, selfAssessment, employee) {
  // 1. Явный target_level в self-assessment
  if (selfAssessment && selfAssessment.target_level !== undefined && selfAssessment.target_level !== null) {
    return toNumber(selfAssessment.target_level, null);
  }

  // 2. default_target_by_grade в competency
  const gradeId = employee?.target_grade_id || employee?.grade_id;
  const fromGradeMap = competency?.default_target_by_grade?.[gradeId];
  if (fromGradeMap !== undefined && fromGradeMap !== null) {
    return toNumber(fromGradeMap, null);
  }

  // 3. fallback_target_level
  if (competency?.targeting_rule?.fallback_target_level !== undefined) {
    return toNumber(competency.targeting_rule.fallback_target_level, null);
  }

  return null;
}

function summarizeEvidence(evidenceList) {
  return evidenceList.map(e => ({
    evidence_id: e.evidence_id ?? null,
    source: e.source ?? null,
    issue_key: e.issue_key ?? null,
    title: e.title ?? null,
    summary: e.summary ?? null,
    relevance: toNumber(e.relevance, null),
    evidence_strength: e.evidence_strength ?? null,
    status: e.status ?? null,
    date: e.date ?? null,
  }));
}

const input = items[0].json;

const employee = input.employee ?? {};
const workflowConfig = input.workflow_config ?? {};
const competencyCatalog = input.competency_catalog ?? {};
const assessmentInput = input.assessment_input ?? {};
const evidenceInput = input.evidence_input ?? {};
const scoringInput = input.scoring_input ?? {};

const competencies = asArray(competencyCatalog.competencies);
const selfAssessments = asArray(assessmentInput.self_assessments);
const jiraEvidence = asArray(evidenceInput.jira_competency_evidence);

const priorityRules = {
  self_assessment_present_bonus:
    toNumber(scoringInput?.priority_rules?.self_assessment_present_bonus, 20),
  comment_present_bonus:
    toNumber(scoringInput?.priority_rules?.comment_present_bonus, 10),
  jira_evidence_present_bonus:
    toNumber(scoringInput?.priority_rules?.jira_evidence_present_bonus, 18),
  gap_to_target_bonus:
    toNumber(scoringInput?.priority_rules?.gap_to_target_bonus, 16),
  hard_skill_bonus:
    toNumber(scoringInput?.priority_rules?.hard_skill_bonus, 8),
  soft_skill_penalty:
    toNumber(scoringInput?.priority_rules?.soft_skill_penalty, -6),
};

const defaultPriorityWeight = toNumber(workflowConfig.default_priority_weight, 2);
const includeWithoutSelfAssessment =
  workflowConfig.include_competencies_without_self_assessment !== false;
const includeWithoutEvidence =
  workflowConfig.include_competencies_without_evidence !== false;

// ---- Индекс самооценки по competency_id
const selfMap = new Map();
for (const sa of selfAssessments) {
  if (!sa?.competency_id) continue;
  selfMap.set(sa.competency_id, sa);
}

// ---- Индекс evidence по competency_id
const evidenceMap = new Map();
for (const ev of jiraEvidence) {
  if (!ev?.competency_id) continue;
  const arr = evidenceMap.get(ev.competency_id) ?? [];
  arr.push(ev);
  evidenceMap.set(ev.competency_id, arr);
}

const results = [];

for (const competency of competencies) {
  const competencyId = competency?.competency_id;
  if (!competencyId) continue;

  const selfAssessment = selfMap.get(competencyId) ?? null;
  const evidenceList = evidenceMap.get(competencyId) ?? [];

  if (!includeWithoutSelfAssessment && !selfAssessment) {
    continue;
  }

  if (!includeWithoutEvidence && evidenceList.length === 0) {
    continue;
  }

  const selfLevel = toNumber(selfAssessment?.self_level, null);
  const targetLevelRaw = getTargetLevel(competency, selfAssessment, employee);
  const levels = asArray(competency.levels);
  const maxLevel =
    levels.length > 0
      ? Math.max(...levels.map(l => toNumber(l.level, 0)))
      : 4;

  const targetLevel =
    targetLevelRaw !== null ? clamp(targetLevelRaw, 0, maxLevel) : null;

  const gap =
    selfLevel !== null && targetLevel !== null
      ? Math.max(targetLevel - selfLevel, 0)
      : null;

  const priorityWeight = toNumber(
    competency.priority_weight,
    defaultPriorityWeight
  );

  const hasSelfAssessment = !!selfAssessment;
  const hasComment = !!(selfAssessment?.comment && String(selfAssessment.comment).trim());
  const hasJiraEvidence = evidenceList.length > 0;
  const skillType = competency.skill_type ?? null;

  let priorityScore = 0;

  // База: важность компетенции из каталога
  priorityScore += priorityWeight * 10;

  // Есть самооценка
  if (hasSelfAssessment) {
    priorityScore += priorityRules.self_assessment_present_bonus;
  }

  // Есть комментарий
  if (hasComment) {
    priorityScore += priorityRules.comment_present_bonus;
  }

  // Есть Jira evidence
  if (hasJiraEvidence) {
    priorityScore += priorityRules.jira_evidence_present_bonus;
  }

  // Есть gap до target
  if (gap !== null && gap > 0) {
    priorityScore += priorityRules.gap_to_target_bonus;
  }

  // Hard / Soft
  if (skillType === 'hard') {
    priorityScore += priorityRules.hard_skill_bonus;
  } else if (skillType === 'soft') {
    priorityScore += priorityRules.soft_skill_penalty;
  }

  // Дополнительный легкий буст по criticality
  if (competency.criticality === 'high') {
    priorityScore += 5;
  } else if (competency.criticality === 'medium') {
    priorityScore += 2;
  }

  // Можно слегка учесть силу evidence
  const strongEvidenceCount = evidenceList.filter(e => e.evidence_strength === 'strong').length;
  const mediumEvidenceCount = evidenceList.filter(e => e.evidence_strength === 'medium').length;

  priorityScore += strongEvidenceCount * 2;
  priorityScore += mediumEvidenceCount * 1;

  const priority = priorityFromScore(priorityScore);

  const missingDataFlags = [];
  if (!hasSelfAssessment) missingDataFlags.push('no_self_assessment');
  if (!hasComment) missingDataFlags.push('no_self_comment');
  if (!hasJiraEvidence) missingDataFlags.push('no_jira_evidence');
  if (targetLevel === null) missingDataFlags.push('no_target_level');

  results.push({
    competency_id: competencyId,
    competency: {
      competency_id: competencyId,
      name: competency.name ?? null,
      group_id: competency.group_id ?? null,
      group_name: competency.group_name ?? null,
      skill_type: competency.skill_type ?? null,
      priority_weight: priorityWeight,
      criticality: competency.criticality ?? null,
      is_required_for_role: competency.is_required_for_role ?? null,
      levels: levels,
    },
    self_assessment_item: selfAssessment
      ? {
          competency_id: selfAssessment.competency_id,
          self_level: selfLevel,
          target_level: targetLevel,
          comment: selfAssessment.comment ?? null,
          confidence: selfAssessment.confidence ?? null,
          updated_at: selfAssessment.updated_at ?? null,
        }
      : null,
    jira_evidence: summarizeEvidence(evidenceList),
    self_level: selfLevel,
    target_level: targetLevel,
    gap,
    priority_score: priorityScore,
    priority,
    missing_data_flags: missingDataFlags,

    // payload прямо для LLM-компетенции
    payload: {
      employee: {
        employee_id: employee.employee_id ?? null,
        full_name: employee.full_name ?? null,
        role_id: employee.role_id ?? null,
        grade_id: employee.grade_id ?? null,
        target_grade_id: employee.target_grade_id ?? null,
        team_id: employee.team_id ?? null,
      },
      competency: {
        competency_id: competencyId,
        name: competency.name ?? null,
        group_name: competency.group_name ?? null,
        skill_type: competency.skill_type ?? null,
        priority_weight: priorityWeight,
        criticality: competency.criticality ?? null,
        levels: levels,
      },
      self_assessment: selfAssessment
        ? {
            self_level: selfLevel,
            target_level: targetLevel,
            comment: selfAssessment.comment ?? null,
            confidence: selfAssessment.confidence ?? null,
            updated_at: selfAssessment.updated_at ?? null,
          }
        : null,
      jira_evidence: summarizeEvidence(evidenceList),
      scoring_context: {
        priority_score: priorityScore,
        priority,
        gap,
        missing_data_flags: missingDataFlags,
      }
    }
  });
}

// сортировка по приоритету
results.sort((a, b) => b.priority_score - a.priority_score);

return results.map(r => ({ json: r }));
```

# Что делать следующей нодой

Сразу после этой ноды обычно ставят:

## `Pick Top Competencies`

Например top 12:

```javascript
// n8n Code node
// Mode: Run Once for All Items

const TOP_N = 12;

return items
  .sort((a, b) => (b.json.priority_score || 0) - (a.json.priority_score || 0))
  .slice(0, TOP_N);
```

---

# Почему этот код лучше прошлого варианта

Потому что тут уже нет:

* поиска soft/hard по маркерам
* генерации `competency_id` из текста
* угадывания target level “на глаз”

Все берется из входного контракта:

* `competency.skill_type`
* `competency.priority_weight`
* `self_assessment.target_level`
* `competency.default_target_by_grade`
