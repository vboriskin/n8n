# 1) HTTP Request: Jira Search Issues

## Настройки

* Method: `POST`
* URL:

```text
https://jira.delta.sbrf.ru/rest/api/2/search
```

* Authentication: как у тебя работает токен
* Headers:

  * `Content-Type: application/json`
  * `Authorization: Bearer {{$json.jira_token}}`

Если у вас не Bearer, а Basic — просто поменяешь это место.

## Body

```javascript
={{
{
  jql: `assignee = "${$json.employee.employee_id}" AND updated >= -180d ORDER BY updated DESC`,
  startAt: 0,

  fields: [
    "summary",
    "description",
    "labels",
    "components",
    "issuetype",
    "status",
    "resolution",
    "assignee",
    "created",
    "updated",
    "comment",
    "worklog",
    "timespent",
    "timeoriginalestimate",
    "aggregatetimespent",

  expand: ["changelog"]
```

# 4) Для developer: HTTP Request / Jira Dev Status

У Jira Server/DC с Bitbucket integration часто работает endpoint вида:

```text
https://jira.delta.sbrf.ru/rest/dev-status/latest/issue/detail
```

Обычно туда передают:

* `issueId`
* `applicationType=stash`
* `dataType=repository`

или отдельно по типам.

На твоих скринах видно Bitbucket Delta и development panel:

* `1 branch`
* `3 commits`
* `2 pull requests`

Так что пробовать надо именно этот путь.

## Вариант запроса

* Method: `GET`
* URL:

```text
=https://jira.delta.sbrf.ru/rest/dev-status/latest/issue/detail?issueId={{$json.issue_id}}&applicationType=stash&dataType=repository
```

### Если не сработает — пробовать:

```text
=https://jira.delta.sbrf.ru/rest/dev-status/latest/issue/summary?issueId={{$json.issue_id}}
```

и/или

```text
=https://jira.delta.sbrf.ru/rest/dev-status/latest/issue/detail?issueId={{$json.issue_id}}&applicationType=bitbucket&dataType=repository
```

У вас, скорее всего, `stash`, потому что Jira Server + Bitbucket Server/Delta.

# Главная идея scoring evidence

## Для всех ролей

`worklog` повышает confidence, но не должен автоматически означать высокий skill.

Пример:

* есть worklog 2 дня
* но нет текстового evidence по содержанию
  => это **medium**, не `high`

## Для developer

`commits + merged PR` — это уже сильнее:

* если есть ветка
* есть коммиты
* есть merged PR
  => evidence становится заметно сильнее

## Код

```javascript
// n8n Code node
// Run Once for All Items

function asArray(v) {
  return Array.isArray(v) ? v : [];
}

function text(v) {
  return String(v ?? '').trim();
}

function lower(v) {
  return text(v).toLowerCase();
}

function uniq(arr) {
  return [...new Set(arr.filter(Boolean))];
}

function cut(str, max = 700) {
  const s = text(str);
  return s.length > max ? s.slice(0, max) + '...' : s;
}

function sum(arr) {
  return arr.reduce((acc, x) => acc + (Number(x) || 0), 0);
}

// ---- fallback mapping ----
const keywordToCompetency = {
  api: 'api_testing',
  rest: 'api_testing',
  swagger: 'api_testing',
  graphql: 'api_testing',
  websocket: 'api_testing',

  sql: 'databases',
  database: 'databases',
  postgres: 'databases',
  oracle: 'databases',

  requirement: 'work_with_requirements',
  acceptance: 'work_with_requirements',
  traceability: 'work_with_requirements',

  defect: 'defect_management',
  regression: 'defect_management',
  bug: 'defect_management',

  communication: 'communication_with_stakeholders',
  коммуникация: 'communication_with_stakeholders',
  согласование: 'communication_with_stakeholders',

  java: 'programming_languages',
  kotlin: 'programming_languages',
  python: 'programming_languages',
  refactor: 'clean_code',
  refactoring: 'clean_code',
  oop: 'oop_principles',
  git: 'version_control',
  branch: 'version_control',
  pull request: 'version_control',
  commit: 'version_control'
};

const input = items[0].json;

const employee = input.employee || {};
const issues = asArray(input.issues);
const devStatusByIssue = input.dev_status_by_issue || {};
const validCompetencyIds = new Set(
  asArray(input.competency_catalog?.competencies).map(c => c.competency_id)
);

function extractCommentText(commentField) {
  const comments = asArray(commentField?.comments);
  return comments
    .slice(0, 3)
    .map(c => text(c.body))
    .filter(Boolean)
    .join('\n');
}

function normalizeWorklog(worklogField) {
  const entries = asArray(worklogField?.worklogs);

  const ownEntries = entries.filter(w => {
    const authorName = lower(w?.author?.displayName);
    const authorKey = lower(w?.author?.name);
    const empName = lower(employee.full_name);
    const empId = lower(employee.employee_id);
    return (
      (empName && authorName.includes(empName)) ||
      (empId && authorKey.includes(empId))
    );
  });

  const targetEntries = ownEntries.length ? ownEntries : entries;

  const secondsTotal = sum(targetEntries.map(w => w.timeSpentSeconds || 0));
  const entriesCount = targetEntries.length;
  const hasComments = targetEntries.some(w => text(w.comment));
  const authors = uniq(targetEntries.map(w => w?.author?.displayName).filter(Boolean));

  return {
    entries_count: entriesCount,
    seconds_total: secondsTotal,
    days_total: Number((secondsTotal / 28800).toFixed(2)), // 8ч = 1 day условно
    has_comments: hasComments,
    authors,
    author_match: ownEntries.length > 0
  };
}

function normalizeDevStatus(raw) {
  if (!raw) {
    return {
      branches: 0,
      commits: 0,
      pull_requests: 0,
      merged_pull_requests: 0,
      commit_messages: [],
      pull_request_titles: []
    };
  }

  const detail = asArray(raw.detail);
  const summary = raw.summary || {};

  let branches = 0;
  let commits = 0;
  let pullRequests = 0;
  let mergedPullRequests = 0;
  const commitMessages = [];
  const prTitles = [];

  for (const d of detail) {
    const repos = asArray(d.repositories);
    for (const repo of repos) {
      branches += asArray(repo.branches).length;

      const repoCommits = asArray(repo.commits);
      commits += repoCommits.length;
      for (const c of repoCommits) {
        if (c?.message) commitMessages.push(c.message);
      }

      const prs = asArray(repo.pullRequests);
      pullRequests += prs.length;
      for (const pr of prs) {
        if (pr?.title) prTitles.push(pr.title);
        const status = lower(pr?.status || pr?.state);
        if (status.includes('merged')) mergedPullRequests += 1;
      }
    }
  }

  // fallback на summary, если detail пустой
  if (!branches && summary.branches) branches = Number(summary.branches) || 0;
  if (!commits && summary.commits) commits = Number(summary.commits) || 0;
  if (!pullRequests && summary.pullRequests) pullRequests = Number(summary.pullRequests) || 0;

  return {
    branches,
    commits,
    pull_requests: pullRequests,
    merged_pull_requests: mergedPullRequests,
    commit_messages: commitMessages.slice(0, 5),
    pull_request_titles: prTitles.slice(0, 5)
  };
}

function detectCompetencyIds(issue, devInfo) {
  const fields = issue.fields || {};

  const haystack = lower([
    fields.summary,
    fields.description,
    extractCommentText(fields.comment),
    ...asArray(fields.labels),
    ...asArray(fields.components).map(c => c?.name),
    ...asArray(devInfo.commit_messages),
    ...asArray(devInfo.pull_request_titles)
  ].join(' '));

  const found = [];
  for (const [keyword, competencyId] of Object.entries(keywordToCompetency)) {
    if (haystack.includes(keyword)) found.push(competencyId);
  }

  let ids = uniq(found);

  if (validCompetencyIds.size > 0) {
    ids = ids.filter(id => validCompetencyIds.has(id));
  }

  return ids;
}

function calcEvidenceStrength(issue, worklogInfo, devInfo) {
  let score = 0;
  const fields = issue.fields || {};

  const status = lower(fields.status?.name);
  if (status.includes('done') || status.includes('closed') || status.includes('resolved')) {
    score += 2;
  }

  if (text(fields.summary)) score += 1;
  if (text(fields.description)) score += 1;

  if (worklogInfo.entries_count > 0) score += 1;
  if (worklogInfo.seconds_total >= 14400) score += 1; // от 4ч
  if (worklogInfo.author_match) score += 1;

  if (devInfo.commits > 0) score += 1;
  if (devInfo.pull_requests > 0) score += 1;
  if (devInfo.merged_pull_requests > 0) score += 1;

  if (score >= 7) return 'strong';
  if (score >= 4) return 'medium';
  return 'weak';
}

function calcRelevance(issue, competencyId, devInfo) {
  const fields = issue.fields || {};
  const haystack = lower([
    fields.summary,
    fields.description,
    extractCommentText(fields.comment),
    ...asArray(fields.labels),
    ...asArray(fields.components).map(c => c?.name),
    ...asArray(devInfo.commit_messages),
    ...asArray(devInfo.pull_request_titles)
  ].join(' '));

  if (haystack.includes(lower(competencyId))) return 0.95;

  const keywords = Object.entries(keywordToCompetency)
    .filter(([, compId]) => compId === competencyId)
    .map(([k]) => k);

  const hits = keywords.filter(k => haystack.includes(k)).length;

  if (hits >= 3) return 0.92;
  if (hits >= 2) return 0.85;
  if (hits === 1) return 0.72;

  // developer signals
  if (devInfo.commits > 0 || devInfo.pull_requests > 0) return 0.65;

  return 0.55;
}

const jira_competency_evidence = [];

for (const issue of issues) {
  const fields = issue.fields || {};
  const issueKey = issue.key;
  const devInfo = normalizeDevStatus(devStatusByIssue[issueKey]);
  const worklogInfo = normalizeWorklog(fields.worklog);

  const competencyIds = detectCompetencyIds(issue, devInfo);
  if (!competencyIds.length) continue;

  const summary = cut([
    text(fields.summary),
    text(fields.description),
    extractCommentText(fields.comment),
    worklogInfo.entries_count
      ? `worklog: ${worklogInfo.entries_count} entries, ${worklogInfo.days_total}d`
      : '',
    devInfo.commits
      ? `development: ${devInfo.branches} branch, ${devInfo.commits} commits, ${devInfo.pull_requests} PR, merged ${devInfo.merged_pull_requests}`
      : ''
  ].filter(Boolean).join(' | '), 900);

  for (const competencyId of competencyIds) {
    jira_competency_evidence.push({
      competency_id: competencyId,
      evidence_id: `JIRA-${issueKey}`,
      source: 'jira',
      issue_key: issueKey,
      issue_url: `https://jira.delta.sbrf.ru/browse/${issueKey}`,
      title: text(fields.summary) || null,
      summary,
      relevance: calcRelevance(issue, competencyId, devInfo),
      evidence_strength: calcEvidenceStrength(issue, worklogInfo, devInfo),
      status: fields.status?.name || null,
      issue_type: fields.issuetype?.name || null,
      resolution: fields.resolution?.name || null,
      created: fields.created || null,
      updated: fields.updated || null,

      worklog: worklogInfo,

      timetracking: {
        timespent_seconds: fields.timespent || 0,
        original_estimate_seconds: fields.timeoriginalestimate || 0,
        aggregate_timespent_seconds: fields.aggregatetimespent || 0,
        aggregate_original_estimate_seconds: fields.aggregatetimeoriginalestimate || 0
      },

      development: devInfo
    });
  }
}

return [
  {
    json: {
      employee: input.employee,
      workflow_config: input.workflow_config,
      competency_catalog: input.competency_catalog,
      assessment_input: input.assessment_input,
      scoring_input: input.scoring_input,
      evidence_input: {
        jira_competency_evidence
      }
    }
  }
];
```

```javascript
// Run Once for All Items

const map = {};

for (const item of items) {
  const issueKey = item.json.issue_key || item.json.base_issue?.key;

  map[issueKey] = {

return [

    json: {
```
