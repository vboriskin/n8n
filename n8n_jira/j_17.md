from __future__ import annotations

from datetime import datetime
from typing import Any, Literal
from uuid import UUID

from pydantic import BaseModel, ConfigDict, EmailStr, Field


ConfidenceLevel = Literal["low", "medium", "high"]
PriorityLevel = Literal["low", "medium", "high"]
ActivityLevel = Literal["low", "medium", "high"]
EffortLevel = Literal["low", "medium", "high"]
ReadinessLevel = Literal["not_ready", "emerging", "ready"]
JobStatus = Literal["queued", "running", "completed", "failed"]
CurrentGradeFit = Literal["low", "partial", "strong"]
FocusMode = Literal["deepening", "broadening", "stabilizing"]
LearningResourceType = Literal["book", "course", "article", "internal_training"]
IdpActionType = Literal["task", "learning", "mentoring", "project", "reading"]


class ErrorResponseDto(BaseModel):
    error: str
    details: list[str] = Field(default_factory=list)


class RunAssessmentFeaturesDto(BaseModel):
    development: bool = True
    one2one: bool = True
    promotion: bool = True


class RunAssessmentRequestDto(BaseModel):
    employee_id: str
    lookback_days: int = 180
    force_refresh: bool = False
    features: RunAssessmentFeaturesDto = Field(default_factory=RunAssessmentFeaturesDto)


class RunAssessmentResponseDto(BaseModel):
    job_id: UUID
    status: JobStatus


class AssessmentJobDto(BaseModel):
    job_id: UUID
    employee_id: str
    status: JobStatus
    snapshot_id: UUID | None = None
    error_message: str | None = None
    created_at: datetime
    started_at: datetime | None = None
    finished_at: datetime | None = None

    employee_id: str
    full_name: str
    email: EmailStr
    role_id: str
    grade_id: str
    team_id: str | None = None

# Что я бы советовал сразу поправить при внедрении

Самое практичное:

1. Вынести `AssessmentSnapshot` в отдельный файл и считать его главным контрактом.
2. Для `inputs/raw_data/normalized` в FastAPI пока оставить `dict[str, Any]`, а не пытаться жестко типизировать всё сразу.
3. В NestJS не делать DTO “на всё подряд” для внутренних сервисов — только для внешнего API.
4. Для LLM-выходов валидировать payload отдельным слоем маппинга, а не пускать в DTO напрямую.

Следующим сообщением я могу добить это до совсем прикладного состояния и собрать тебе **готовую структуру папок проекта для FastAPI или NestJS + repository/service/controller слои + naming conventions**.
