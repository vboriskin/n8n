Вот fallback через `changelog/history`, если `dev-status` недоступен.

Идея простая: из истории задачи вытаскиваем признаки:

* создана ветка
* создан PR
* PR merged
* переходы в `Code Review` / `In Review` / `Ready for Test`
* коммиты по текстовым следам, если они есть

Это слабее, чем `dev-status`, но на Jira Server часто спасает.

---

# Что добавить в запрос Jira

В `HTTP Request / Jira Issue Details` уже должен быть:

* `expand=changelog`

Иначе истории не будет.

# Code node: `Extract Dev Signals From Changelog`

Режим:

* `Run Once for All Items`

На входе: items из `Jira Issue Details`

```javascript
function asArray(v) {
  return Array.isArray(v) ? v : [];
}

function text(v) {
  return String(v ?? '').trim();
}

function lower(v) {
  return text(v).toLowerCase();
}

function uniq(arr) {
  return [...new Set(arr.filter(Boolean))];
}

function safeMatchAll(str, regex) {
  try {
    return [...str.matchAll(regex)];
  } catch {
    return [];
  }
}

function extractStringsFromHistory(history) {
  const parts = [];

  parts.push(text(history.author?.displayName));
  parts.push(text(history.author?.name));
  parts.push(text(history.created));

  for (const item of asArray(history.items)) {
    parts.push(text(item.field));
    parts.push(text(item.fieldtype));
    parts.push(text(item.fromString));
    parts.push(text(item.toString));
  }

  // иногда интеграции Jira/Bitbucket кладут текст в historyMetadata / description / activityDescription
  parts.push(text(history.historyMetadata?.description));
  parts.push(text(history.historyMetadata?.activityDescription));
  parts.push(text(history.description));
  parts.push(text(history.activityDescription));

  return parts.filter(Boolean);
}

function countMatches(haystack, patterns) {
  let count = 0;
  for (const p of patterns) {
    if (p.test(haystack)) count++;
  }
  return count;
}

const branchPatterns = [
  /\bcreated a branch\b/i,
  /\bbranch created\b/i,
  /\bсоздал(?:а)? ветк/i,
  /\bветк[ау]\b/i,
  /\bbitbucket.*branch\b/i,
  /\bstash.*branch\b/i
];

const prCreatePatterns = [
  /\bcreated pull request\b/i,
  /\bopened pull request\b/i,
  /\bpull request created\b/i,
  /\bсоздал(?:а)? pull request\b/i,
  /\bсоздал(?:а)? pr\b/i,
  /\bmerge request created\b/i
];

const prMergePatterns = [
  /\bmerged pull request\b/i,
  /\bpull request merged\b/i,
  /\bmerge completed\b/i,
  /\bсмёржен/i,
  /\bmerged pr\b/i
];

const commitPatterns = [
  /\bcommit\b/i,
  /\bcommits\b/i,
  /\bзафиксировал/i,
  /\bкоммит/i
];

const reviewStatusNames = [
  'code review',
  'in review',
  'review',
  'ready for review',
  'peer review',
  'на ревью',
  'ревью',
  'code_review'
];

const readyForTestStatuses = [
  'ready for test',
  'testing',
  'qa',
  'ready for qa',
  'готово к тестированию',
  'на тестировании'
];

const results = [];

for (const item of items) {
  const issue = item.json;
  const key = issue.key;
  const changelog = asArray(issue.changelog?.histories);

  let branches = 0;
  let pullRequests = 0;
  let mergedPullRequests = 0;
  let commitSignals = 0;
  let codeReviewTransitions = 0;
  let readyForTestTransitions = 0;

  const branchEvidence = [];
  const prEvidence = [];
  const mergedPrEvidence = [];
  const commitEvidence = [];
  const reviewEvidence = [];

  for (const history of changelog) {
    const rawText = extractStringsFromHistory(history).join(' | ');
    const haystack = lower(rawText);

    // branch events
    if (countMatches(rawText, branchPatterns) > 0) {
      branches += 1;
      branchEvidence.push({
        created: history.created ?? null,
        author: history.author?.displayName ?? history.author?.name ?? null,
        snippet: rawText.slice(0, 300)
      });
    }

    // PR created
    if (countMatches(rawText, prCreatePatterns) > 0) {
      pullRequests += 1;
      prEvidence.push({
        created: history.created ?? null,
        author: history.author?.displayName ?? history.author?.name ?? null,
        snippet: rawText.slice(0, 300)
      });
    }

    // PR merged
    if (countMatches(rawText, prMergePatterns) > 0) {
      mergedPullRequests += 1;
      mergedPrEvidence.push({
        created: history.created ?? null,
        author: history.author?.displayName ?? history.author?.name ?? null,
        snippet: rawText.slice(0, 300)
      });
    }

    // commit signals
    if (countMatches(rawText, commitPatterns) > 0) {
      commitSignals += 1;
      commitEvidence.push({
        created: history.created ?? null,
        author: history.author?.displayName ?? history.author?.name ?? null,
        snippet: rawText.slice(0, 300)
      });
    }

    // workflow transitions
    for (const chItem of asArray(history.items)) {
      if (lower(chItem.field) !== 'status') continue;

      const toStatus = lower(chItem.toString);
      const fromStatus = lower(chItem.fromString);

      if (reviewStatusNames.some(s => toStatus.includes(s))) {
        codeReviewTransitions += 1;
        reviewEvidence.push({
          type: 'code_review_transition',
          created: history.created ?? null,
          from: chItem.fromString ?? null,
          to: chItem.toString ?? null
        });
      }

      if (readyForTestStatuses.some(s => toStatus.includes(s))) {
        readyForTestTransitions += 1;
        reviewEvidence.push({
          type: 'ready_for_test_transition',
          created: history.created ?? null,
          from: chItem.fromString ?? null,
          to: chItem.toString ?? null
        });
      }

      // иногда уже полезно просто видеть движение dev-пайплайна
      if (
        reviewStatusNames.some(s => fromStatus.includes(s)) &&
        readyForTestStatuses.some(s => toStatus.includes(s))
      ) {
        reviewEvidence.push({
          type: 'review_to_test_flow',
          created: history.created ?? null,
          from: chItem.fromString ?? null,
          to: chItem.toString ?? null
        });
      }
    }
  }

  results.push({
    json: {
      issue_key: key,
      issue_id: issue.id ?? null,
      changelog_dev_signals: {
        branches,
        pull_requests: pullRequests,
        merged_pull_requests: mergedPullRequests,
        commit_signals: commitSignals,
        code_review_transitions: codeReviewTransitions,
        ready_for_test_transitions: readyForTestTransitions,
        branch_evidence: branchEvidence.slice(0, 5),
        pr_evidence: prEvidence.slice(0, 5),
        merged_pr_evidence: mergedPrEvidence.slice(0, 5),
        commit_evidence: commitEvidence.slice(0, 5),
        review_evidence: reviewEvidence.slice(0, 8)
      }
    }
  });
}

return results;
```

Если `dev-status` работает:

* используй его как основной источник
* changelog extractor можно оставить как fallback

Если `dev-status` не работает:

# Как подмешать это в normalizer

В `Normalize Jira Evidence` добавь чтение:

```javascript
const changelogDevByIssue = input.changelog_dev_by_issue || {};
```

И замени `normalizeDevStatus(...)` на объединение двух источников.

Вот функция:

```javascript
function mergeDevSignals(devStatusInfo, changelogInfo) {
  const a = devStatusInfo || {
    branches: 0,
    commits: 0,
    pull_requests: 0,
    merged_pull_requests: 0,
    commit_messages: [],
    pull_request_titles: []
  };

  const b = changelogInfo || {
    branches: 0,
    pull_requests: 0,
    merged_pull_requests: 0,
    commit_signals: 0,
    code_review_transitions: 0,
    ready_for_test_transitions: 0,
    branch_evidence: [],
    pr_evidence: [],
    merged_pr_evidence: [],
    commit_evidence: [],
    review_evidence: []
  };

  return {
    branches: Math.max(a.branches || 0, b.branches || 0),
    commits: Math.max(a.commits || 0, b.commit_signals || 0),
    pull_requests: Math.max(a.pull_requests || 0, b.pull_requests || 0),
    merged_pull_requests: Math.max(a.merged_pull_requests || 0, b.merged_pull_requests || 0),
    code_review_transitions: b.code_review_transitions || 0,
    ready_for_test_transitions: b.ready_for_test_transitions || 0,
    commit_messages: a.commit_messages || [],
    pull_request_titles: a.pull_request_titles || [],
    changelog_evidence: {
      branch_evidence: b.branch_evidence || [],
      pr_evidence: b.pr_evidence || [],
      merged_pr_evidence: b.merged_pr_evidence || [],
      commit_evidence: b.commit_evidence || [],
      review_evidence: b.review_evidence || []
    }
  };
}
```

А в основном цикле по задачам вместо:

```javascript
const devInfo = normalizeDevStatus(devStatusByIssue[issueKey]);
```

делаешь:

```javascript
const devInfo = mergeDevSignals(
  normalizeDevStatus(devStatusByIssue[issueKey]),
  changelogDevByIssue[issueKey]
);
```

# Как это влияет на evidence strength

Для developer это хороший сигнал даже без dev-status:

* `branches > 0` → есть работа в ветке
* `pull_requests > 0` → есть кодовая поставка
* `merged_pull_requests > 0` → есть завершенный merge
* `code_review_transitions > 0` → есть прохождение review-пайплайна
* `ready_for_test_transitions > 0` → есть движение в delivery flow

Я бы трактовал так:

* только worklog → weak/medium
* worklog + status done → medium
* worklog + PR/branch/review → medium/high
* merged PR + done + worklog → strong
