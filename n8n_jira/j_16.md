# 2. Postgres DDL

Ниже минимальный, но уже хороший каркас.

## 2.1 Справочники и сотрудники

```sql
create table employees (
  employee_id text primary key,
  full_name text not null,
  email text not null unique,
  role_id text not null,
  grade_id text not null,
  team_id text,
  manager_id text,
  is_active boolean not null default true,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table competency_catalogs (
  catalog_version text primary key,
  payload jsonb not null,
  created_at timestamptz not null default now()
);

create table career_tracks (
  track_version text primary key,
  payload jsonb not null,
  created_at timestamptz not null default now()
);

create table training_catalogs (
  training_version text primary key,
  payload jsonb not null,
  created_at timestamptz not null default now()
);
```

---

## 2.2 Self-assessment

```sql
create extension if not exists pgcrypto;

create table self_assessments (
  assessment_id uuid primary key default gen_random_uuid(),
  employee_id text not null references employees(employee_id),
  submitted_at timestamptz not null,
  source text not null default 'manual',
  payload jsonb not null,
  created_at timestamptz not null default now()
);

create index idx_self_assessments_employee_submitted_at
  on self_assessments(employee_id, submitted_at desc);
```

---

## 2.3 Снапшоты

```sql
create table assessment_snapshots (
  snapshot_id uuid primary key default gen_random_uuid(),
  employee_id text not null references employees(employee_id),
  generated_at timestamptz not null default now(),
  lookback_days integer not null,
  source text not null default 'jira+self_assessment',
  status text not null check (status in ('queued', 'running', 'completed', 'failed')),
  snapshot_hash text not null,
  source_max_updated_at timestamptz,
  payload jsonb not null,
  created_at timestamptz not null default now()
);

create index idx_assessment_snapshots_employee_generated
  on assessment_snapshots(employee_id, generated_at desc);

create index idx_assessment_snapshots_status
  on assessment_snapshots(status);

create index idx_assessment_snapshots_payload_gin
  on assessment_snapshots using gin (payload jsonb_path_ops);
```

---

## 2.4 Последний актуальный snapshot

```sql
create table latest_assessment_snapshots (
  employee_id text primary key references employees(employee_id),
  snapshot_id uuid not null references assessment_snapshots(snapshot_id),
  generated_at timestamptz not null,
  source_max_updated_at timestamptz,
  payload jsonb not null,
  updated_at timestamptz not null default now()
);
```

---

## 2.5 Быстрые projections для API

```sql
create table latest_assessment_views (
  employee_id text primary key references employees(employee_id),
  snapshot_id uuid not null references assessment_snapshots(snapshot_id),
  generated_at timestamptz not null,

  overview jsonb not null,
  competencies jsonb not null,

```

---

## 2.6 Jobs для запуска heavy workflow

```sql
create table assessment_jobs (
  job_id uuid primary key default gen_random_uuid(),
  employee_id text not null references employees(employee_id),
  requested_by text,
  lookback_days integer not null,
  force_refresh boolean not null default false,
  status text not null check (status in ('queued', 'running', 'completed', 'failed')),
  error_message text,
  snapshot_id uuid references assessment_snapshots(snapshot_id),
  created_at timestamptz not null default now(),
  started_at timestamptz,
  finished_at timestamptz
);

create index idx_assessment_jobs_employee_created
  on assessment_jobs(employee_id, created_at desc);

create index idx_assessment_jobs_status
  on assessment_jobs(status);
```

## 2.8 Полезные SQL upsert-шаблоны

### upsert latest snapshot

```sql
insert into latest_assessment_snapshots (
  employee_id,
  snapshot_id,
  generated_at,
  source_max_updated_at,
  payload,
  updated_at
)
values ($1, $2, $3, $4, $5::jsonb, now())
on conflict (employee_id)
do update set
  snapshot_id = excluded.snapshot_id,
  generated_at = excluded.generated_at,
  source_max_updated_at = excluded.source_max_updated_at,
  payload = excluded.payload,
  updated_at = now();
```

# 4. Финальный список n8n нод

Ниже уже **рабочая логика heavy workflow под batching**.

---

## Блок A. Start + cache

### 1. `Webhook / Start Assessment`

**Input:** request body
**Output:**

```json
{
  "employee_id": "87654321",
  "lookback_days": 180,
  "force_refresh": false,
  "features": {}
}
```

### 2. `Code / Normalize Input`

**Input:** webhook input
**Output:**

```json
{
  "employee_id": "87654321",
  "lookback_days": 180,
  "force_refresh": false,
  "features": {
    "development": true,
    "one2one": true,
    "promotion": true
  }
}
```

### 3. `Postgres / Create Assessment Job`

**Input:** normalized input
**Output:** `job_id`

### 4. `Postgres / Find Fresh Latest Snapshot`

Ищет в `latest_assessment_snapshots` по `employee_id`.

**Input:** normalized input
**Output:**

```json
{
  "found": true,
  "payload": {}
}
```

### 5. `IF / Use Cached Snapshot`

Условие:

* есть snapshot
* `force_refresh = false`
* snapshot достаточно свежий

### 6. `Respond / Return Cached Snapshot`

Возвращает cached snapshot и завершает workflow.

---

## Блок B. Load reference data

### 7. `Postgres / Load Employee`

**Output:** employee object

### 8. `Postgres / Load Latest Self Assessment`

**Output:** self assessment

### 9. `Postgres / Load Competency Catalog`

**Output:** competency catalog

### 10. `Postgres / Load Career Track`

**Output:** career track payload

### 11. `Postgres / Load Training Catalog`

**Output:** training catalog payload

### 12. `Merge / Build Root Context`

Собирает employee + self + catalogs.

### 16. `HTTP / Search Jira Issues`

### 19. `Code / Extract Dev Signals From Changelog`

# 5. Принципы, чтобы workflow не убить

## 5.1 LLM только на крупные смысловые блоки

Оставляем 7 вызовов максимум:

* Analyze All Competencies
* Build Employee Summary
* Jira Summary
* Build IDP
* Development Strategy
* External Learning Suggestions
* One2One Employee
* One2One Manager

Если надо резать дальше — `External Learning` и `Development Strategy` можно слить.

---

## 5.2 Перед каждым LLM — trim payload

Отдельная code-нода, которая:

* режет comment snippets до 300–400 символов
* берет top 2–3 evidence на competency
* выкидывает raw descriptions целиком, если уже есть summary

---

## 5.3 Не гонять heavy workflow на каждый UI запрос

UI всегда читает:

* `latest_assessment_views`
* или `latest_assessment_snapshots`

Heavy workflow — только по кнопке, cron или если snapshot протух.

---

## 5.4 Версионировать каталоги

У snapshot всегда должны быть:

* `competency_catalog_version`
* `career_track_version`
* `training_catalog_version`

Потом это сильно упростит дебаг и сравнение старых результатов.
