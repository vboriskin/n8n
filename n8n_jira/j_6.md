# 1. Финальная `Build Jira Summary Payload`

Эта нода делает **отдельную summary-ветку только по Jira**, без компетенций.

Она учитывает:

* issues
* statuses
* issue types
* worklog
* timetracking
* dev-status
* changelog fallback

На вход ждет один item примерно такого вида:

```json
{
  "employee": { ... },
  "workflow_config": { ... },
  "issues": [ ... ],
  "dev_status_by_issue": { ... },
  "changelog_dev_by_issue": { ... }
}
```

## n8n Code node: `Build Jira Summary Payload`

Режим:
`Run Once for All Items`

```javascript
function asArray(v) {
  return Array.isArray(v) ? v : [];
}

function text(v) {
  return String(v ?? '').trim();
}

function lower(v) {
  return text(v).toLowerCase();
}

function uniq(arr) {
  return [...new Set(arr.filter(Boolean))];
}

function toNum(v, fallback = 0) {
  const n = Number(v);
  return Number.isFinite(n) ? n : fallback;
}

function sum(arr) {
  return arr.reduce((acc, x) => acc + (Number(x) || 0), 0);
}

const input = items[0].json;
const employee = input.employee || {};
const issues = asArray(input.issues);
const devStatusByIssue = input.dev_status_by_issue || {};
const changelogDevByIssue = input.changelog_dev_by_issue || {};

function normalizeWorklog(worklogField) {
  const worklogs = asArray(worklogField?.worklogs);

  const ownEntries = worklogs.filter(w => {
    const authorDisplayName = lower(w?.author?.displayName);
    const authorName = lower(w?.author?.name);
    const authorEmail = lower(w?.author?.emailAddress);

    const empName = lower(employee.full_name);
    const empId = lower(employee.employee_id);
    const empEmail = lower(employee.email);

    return (
      (empName && authorDisplayName.includes(empName)) ||
      (empId && authorName.includes(empId)) ||
      (empEmail && authorEmail.includes(empEmail))
    );
  });

  const targetEntries = ownEntries.length ? ownEntries : worklogs;

  const secondsTotal = sum(targetEntries.map(w => toNum(w.timeSpentSeconds, 0)));
  const activeDates = uniq(
    targetEntries.map(w => text(w.started).slice(0, 10)).filter(Boolean)
  );

  return {
    entries_count: targetEntries.length,
    seconds_total: secondsTotal,
    days_total: Number((secondsTotal / 28800).toFixed(2)),
    active_days: activeDates.length,
    has_comments: targetEntries.some(w => text(w.comment)),
    author_match: ownEntries.length > 0
  };
}

function normalizeDevStatus(raw) {
  if (!raw) {
    return {
      branches: 0,
      commits: 0,
      pull_requests: 0,
      merged_pull_requests: 0
    };
  }

  const detail = asArray(raw.detail);
  const summary = raw.summary || {};

  let branches = 0;
  let commits = 0;
  let pullRequests = 0;
  let mergedPullRequests = 0;

  for (const d of detail) {
    const repos = asArray(d.repositories);
    for (const repo of repos) {
      branches += asArray(repo.branches).length;

      const repoCommits = asArray(repo.commits);
      commits += repoCommits.length;

      const prs = asArray(repo.pullRequests);
      pullRequests += prs.length;
      mergedPullRequests += prs.filter(pr =>
        lower(pr?.status || pr?.state).includes('merged')
      ).length;
    }
  }

  if (!branches) branches = toNum(summary.branches, 0);
  if (!commits) commits = toNum(summary.commits, 0);
  if (!pullRequests) pullRequests = toNum(summary.pullRequests, 0);

  return {
    branches,
    commits,
    pull_requests: pullRequests,
    merged_pull_requests: mergedPullRequests
  };
}

function mergeDevSignals(devStatusInfo, changelogInfo) {
  const a = devStatusInfo || {
    branches: 0,
    commits: 0,
    pull_requests: 0,
    merged_pull_requests: 0
  };

  const b = changelogInfo || {
    branches: 0,
    pull_requests: 0,
    merged_pull_requests: 0,
    commit_signals: 0,
    code_review_transitions: 0,
    ready_for_test_transitions: 0
  };

  return {
    branches: Math.max(toNum(a.branches, 0), toNum(b.branches, 0)),
    commits: Math.max(toNum(a.commits, 0), toNum(b.commit_signals, 0)),
    pull_requests: Math.max(toNum(a.pull_requests, 0), toNum(b.pull_requests, 0)),
    merged_pull_requests: Math.max(toNum(a.merged_pull_requests, 0), toNum(b.merged_pull_requests, 0)),
    code_review_transitions: toNum(b.code_review_transitions, 0),
    ready_for_test_transitions: toNum(b.ready_for_test_transitions, 0)
  };
}

let completedIssues = 0;
let inProgressIssues = 0;
let openIssues = 0;
let reopenedSignals = 0;

let totalWorklogSeconds = 0;
let totalWorklogEntries = 0;
let totalWorklogActiveDays = 0;
let worklogWithCommentsCount = 0;
let worklogAuthorMatchCount = 0;

let totalOriginalEstimateSeconds = 0;
let totalTimeSpentSeconds = 0;
let totalAggregateOriginalEstimateSeconds = 0;
let totalAggregateTimeSpentSeconds = 0;

let totalBranches = 0;
let totalCommits = 0;
let totalPR = 0;
let totalMergedPR = 0;
let totalCodeReviewTransitions = 0;
let totalReadyForTestTransitions = 0;

const typeDistribution = {};
const statusDistribution = {};
const activeIssueKeys = [];
const deliveredIssueKeys = [];
const issueSnapshots = [];

for (const issue of issues) {
  const f = issue.fields || {};
  const issueKey = issue.key;

  const status = text(f.status?.name || 'Unknown');
  const statusLower = lower(status);
  const issueType = text(f.issuetype?.name || 'Unknown');

  statusDistribution[status] = (statusDistribution[status] || 0) + 1;
  typeDistribution[issueType] = (typeDistribution[issueType] || 0) + 1;

  if (
    statusLower.includes('done') ||
    statusLower.includes('closed') ||
    statusLower.includes('resolved')
  ) {
    completedIssues += 1;
    deliveredIssueKeys.push(issueKey);
  } else if (
    statusLower.includes('progress') ||
    statusLower.includes('review') ||
    statusLower.includes('testing') ||
    statusLower.includes('qa')
  ) {
    inProgressIssues += 1;
    activeIssueKeys.push(issueKey);
  } else {
    openIssues += 1;
    activeIssueKeys.push(issueKey);
  }

  if (statusLower.includes('reopen')) {
    reopenedSignals += 1;
  }

  const worklogInfo = normalizeWorklog(f.worklog);
  totalWorklogSeconds += worklogInfo.seconds_total;
  totalWorklogEntries += worklogInfo.entries_count;
  totalWorklogActiveDays += worklogInfo.active_days;
  if (worklogInfo.has_comments) worklogWithCommentsCount += 1;
  if (worklogInfo.author_match) worklogAuthorMatchCount += 1;

  totalOriginalEstimateSeconds += toNum(f.timeoriginalestimate, 0);
  totalTimeSpentSeconds += toNum(f.timespent, 0);
  totalAggregateOriginalEstimateSeconds += toNum(f.aggregatetimeoriginalestimate, 0);
  totalAggregateTimeSpentSeconds += toNum(f.aggregatetimespent, 0);

  const devInfo = mergeDevSignals(
    normalizeDevStatus(devStatusByIssue[issueKey]),
    changelogDevByIssue[issueKey]
  );

  totalBranches += devInfo.branches;
  totalCommits += devInfo.commits;
  totalPR += devInfo.pull_requests;
  totalMergedPR += devInfo.merged_pull_requests;
  totalCodeReviewTransitions += devInfo.code_review_transitions;
  totalReadyForTestTransitions += devInfo.ready_for_test_transitions;

  issueSnapshots.push({
    issue_key: issueKey,
    summary: f.summary || null,
    status,
    issue_type: issueType,
    updated: f.updated || null,
    worklog_days: worklogInfo.days_total,
    worklog_entries: worklogInfo.entries_count,
    commits: devInfo.commits,
    pull_requests: devInfo.pull_requests,
    merged_pull_requests: devInfo.merged_pull_requests
  });
}

const totalIssues = issues.length;
const completionRatio = totalIssues ? Number((completedIssues / totalIssues).toFixed(2)) : 0;
const worklogDays = Number((totalWorklogSeconds / 28800).toFixed(2));
const estimateDays = Number((totalOriginalEstimateSeconds / 28800).toFixed(2));
const loggedVsEstimateRatio =
  totalOriginalEstimateSeconds > 0
    ? Number((totalTimeSpentSeconds / totalOriginalEstimateSeconds).toFixed(2))
    : null;

let activityLevelHint = 'medium';
if (totalIssues === 0 && worklogDays === 0 && totalCommits === 0) {
  activityLevelHint = 'low';
} else if (
  completedIssues >= 8 ||
  worklogDays >= 10 ||
  totalMergedPR >= 5 ||
  totalCommits >= 20
) {
  activityLevelHint = 'high';
}

let confidenceHint = 'medium';
if (totalIssues < 3 && worklogDays === 0 && totalCommits === 0) {
  confidenceHint = 'low';
} else if (
  totalIssues >= 5 &&
  (worklogDays > 0 || totalCommits > 0 || totalMergedPR > 0)
) {
  confidenceHint = 'high';
}

const risksHint = [];
if (totalIssues > 0 && completedIssues === 0) risksHint.push('no_completed_issues');
if (totalOriginalEstimateSeconds > 0 && totalTimeSpentSeconds === 0) risksHint.push('no_logged_time_against_estimates');
if (totalPR > 0 && totalMergedPR === 0) risksHint.push('pull_requests_not_merged');
if (totalIssues > 0 && worklogAuthorMatchCount === 0) risksHint.push('worklog_author_not_matched');
if (reopenedSignals > 0) risksHint.push('reopened_or_unstable_flow_signals');

return [
  {
    json: {
      jira_summary_payload: {
        employee: {
          employee_id: employee.employee_id || null,
          full_name: employee.full_name || null,
          email: employee.email || null,
          role_id: employee.role_id || null,
          grade_id: employee.grade_id || null
        },
        activity_summary: {
          total_issues: totalIssues,
          completed_issues: completedIssues,
          in_progress_issues: inProgressIssues,
          open_issues: openIssues,
          completion_ratio: completionRatio,
          reopened_signals: reopenedSignals,

          worklog: {
            total_days: worklogDays,
            entries: totalWorklogEntries,
            active_days: totalWorklogActiveDays,
            with_comments_issue_count: worklogWithCommentsCount,
            author_match_issue_count: worklogAuthorMatchCount
          },

          timetracking: {
            estimated_days: estimateDays,
            logged_days: Number((totalTimeSpentSeconds / 28800).toFixed(2)),
            aggregate_estimated_days: Number((totalAggregateOriginalEstimateSeconds / 28800).toFixed(2)),
            aggregate_logged_days: Number((totalAggregateTimeSpentSeconds / 28800).toFixed(2)),
            logged_vs_estimate_ratio: loggedVsEstimateRatio
          },

          development: {
            branches: totalBranches,
            commits: totalCommits,
            pull_requests: totalPR,
            merged_pull_requests: totalMergedPR,
            code_review_transitions: totalCodeReviewTransitions,
            ready_for_test_transitions: totalReadyForTestTransitions
          }
        },

        distribution: {
          issue_types: typeDistribution,
          statuses: statusDistribution
        },

        signals: {
          has_worklog: totalWorklogEntries > 0,
          has_dev_activity: totalCommits > 0 || totalPR > 0 || totalBranches > 0,
          has_merged_pr: totalMergedPR > 0,
          has_code_review_flow: totalCodeReviewTransitions > 0,
          has_delivery_flow: completedIssues > 0 || totalReadyForTestTransitions > 0
        },

        precomputed: {
          activity_level_hint: activityLevelHint,
          confidence_hint: confidenceHint,
          risks_hint: risksHint,
          active_issue_keys: activeIssueKeys.slice(0, 20),
          delivered_issue_keys: deliveredIssueKeys.slice(0, 20),
          issue_snapshots: issueSnapshots.slice(0, 30)
        }
      }
    }
  }
];
```

## LLM для Jira summary

### System prompt

```text
Ты анализируешь активность сотрудника по данным Jira.

Правила:
- Опирайся только на переданные данные
- Не выдумывай данные
- Worklog означает участие, но сам по себе не доказывает высокий уровень
- Commits, pull requests и merged pull requests — инженерный сигнал для developer
- Если данных мало, укажи это явно
- Не путай отсутствие данных с низкой активностью

Верни строго JSON без markdown.
```

```text
  "engineering_signal": "string",
  "risks": ["string"],

Данные:
{{ JSON.stringify($json.jira_summary_payload) }}
```

# 2. Полная схема workflow по нодам n8n

Ниже дам **практическую схему**, чтобы можно было собрать без додумывания.

## Ветка A — получение Jira данных

### `Set Input`

На входе:

* `employee.email`
* `employee.full_name`
* `employee.role_id`
* `workflow_config.jira_base_url`
* `workflow_config.lookback_days`
* `competency_catalog`
* `assessment_input`
* `scoring_input`
* `jira_token`

### `HTTP Request / Find Jira User By Email`

`GET`

```text
{{$json.workflow_config.jira_base_url}}/rest/api/2/user/search
```

Query param:

```text
username = {{$json.employee.email}}
```

### `Code / Pick Jira User`

Берет первого найденного пользователя и возвращает:

* `jira_user.name`
* `jira_user.key`
* `jira_user.emailAddress`

### `Merge / Attach Jira User`

Mode:
`Combine -> By Position`

Соединяет:

* исходный input
* найденного jira user

---

## Ветка B — поиск задач

### `HTTP Request / Jira Search Issues`

`POST`

```text
{{$json.workflow_config.jira_base_url}}/rest/api/2/search
```

Body:

```javascript
={{
{
  jql: `(
    assignee = "${$json.jira_user.key || $json.jira_user.name}"
    OR reporter = "${$json.jira_user.key || $json.jira_user.name}"
    OR worklogAuthor = "${$json.jira_user.key || $json.jira_user.name}"
  ) AND updated >= -${$json.workflow_config.lookback_days || 180}d ORDER BY updated DESC`,
  startAt: 0,
  maxResults: 100,
  fields: [
    "summary",
    "description",
    "labels",
    "components",
    "issuetype",
    "status",
    "resolution",
    "assignee",
    "reporter",
    "created",
    "updated",
    "comment",
    "worklog",
    "timespent",
    "timeoriginalestimate",
    "aggregatetimespent",
    "aggregatetimeoriginalestimate",
    "timetracking"
  ],
  expand: ["changelog"]
}
}}
```

### `Code / Split Issues`

Разбивает `issues[]` в отдельные items:

* `issue_key`
* `issue_id`
* `base_issue`

### `HTTP Request / Jira Issue Details`

`GET`

```text
{{$item(0).$node["Merge / Attach Jira User"].json.workflow_config.jira_base_url}}/rest/api/2/issue/{{$json.issue_key}}
```

Query params:

* `fields=summary,description,labels,components,issuetype,status,resolution,assignee,reporter,created,updated,comment,worklog,timespent,timeoriginalestimate,aggregatetimespent,aggregatetimeoriginalestimate,timetracking`
* `expand=changelog`

---

## Ветка C — developer enrichment

### `IF / Is Developer`

Condition:

```text
{{$item(0).$node["Merge / Attach Jira User"].json.employee.role_id === "developer"}}
```

### True branch → `HTTP Request / Jira Dev Status`

`GET`

```text
{{$item(0).$node["Merge / Attach Jira User"].json.workflow_config.jira_base_url}}/rest/dev-status/latest/issue/detail?issueId={{$json.id || $json.issue_id}}&applicationType=stash&dataType=repository
```

`Continue On Fail = true`

### `Code / Build Dev Status Map`

Собирает map:

```json
{
  "SMECM-22185": {
    "summary": {},
    "detail": []
  }
}
```

`Code / Extract Dev Signals From Changelog`

Это та нода, которую мы уже написали.

### `Code / Build Changelog Dev Map`

Собирает map:

```json
{
  "SMECM-22185": {

    "merged_pull_requests": 1,
    "commit_signals": 3,
    "code_review_transitions": 1,
```

## Ветка D — сборка полного input

### `Code / Build Full Jira Input`

Эта нода собирает:

* root input из `Merge / Attach Jira User`
* all issue details
* `dev_status_by_issue`
* `changelog_dev_by_issue`

То есть на выходе один item:

```json
{
  "employee": {...},
  "workflow_config": {...},
  "competency_catalog": {...},
  "assessment_input": {...},
  "scoring_input": {...},
  "issues": [...],
  "dev_status_by_issue": {...},
  "changelog_dev_by_issue": {...}
}
```

## Ветка E — две независимые аналитики

1. `Code / Build Jira Summary Payload`
2. `LLM / Jira Summary`

# Что лучше сделать следующим шагом

Сейчас самый полезный следующий шаг — это не еще одна концепция, а **готовый JSON workflow-шаблон по нодам n8n**, хотя бы в упрощенном виде с названиями и основными параметрами.

Я могу следующим сообщением дать тебе именно это:
**каркас workflow в виде списка нод с точными именами, режимами и связями “из какой ноды в какую”**.
