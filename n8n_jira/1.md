Да — это лучше собрать в **3 Code-ноды + 2 LLM-ноды + 1 финальную ноду-сборщик**.

Судя по файлу, у матрицы действительно структура вида **role → competency_groups → competencies → levels 0..4** , так что удобнее сначала нормализовать всё в единый список компетенций, потом отранжировать, потом отдавать в LLM только top N.

Ниже дам рабочую схему для n8n и готовый JS.

---

# Что лучше сделать по пайплайну

## Узел 1. `Code / Build Competency Payloads`

Из:

* matrix
* self_assessment
* jira evidence

собираем список items, где каждый item содержит:

* `competency_id`
* `competency`
* `self_assessment_item`
* `jira_evidence`
* `target_level`
* `priority_score`
* `priority`
* `payload` для LLM по одной компетенции

## Узел 2. `Code / Pick Top Competencies`

Сортируем по `priority_score`, берем top N, например 15.

## Узел 3. `LLM / Analyze One Competency`

Запускается по одному item.
Возвращает JSON по одной компетенции.

## Узел 4. `Code / Build Employee Summary Payload`

Собирает результаты компетенций в payload для summary LLM.

## Узел 5. `LLM / Build Employee Summary`

Возвращает summary JSON.

## Узел 6. `Code / Build WF30 Output`

Формирует итог:

```json
{
  "competency_analysis": {
    "competency_results": [],
    "employee_summary": {},
    "top_priority_competencies": [],
    "analysis_quality": "medium",
    "flags": []
  }
}
```

# 1) Code node: Build Competency Payloads

Название ноды: `Build Competency Payloads`

```javascript
// n8n Code node
// Mode: Run Once for All Items

function asArray(v) {
  if (Array.isArray(v)) return v;
  if (v == null) return [];
  return [v];
}

function normalizeStr(v) {
  return String(v || '')
    .trim()
    .toLowerCase();
}

function slugify(v) {
  return normalizeStr(v)
    .replace(/[^\p{L}\p{N}]+/gu, '_')
    .replace(/^_+|_+$/g, '');
}

function firstNonEmpty(...values) {
  for (const v of values) {
    if (v !== undefined && v !== null && v !== '') return v;
  }
  return null;
}

function parseLevel(v) {
  if (v === null || v === undefined || v === '') return null;
  const n = Number(v);
  return Number.isFinite(n) ? n : null;
}

function isSoftSkill(competency, groupName) {
  const text = normalizeStr(
    [
      competency?.competency_name,
      competency?.name,
      competency?.description,
      groupName,
    ].join(' ')
  );

  const softMarkers = [
    'soft',
    'коммуника',
    'заинтересован',
    'встреч',
    'фасилита',
    'презентац',
    'переговор',
    'обратн',
    'лидер',
    'ментор',
    'collaboration',
    'stakeholder',
  ];

  return softMarkers.some(m => text.includes(m));
}

function inferHardnessBoost(competency, groupName) {
  const text = normalizeStr(
    [
      competency?.competency_name,
      competency?.name,
      competency?.description,
      groupName,
    ].join(' ')
  );

  const hardMarkers = [
    'api', 'sql', 'архитект', 'баз', 'данн', 'cicd', 'jenkins',
    'mobile', 'web', 'backend', 'frontend', 'integration', 'интеграц',
    'лог', 'монитор', 'безопас', 'test design', 'тест', 'автоматиза',
    'framework', 'фреймворк', 'code', 'ооп', 'паттерн', 'git'
  ];

  return hardMarkers.some(m => text.includes(m)) ? 1 : 0;
}

// target level heuristic
function calculateTargetLevel({ selfLevel, currentGrade, matrixMaxLevel = 4 }) {
  const gradeMap = {
    junior: 1,
    middle: 2,
    middle_high: 2,
    senior: 3,
    senior_high: 4,
    lead: 4,
    principal: 4,
  };

  const baseFromGrade = gradeMap[normalizeStr(currentGrade)] ?? 2;
  let target = baseFromGrade;

  if (selfLevel !== null) {
    // мягкая логика: target не ниже self, но и не улетает слишком далеко
    target = Math.max(target, Math.min(selfLevel + 1, matrixMaxLevel));
  }

  return Math.min(target, matrixMaxLevel);
}

function calcPriorityScore({
  matrixPriorityWeight,
  hasSelfAssessment,
  hasComment,
  hasJiraEvidence,
  hasGap,
  isSoft,
  hardBoost,
}) {
  let score = 0;

  // базовый вес компетенции
  score += matrixPriorityWeight * 10;

  // наличие самооценки сильно повышает значимость для анализа
  if (hasSelfAssessment) score += 20;

  // есть текстовый комментарий => LLM есть с чем работать
  if (hasComment) score += 10;

  // jira evidence — очень ценно
  if (hasJiraEvidence) score += 18;

  // gap до target — важный сигнал
  if (hasGap) score += 16;

  // hard skill немного приоритетнее
  if (hardBoost) score += 8;

  // soft skills чуть ниже, если есть критичные hard
  if (isSoft) score -= 6;

  return Math.max(score, 0);
}

function scoreToPriority(score) {
  if (score >= 60) return 'high';
  if (score >= 35) return 'medium';
  return 'low';
}

// ---- collect source data ----
const input = items[0].json;

// matrix can be in different shapes
const matrices =
  asArray(input.matrices).length
    ? asArray(input.matrices)
    : asArray(input.matrix);

// self assessment can be array or object
const selfAssessmentRaw =
  input.self_assessment ??
  input.selfAssessment ??
  input.assessment ??
  [];

const jiraEvidenceRaw =
  input.jira_evidence ??
  input.jiraEvidence ??
  input.evidence ??
  [];

const employee =
  input.employee ??
  input.employee_profile ??
  input.employeeInfo ??
  {};

const employeeGrade = firstNonEmpty(employee.grade, input.grade, input.employee_grade);

// ---- normalize self assessment into map ----
const selfMap = new Map();

for (const row of asArray(selfAssessmentRaw)) {
  const key =
    firstNonEmpty(
      row.competency_id,
      row.competencyId,
      row.id,
      slugify(row.competency_name),
      slugify(row.name)
    );

  if (!key) continue;

  selfMap.set(key, {
    competency_id: key,
    self_level: parseLevel(firstNonEmpty(row.self_level, row.level, row.selfLevel)),
    comment: firstNonEmpty(row.comment, row.notes, row.text, row.justification),
    raw: row,
  });
}

// ---- normalize jira evidence into map ----
const evidenceMap = new Map();

for (const row of asArray(jiraEvidenceRaw)) {
  const key =
    firstNonEmpty(
      row.competency_id,
      row.competencyId,
      row.id,
      slugify(row.competency_name),
      slugify(row.name)
    );

  if (!key) continue;

  const prev = evidenceMap.get(key) || [];
  prev.push({
    summary: firstNonEmpty(row.summary, row.title, row.issue_summary),
    key: firstNonEmpty(row.key, row.issueKey, row.ticket),
    status: firstNonEmpty(row.status),
    comment: firstNonEmpty(row.comment, row.description, row.evidence),
    raw: row,
  });
  evidenceMap.set(key, prev);
}

// ---- flatten matrix competencies ----
const competencyItems = [];

for (const matrix of matrices) {
  const role = firstNonEmpty(matrix.role, input.role, employee.role);
  const groups = asArray(matrix.compentecy_groups || matrix.competency_groups);

  for (const group of groups) {
    const groupName = firstNonEmpty(group.group_name, group.name, 'unknown_group');
    const competencies = asArray(group.competencies);

    for (const competency of competencies) {
      const competencyName = firstNonEmpty(
        competency.competency_name,
        competency.name,
        competency.title
      );

      if (!competencyName) continue;

      const competencyId = firstNonEmpty(
        competency.competency_id,
        competency.id,
        slugify(competencyName)
      );

      const levels = asArray(competency.levels);
      const matrixMaxLevel = levels.length
        ? Math.max(...levels.map(l => parseLevel(l.level)).filter(v => v !== null))
        : 4;

      const selfItem = selfMap.get(competencyId) || null;
      const jiraEvidence = evidenceMap.get(competencyId) || [];

      const selfLevel = selfItem?.self_level ?? null;
      const targetLevel = calculateTargetLevel({
        selfLevel,
        currentGrade: employeeGrade,
        matrixMaxLevel,
      });

      const gap =
        selfLevel !== null && targetLevel !== null
          ? Math.max(targetLevel - selfLevel, 0)
          : null;

      const soft = isSoftSkill(competency, groupName);
      const hardBoost = inferHardnessBoost(competency, groupName);

      // matrix priority weight heuristic
      // if explicitly set in matrix - use it, else infer
      const matrixPriorityWeight =
        Number(firstNonEmpty(
          competency.priority_weight,
          competency.priorityWeight,
          group.priority_weight,
          group.priorityWeight,
          hardBoost ? 3 : 2
        )) || 2;

      const priorityScore = calcPriorityScore({
        matrixPriorityWeight,
        hasSelfAssessment: !!selfItem,
        hasComment: !!selfItem?.comment,
        hasJiraEvidence: jiraEvidence.length > 0,
        hasGap: gap !== null && gap > 0,
        isSoft: soft,
        hardBoost,
      });

      competencyItems.push({
        competency_id: competencyId,
        competency: {
          competency_id: competencyId,
          competency_name: competencyName,
          group_name: groupName,
          role,
          levels,
          raw: competency,
        },
        self_assessment_item: selfItem?.raw ?? null,
        jira_evidence: jiraEvidence,
        self_level: selfLevel,
        target_level: targetLevel,
        gap,
        matrix_priority_weight: matrixPriorityWeight,
        priority_score: priorityScore,
        priority: scoreToPriority(priorityScore),
        payload: {
          employee: {
            grade: employeeGrade,
            role: firstNonEmpty(employee.role, role),
          },
          competency: {
            competency_id: competencyId,
            competency_name: competencyName,
            group_name: groupName,
            role,
            levels,
          },
          self_assessment: selfItem?.raw ?? null,
          jira_evidence: jiraEvidence,
          target_level: targetLevel,
          priority_score: priorityScore,
          priority: scoreToPriority(priorityScore),
        },
      });
    }
  }
}

// sort descending right here
competencyItems.sort((a, b) => b.priority_score - a.priority_score);

return competencyItems.map(item => ({ json: item }));
```

# 5) Code node: Build Employee Summary Payload

Эта нода должна собрать все competency_results в единый payload.

Название: `Build Employee Summary Payload`

```javascript
// n8n Code node
// Mode: Run Once for All Items

const competencyResults = items.map(i => i.json);

function topPriorityCompetencies(results, topN = 5) {
  const scored = results.map(r => {
    let s = 0;
    if (r.priority === 'high') s += 3;
    else if (r.priority === 'medium') s += 2;
    else s += 1;

    if (r.confidence === 'high') s += 2;
    else if (r.confidence === 'medium') s += 1;

    if (Array.isArray(r.risk_flags) && r.risk_flags.length) s += 2;

    const selfLevel = Number(r.self_level ?? 0);
    const estimatedLevel = Number(r.estimated_level ?? 0);
    const gap = Math.max(selfLevel - estimatedLevel, estimatedLevel - selfLevel);

    s += gap;

    return {
      competency_id: r.competency_id,
      score: s,
    };
  });

  scored.sort((a, b) => b.score - a.score);
  return scored.slice(0, topN).map(x => x.competency_id);
}

return [
  {
    json: {
      employee_summary_payload: {
        competency_results: competencyResults,
        top_priority_candidates: topPriorityCompetencies(competencyResults, 6),
      },
    },
  },
];
```

# 7) Code node: Build WF30 Output

Тут уже формируем финальный результат.

Название: `Build WF30 Output`

Эта нода ожидает:

* из одной ветки: список competency_results
* из другой ветки: employee_summary

Обычно это делается через Merge node перед ней.

```javascript
// n8n Code node
// Mode: Run Once for All Items

function parseMaybeJson(v) {
  if (typeof v === 'string') {
    try {
      return JSON.parse(v);
    } catch {
      return v;
    }
  }
  return v;
}

const parsed = items.map(i => parseMaybeJson(i.json));

// Пытаемся разделить summary и competency results
const competencyResults = [];
let employeeSummary = null;

for (const item of parsed) {
  if (
    item &&
    typeof item === 'object' &&
    'competency_id' in item &&
    'estimated_level' in item
  ) {
    competencyResults.push(item);
    continue;
  }

  if (
    item &&
    typeof item === 'object' &&
    'overall_summary' in item &&
    'strengths' in item
  ) {
    employeeSummary = item;
  }
}

const topPriorityCompetencies = competencyResults
  .filter(r => r.priority === 'high')
  .slice(0, 5)
  .map(r => r.competency_id);

let analysisQuality = 'medium';
const lowConfidenceCount = competencyResults.filter(r => r.confidence === 'low').length;

if (competencyResults.length > 0) {
  if (lowConfidenceCount / competencyResults.length > 0.5) {
    analysisQuality = 'low';
  } else if (lowConfidenceCount === 0) {
    analysisQuality = 'high';
  }
}

const flags = [];
if (!employeeSummary) {
  flags.push('employee_summary_missing');
}
if (!competencyResults.length) {
  flags.push('competency_results_missing');
}
if (lowConfidenceCount > 0) {
  flags.push('some_competencies_low_confidence');
}

return [
  {
    json: {
      competency_analysis: {
        competency_results: competencyResults,
        employee_summary: employeeSummary || {
          overall_summary: '',
          strengths: [],
          growth_areas: [],
          alignment: {
            overestimated: [],
            underestimated: [],
            aligned: [],
          },
          top_priorities_next_6m: [],
          confidence: 'low',
        },
        top_priority_competencies: employeeSummary?.top_priorities_next_6m?.length
          ? employeeSummary.top_priorities_next_6m
          : topPriorityCompetencies,
        analysis_quality: analysisQuality,
        flags,
      },
    },
  },
];
```
