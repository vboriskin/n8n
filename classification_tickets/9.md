function cleanText(value) {
  if (value === null || value === undefined) return '';
  let text = String(value).replace(/\s+/g, ' ').trim();
  text = text.replace(/[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}/gi, '[EMAIL]');
  text = text.replace(/(\+7|8)\s*\(?\d{3}\)?[\s-]*\d{3}[\s-]*\d{2}[\s-]*\d{2}/g, '[PHONE]');
  text = text.replace(/\b\d{10,20}\b/g, '[LONG_NUMBER]');
  text = text.replace(/\b[А-ЯЁ][а-яё]+\s+[А-ЯЁ]\.\s?[А-ЯЁ]\.\b/g, '[PERSON]');
  return text;
}

function truncateSmart(text, maxLen = 3000) {
  if (!text) return '';
  if (text.length <= maxLen) return text;
  const head = text.slice(0, Math.floor(maxLen * 0.7));
  const tail = text.slice(-Math.floor(maxLen * 0.3));
  return `${head} ... [TRUNCATED] ... ${tail}`;
}

function parseReturnCount(value) {
  if (value === null || value === undefined || value === '') return 0;
  const num = String(value).match(/\d+/);
  return num ? Number(num[0]) : 0;
}

function parseDateFlexible(value) {
  if (!value) return null;
  const str = String(value).trim();
  const m = str.match(/^(\d{2})\.(\d{2})\.(\d{4})(?:\s+(\d{2}):(\d{2})(?::(\d{2}))?)?$/);
  if (m) {
    const [, dd, mm, yyyy, hh = '00', mi = '00', ss = '00'] = m;
    return new Date(`${yyyy}-${mm}-${dd}T${hh}:${mi}:${ss}`);
  }
  const d = new Date(str);
  return isNaN(d.getTime()) ? null : d;
}

function parseDurationToHours(value) {
  if (!value) return null;
  const str = String(value).trim();
  let m = str.match(/^(\d+)\s+(\d{2}):(\d{2}):(\d{2})$/);
  if (m) {
    const [, days, hh, mm, ss] = m;
    return Number(days) * 24 + Number(hh) + Number(mm) / 60 + Number(ss) / 3600;
  }
  m = str.match(/^(\d{2}):(\d{2}):(\d{2})$/);
  if (m) {
    const [, hh, mm, ss] = m;
    return Number(hh) + Number(mm) / 60 + Number(ss) / 3600;
  }
  return null;
}

return items.map((item) => {
  const j = item.json;
  const ddl = parseDateFlexible(j.deadline_datetime);
  const fin = parseDateFlexible(j.finish_datetime);

  let overdue_flag = 'unknown';
  let overdue_hours = null;

  if (ddl && fin) {
    const diffMs = fin.getTime() - ddl.getTime();

      overdue_flag,

    }

function containsAny(text, patterns) {
  if (!text) return false;

  if (hours === null || hours === undefined || isNaN(hours)) return 'unknown';

function norm(value) {
  return String(value || '')
    .toLowerCase()
    .replace(/ё/g, 'е')
    .replace(/[^\p{L}\p{N}\s#/_-]/gu, ' ')
    .replace(/\s+/g, ' ')
    .trim();
}

function canonicalTeam(value) {
  const v = norm(value);
  if (v.includes('hydra') || v.includes('гидра')) return 'Hydra';
  return value || 'unknown';
}

function canonicalExecutorGroup(value) {
  const v = norm(value);
  if (v.includes('поддержка hydra')) return 'Поддержка Hydra';
  if (v.includes('рг гидра')) return 'РГ Гидра';
  return value || 'unknown';
}

function extractTags(tagRaw) {
  const raw = String(tagRaw || '');
  return [...new Set(raw.split('#').map(s => norm(s)).filter(Boolean))];
}

function canonicalizeTag(t) {
  if (!t) return null;
  if (t.includes('it проблема')) return 'it_problem';
  if (t.includes('перемиграц')) return 'migration_request';
  if (t.includes('доступ')) return 'access';
  return t.replace(/\s+/g, '_');
}

return items.map(item => {

function simplify(text) {
  return String(text || '')
    .toLowerCase()
    .replace(/ё/g, 'е')
    .replace(/\b\d{2,}\b/g, ' ')
    .replace(/[^\p{L}\p{N}\s]/gu, ' ')
    .replace(/\s+/g, ' ')
    .trim();
}

function short(text, len) {
  return simplify(text).slice(0, len);
}

function hashCode(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    hash = ((hash << 5) - hash) + str.charCodeAt(i);
    hash |= 0;
  }
  return `fp_${Math.abs(hash)}`;
}

return items.map(item => {
  const j = item.json;
  const base = [
    short(j.topic, 120),
    short((j.tags_canonical || []).join(' '), 120),
    short(j.description, 300),
    short(j.resolution, 200),
    short(j.close_code, 80),
    short(j.executor_group_canonical, 80)
  ].join(' | ');

  return {
    json: {
      ...j,
      dedup_base: base,
      dedup_fingerprint: hashCode(base)
    }
  };
});

function prio(v) {
  return v === 'high' ? 0 : 1;
}

const sorted = items
  .map(i => i.json)
  .sort((a, b) => {
    if (prio(a.llm_priority) !== prio(b.llm_priority)) {
      return prio(a.llm_priority) - prio(b.llm_priority);
    }
    if ((Number(b.rework_count || 0)) !== (Number(a.rework_count || 0))) {
      return Number(b.rework_count || 0) - Number(a.rework_count || 0);
    }
    const ao = a.overdue_flag === 'yes' ? 1 : 0;
    const bo = b.overdue_flag === 'yes' ? 1 : 0;
    return bo - ao;
  })
  .slice(0, 300);

return sorted.map(r => ({ json: r }));

## 28. Code — `aggregate_team_metrics`

**Node:** `Code`
**Mode:** `Run Once for All Items`

```javascript
function num(v) {
  const x = Number(v);
  return Number.isFinite(x) ? x : 0;
}

const groups = {};

for (const item of items) {
  const r = item.json;
  const team = r.team_canonical || r.team || 'unknown';

  if (!groups[team]) {
    groups[team] = {
      team,
      tickets_total: 0,
      team_fault_count: 0,
      high_conf_fault_count: 0,
      overdue_count: 0,
      rework_count: 0,
      failure_demand_count: 0,
      avoidable_count: 0,
      automation_candidate_count: 0,
      system_or_process_issue_count: 0,
      durations: []
    };
  }

  const g = groups[team];
  g.tickets_total += 1;

  if (['yes','partial'].includes(r.team_fault_flag)) g.team_fault_count += 1;
  if (r.team_fault_flag === 'yes' && num(r.confidence) >= 0.85) g.high_conf_fault_count += 1;
  if (r.overdue_flag === 'yes') g.overdue_count += 1;
  if (num(r.rework_count) > 0 || r.rework_flag === 'yes') g.rework_count += 1;
  if (r.failure_demand_flag === 'yes') g.failure_demand_count += 1;
  if (r.avoidable_flag === 'yes') g.avoidable_count += 1;
  if (r.automation_candidate === 'yes') g.automation_candidate_count += 1;
  if (['system','process','integration'].includes(r.root_cause_group)) g.system_or_process_issue_count += 1;

  if (r.total_duration_hours !== null && r.total_duration_hours !== undefined) {
    g.durations.push(num(r.total_duration_hours));
  }
}

return Object.values(groups).map(g => {
  const avgDuration = g.durations.length
    ? g.durations.reduce((a,b) => a+b, 0) / g.durations.length
    : 0;

  const team_fault_rate = g.tickets_total ? g.team_fault_count / g.tickets_total : 0;
  const high_conf_fault_rate = g.tickets_total ? g.high_conf_fault_count / g.tickets_total : 0;
  const overdue_rate = g.tickets_total ? g.overdue_count / g.tickets_total : 0;
  const rework_rate = g.tickets_total ? g.rework_count / g.tickets_total : 0;
  const failure_demand_rate = g.tickets_total ? g.failure_demand_count / g.tickets_total : 0;
  const avoidable_rate = g.tickets_total ? g.avoidable_count / g.tickets_total : 0;
  const automation_candidate_rate = g.tickets_total ? g.automation_candidate_count / g.tickets_total : 0;
  const system_or_process_issue_rate = g.tickets_total ? g.system_or_process_issue_count / g.tickets_total : 0;

  const team_problem_score =
    0.25 * high_conf_fault_rate +
    0.15 * rework_rate +
    0.15 * overdue_rate +
    0.15 * failure_demand_rate +
    0.10 * avoidable_rate +
    0.10 * Math.min(avgDuration / 72, 1) +
    0.10 * system_or_process_issue_rate;

  return {
    json: {
      team: g.team,
      tickets_total: g.tickets_total,
      team_fault_rate: Number(team_fault_rate.toFixed(4)),
      high_conf_fault_rate: Number(high_conf_fault_rate.toFixed(4)),
      overdue_rate: Number(overdue_rate.toFixed(4)),
      rework_rate: Number(rework_rate.toFixed(4)),
      failure_demand_rate: Number(failure_demand_rate.toFixed(4)),
      avoidable_rate: Number(avoidable_rate.toFixed(4)),
      automation_candidate_rate: Number(automation_candidate_rate.toFixed(4)),
      avg_duration_hours: Number(avgDuration.toFixed(2)),
      system_or_process_issue_rate: Number(system_or_process_issue_rate.toFixed(4)),
      team_problem_score: Number(team_problem_score.toFixed(4))
    }
  };
});
```
