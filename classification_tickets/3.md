# 5. Code node №1 — normalize rows

Этот узел:

* чистит поля
* убирает мусор
* приводит пустые значения
* маскирует чувствительные данные
* ограничивает длину текста

Код:

```javascript
function cleanText(value) {
  if (value === null || value === undefined) return '';
  let text = String(value).replace(/\s+/g, ' ').trim();

  // Маскирование email
  text = text.replace(/[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}/gi, '[EMAIL]');

  // Маскирование телефонов РФ и похожих
  text = text.replace(/(\+7|8)\s*\(?\d{3}\)?[\s-]*\d{3}[\s-]*\d{2}[\s-]*\d{2}/g, '[PHONE]');

  // Маскирование длинных числовых идентификаторов/счетов/договоров
  text = text.replace(/\b\d{10,20}\b/g, '[LONG_NUMBER]');

  // Маскирование ФИО вида "Иванов И. И."
  text = text.replace(/\b[А-ЯЁ][а-яё]+\s+[А-ЯЁ]\.\s?[А-ЯЁ]\.\b/g, '[PERSON]');

  return text;
}

function truncateSmart(text, maxLen = 3000) {
  if (!text) return '';
  if (text.length <= maxLen) return text;

  const head = text.slice(0, Math.floor(maxLen * 0.7));
  const tail = text.slice(-Math.floor(maxLen * 0.3));
  return `${head} ... [TRUNCATED] ... ${tail}`;
}

function parseReturnCount(value) {
  if (value === null || value === undefined || value === '') return 0;
  const num = String(value).match(/\d+/);
  return num ? Number(num[0]) : 0;
}

return items.map((item) => {
  const r = item.json;

  const description = truncateSmart(cleanText(r["Подробное описание"]), 3500);
  const resolution = truncateSmart(cleanText(r["Решение"]), 2500);
  const tag = cleanText(r["ТЕГ"]);
  const topic = cleanText(r["Тема ЗНО"]);

  return {
    json: {
      id: cleanText(r["ID"]),
      team: cleanText(r["Команда"]),
      topic,
      tag,
      status: cleanText(r["Статус"]),
      description,
      registration_datetime: cleanText(r["Дата и время регистрации"]),
      deadline_datetime: cleanText(r["Контрольный срок"]),
      executor_group: cleanText(r["Рабочая группа исполнителя"]),
      executor: cleanText(r["Исполнитель (Сотрудник РГ)"]),
      priority: cleanText(r["Приоритет"]),
      service: cleanText(r["Услуга (Объект)"]),
      resolution,
      finish_datetime: cleanText(r["Дата и время окончания работ"]),
      initiator: cleanText(r["Инициатор ЗНО"] || r["Инициатор"]),
      initiator_department: cleanText(r["Подразделение инициатора"]),
      close_code: cleanText(r["Код закрытия"]),
      total_duration_raw: cleanText(r["Общая длительность"]),
      rework_count: parseReturnCount(r["Счетчик возвратов от инициатора ЗНО"]),
      code: cleanText(r["КОД"]),
      work_duration_raw: cleanText(r["Длит-ть работ"]),
      tb: cleanText(r["ТБ"])
    }
  };
});
```

# 6. Code node №2 — add derived metrics

Этот узел добавляет техполя, которые потом пригодятся и модели, и аналитике.

```javascript
function parseDateFlexible(value) {
  if (!value) return null;

  const str = String(value).trim();

  // Форматы вида 19.12.2025 или 14.01.2026 17:46
  const m = str.match(/^(\d{2})\.(\d{2})\.(\d{4})(?:\s+(\d{2}):(\d{2})(?::(\d{2}))?)?$/);
  if (m) {
    const [, dd, mm, yyyy, hh = '00', mi = '00', ss = '00'] = m;
    return new Date(`${yyyy}-${mm}-${dd}T${hh}:${mi}:${ss}`);
  }

  const d = new Date(str);
  return isNaN(d.getTime()) ? null : d;
}

function parseDurationToHours(value) {
  if (!value) return null;
  const str = String(value).trim();

  // Формат "2 19:12:50" = 2 дня 19:12:50
  let m = str.match(/^(\d+)\s+(\d{2}):(\d{2}):(\d{2})$/);
  if (m) {
    const [, days, hh, mm, ss] = m;
    return Number(days) * 24 + Number(hh) + Number(mm) / 60 + Number(ss) / 3600;
  }

  // Формат "00:00:00"
  m = str.match(/^(\d{2}):(\d{2}):(\d{2})$/);
  if (m) {
    const [, hh, mm, ss] = m;
    return Number(hh) + Number(mm) / 60 + Number(ss) / 3600;
  }

  return null;
}

return items.map((item) => {
  const j = item.json;

  const reg = parseDateFlexible(j.registration_datetime);
  const ddl = parseDateFlexible(j.deadline_datetime);
  const fin = parseDateFlexible(j.finish_datetime);

  let overdue_flag = 'unknown';
  let overdue_hours = null;

  if (ddl && fin) {
    const diffMs = fin.getTime() - ddl.getTime();
    overdue_hours = diffMs / (1000 * 60 * 60);
    overdue_flag = diffMs > 0 ? 'yes' : 'no';
  }

  const total_duration_hours = parseDurationToHours(j.total_duration_raw);
  const work_duration_hours = parseDurationToHours(j.work_duration_raw);

  return {
    json: {
      ...j,
      total_duration_hours,
      work_duration_hours,
      overdue_flag,
      overdue_hours,
      rework_flag_from_counter: j.rework_count > 0 ? 'yes' : 'no'
    }
  };
});
```

# 9. System prompt для OpenAI node

Вставь в `System Message`:

```text
Ты аналитик сервисных тикетов Service Manager.

Твоя задача — классифицировать каждый тикет для управленческой аналитики качества процессов, ошибок команд, повторяемости ручных операций и кандидатов на автоматизацию.

Строгие правила:
1. Используй только информацию из входных данных.
2. Не выдумывай факты и не добавляй сведения, которых нет.
3. Если данных недостаточно, ставь "unknown".
4. Ответ возвращай только как JSON-массив без markdown, без пояснений, без префиксов.
5. Для каждого тикета обязательно верни исходный "id".
6. Формулируй "ticket_summary" кратко, по делу, 1 предложение.
7. "problem_theme" и "root_cause" должны быть нормализованы и коротки.
8. "suspected_fault_team" указывай только если это явно или вероятно следует из тикета. Иначе "unknown".
9. "team_fault_flag":
   - yes: есть явные признаки ошибки/недоработки исполнителя или команды
   - no: это обычный запрос, либо источник проблемы на стороне инициатора/системы, а не исполнителя
   - partial: смешанная ответственность
   - unknown: недостаточно данных
10. "responsible_side":
   - initiator: ошибка или некорректные действия инициатора
   - executor_team: проблема на стороне исполняющей команды
   - other_team: проблема у другой внутренней команды
   - system: системный сбой, дефект ПО, баг, автоматизация не сработала
   - vendor: внешняя сторона
   - unknown: недостаточно данных
11. "automation_candidate" = yes, если кейс выглядит повторяемым, ручным, шаблонным и потенциально автоматизируемым.
12. "confidence" — число от 0 до 1.
13. Возвращай только допустимые значения enum.

Допустимые значения:
request_type = incident | service_request | consultation | data_fix | access | process_issue | system_defect | other
root_cause_group = system | process | human_error | access | integration | data_quality | knowledge_gap | unknown
resolution_type = консультация | исправление данных | ручная обработка | настройка | эскалация | исправление бага | предоставление доступа | без изменений | unknown
team_fault_flag = yes | no | partial | unknown
responsible_side = initiator | executor_team | other_team | system | vendor | unknown
rework_flag = yes | no | unknown
user_error_flag = yes | no | unknown
automation_candidate = yes | no | unknown
```

# 12. Code node №4 — parse OpenAI response

Этот узел должен:

* вытащить текст ответа
* распарсить JSON
* проверить обязательные поля
* вернуть по одному item на каждый результат

В n8n структура ответа OpenAI может немного отличаться. Ниже код с мягким fallback.

```javascript
function safeJsonParse(text) {
  try {
    return JSON.parse(text);
  } catch (e) {
    // Попытка вытащить JSON из текста
    const match = text.match(/\[\s*{[\s\S]*}\s*\]/);
    if (match) {
      return JSON.parse(match[0]);
    }
    throw e;
  }
}

function normalizeEnum(value, allowed, fallback = 'unknown') {
  if (!value) return fallback;
  const v = String(value).trim();
  return allowed.includes(v) ? v : fallback;
}

const allowedRequestType = ['incident', 'service_request', 'consultation', 'data_fix', 'access', 'process_issue', 'system_defect', 'other'];
const allowedRootCauseGroup = ['system', 'process', 'human_error', 'access', 'integration', 'data_quality', 'knowledge_gap', 'unknown'];
const allowedResolutionType = ['консультация', 'исправление данных', 'ручная обработка', 'настройка', 'эскалация', 'исправление бага', 'предоставление доступа', 'без изменений', 'unknown'];
const allowedTeamFault = ['yes', 'no', 'partial', 'unknown'];
const allowedResponsible = ['initiator', 'executor_team', 'other_team', 'system', 'vendor', 'unknown'];
const allowedTri = ['yes', 'no', 'unknown'];

const raw =
  $json.message?.content ||
  $json.content ||
  $json.text ||
  $json.output_text ||
  $json.response ||
  '';

const parsed = safeJsonParse(raw);

if (!Array.isArray(parsed)) {
  throw new Error('OpenAI response is not an array');
}

return parsed.map((row) => {
  const confidenceNum = Number(row.confidence);
  return {
    json: {
      id: row.id || '',
      ticket_summary: row.ticket_summary || '',
      request_type: normalizeEnum(row.request_type, allowedRequestType, 'other'),
      business_topic: row.business_topic || 'unknown',
      problem_theme: row.problem_theme || 'unknown',
      root_cause: row.root_cause || 'unknown',
      root_cause_group: normalizeEnum(row.root_cause_group, allowedRootCauseGroup, 'unknown'),
      resolution_type: normalizeEnum(row.resolution_type, allowedResolutionType, 'unknown'),
      team_fault_flag: normalizeEnum(row.team_fault_flag, allowedTeamFault, 'unknown'),
      responsible_side: normalizeEnum(row.responsible_side, allowedResponsible, 'unknown'),
      suspected_fault_team: row.suspected_fault_team || 'unknown',
      rework_flag: normalizeEnum(row.rework_flag, allowedTri, 'unknown'),
      user_error_flag: normalizeEnum(row.user_error_flag, allowedTri, 'unknown'),
      automation_candidate: normalizeEnum(row.automation_candidate, allowedTri, 'unknown'),
      automation_comment: row.automation_comment || '',
      confidence: Number.isFinite(confidenceNum) ? Math.max(0, Math.min(1, confidenceNum)) : 0.0
    }
  };
});
```

# 14. Code node №5 — flatten merged object

Иногда после Merge структура становится вложенной. Этот узел делает плоский JSON.

```javascript
return items.map(item => {
  const j = item.json;

  return {
    json: {
      ...j
    }
  };
});
```

Если Merge даёт разные namespace, можно сделать так:

```javascript
return items.map(item => {
  const source = item.json.input1 || item.json.source || {};
  const ai = item.json.input2 || item.json.enrichment || {};

  return {
    json: {
      ...source,
      ...ai
    }
  };
});
```

# 16. Рекомендуемый fallback на ошибки JSON

В реальной жизни LLM иногда ломает JSON. Поэтому лучше сделать два сценария:

## Вариант попроще

Если parse упал:

* писать batch в error storage
* потом прогонять вручную

## Вариант поумнее

Добавить `If`:

* если parse failed → второй OpenAI вызов с repair prompt

Repair prompt:

### System

```text
Ты исправляешь JSON. На входе дан неидеальный ответ модели. 
Верни только валидный JSON-массив без пояснений.
Не меняй смысл, только исправь формат.
```

### User

```text
Исправь в валидный JSON этот текст:
{{$json.raw_response}}
```

# 20. Code node для агрегации по командам

Если потом захочешь сделать отдельный workflow по аналитике, вот готовый скрипт.

```javascript
function median(arr) {
  if (!arr.length) return null;
  const sorted = [...arr].sort((a, b) => a - b);
  const mid = Math.floor(sorted.length / 2);
  return sorted.length % 2
    ? sorted[mid]
    : (sorted[mid - 1] + sorted[mid]) / 2;
}

const groups = {};

for (const item of items) {
  const r = item.json;
  const team = r.team || 'unknown';

  if (!groups[team]) {
    groups[team] = {
      team,
      tickets_total: 0,
      team_fault_count: 0,
      overdue_count: 0,
      rework_count: 0,
      system_defect_count: 0,
      automation_candidate_count: 0,
      durations: []
    };
  }

  const g = groups[team];
  g.tickets_total += 1;

  if (r.team_fault_flag === 'yes' || r.team_fault_flag === 'partial') g.team_fault_count += 1;
  if (r.overdue_flag === 'yes') g.overdue_count += 1;
  if (r.rework_flag === 'yes' || Number(r.rework_count || 0) > 0) g.rework_count += 1;
  if (r.request_type === 'system_defect' || r.root_cause_group === 'system') g.system_defect_count += 1;
  if (r.automation_candidate === 'yes') g.automation_candidate_count += 1;

  if (typeof r.total_duration_hours === 'number' && !isNaN(r.total_duration_hours)) {
    g.durations.push(r.total_duration_hours);
  }
}

const result = Object.values(groups).map(g => {
  const avgDuration = g.durations.length
    ? g.durations.reduce((a, b) => a + b, 0) / g.durations.length
    : null;

  const faultRate = g.tickets_total ? g.team_fault_count / g.tickets_total : 0;
  const overdueRate = g.tickets_total ? g.overdue_count / g.tickets_total : 0;
  const reworkRate = g.tickets_total ? g.rework_count / g.tickets_total : 0;
  const defectRate = g.tickets_total ? g.system_defect_count / g.tickets_total : 0;

  const normalizedDuration = avgDuration ? Math.min(avgDuration / 72, 1) : 0;

  const teamProblemScore =
    0.35 * faultRate +
    0.20 * reworkRate +
    0.20 * overdueRate +
    0.15 * normalizedDuration +
    0.10 * defectRate;

  return {
    json: {
      team: g.team,
      tickets_total: g.tickets_total,
      team_fault_count: g.team_fault_count,
      team_fault_rate: Number(faultRate.toFixed(4)),
      overdue_count: g.overdue_count,
      overdue_rate: Number(overdueRate.toFixed(4)),
      rework_count: g.rework_count,
      rework_rate: Number(reworkRate.toFixed(4)),
      system_defect_count: g.system_defect_count,
      system_defect_rate: Number(defectRate.toFixed(4)),
      automation_candidate_count: g.automation_candidate_count,
      automation_candidate_rate: Number((g.automation_candidate_count / g.tickets_total).toFixed(4)),
      avg_duration_hours: avgDuration ? Number(avgDuration.toFixed(2)) : null,
      median_duration_hours: median(g.durations),
      team_problem_score: Number(teamProblemScore.toFixed(4))
    }
  };
});

return result.sort((a, b) => b.json.team_problem_score - a.json.team_problem_score);
```

# 22. Что я бы ещё добавил

Очень полезно в prompt включить короткие правила интерпретации:

* если запрос типовой и выполнен без признаков дефекта → `service_request`
* если в тексте ошибка, сбой, не работает, некорректно отображается → `incident` или `system_defect`
* если нужна ручная корректировка данных → часто `data_fix`
* если причина — неправильные действия пользователя → `user_error_flag=yes`, `responsible_side=initiator`
* если тикет связан с правами, доступами, ролями → `access`

Это можно зашить прямо в system prompt.

```text
Подсказки для классификации:
- service_request: стандартный запрос на обслуживание без признаков сбоя

- system_defect: явно прослеживается баг, дефект системы, некорректная логика ПО, сбой интеграции

- consultation: основное действие — объяснение, консультация, разъяснение
- process_issue: проблема вызвана процессом, регламентом, маршрутом, отсутствием шага, ручным процессом
- root_cause_group=knowledge_gap, если обращение вызвано нехваткой знаний или непониманием процедуры
- root_cause_group=data_quality, если проблема связана с неверными, неполными или испорченными данными
```
