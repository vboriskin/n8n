````text
from typing import List, Optional


def quote_list(values: List[str]) -> str:
    return ", ".join([f'"{v}"' for v in values])


def build_project_jql(
    project_key: str,
    start_date: str,
    end_date: str,
    include_issue_types: Optional[List[str]] = None,
    exclude_issue_types: Optional[List[str]] = None,
) -> str:
    parts = [
        f'project = "{project_key}"',
        f'created <= "{end_date}"',
        f'updated >= "{start_date}"',
    ]

    if include_issue_types:
        parts.append(f"issuetype in ({quote_list(include_issue_types)})")

    if exclude_issue_types:
        parts.append(f"issuetype not in ({quote_list(exclude_issue_types)})")

    return " AND ".join(parts)

    path = os.path.join(base_out_dir, "raw", "issues", project_key)

def save_json(payload: Dict[str, Any], file_path: str) -> str:
    with open(file_path, "w", encoding="utf-8") as f:
        json.dump(payload, f, ensure_ascii=False, indent=2)
    return file_path

import logging
from typing import Any, Dict, Optional

import requests
from requests import Response, Session
from tenacity import retry, stop_after_attempt, wait_exponential, retry_if_exception_type

from jira_backlog_audit.jira.auth import build_auth_headers
from jira_backlog_audit.utils.errors import JiraAuthError, JiraRequestError, JiraResponseError

logger = logging.getLogger(__name__)


class JiraClient:
    def __init__(
        self,
        base_url: str,
        token: str,
        timeout_seconds: int = 60,
    ) -> None:
        self.base_url = base_url.rstrip("/")
        self.timeout_seconds = timeout_seconds
        self.session: Session = requests.Session()
        self.session.headers.update(build_auth_headers(token))

    def _url(self, path: str) -> str:
        return f"{self.base_url}{path}"

    @retry(
        stop=stop_after_attempt(4),
        wait=wait_exponential(multiplier=1, min=1, max=10),
        retry=retry_if_exception_type((requests.RequestException, JiraRequestError)),
        reraise=True,
    )
    def get(self, path: str, params: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
        url = self._url(path)
        try:
            response = self.session.get(url, params=params, timeout=self.timeout_seconds)
        except requests.RequestException as exc:
            logger.warning("Network error while requesting Jira: %s", exc)
            raise JiraRequestError(str(exc)) from exc

        self._raise_for_status(response)

        try:
            return response.json()
        except ValueError as exc:
            raise JiraResponseError(f"Invalid JSON from Jira at {url}") from exc

    def _raise_for_status(self, response: Response) -> None:
        if response.status_code in (401, 403):
            raise JiraAuthError(
                f"Jira authentication/authorization failed with status {response.status_code}"
            )

        if response.status_code >= 400:
            text = response.text[:1000] if response.text else ""
            raise JiraRequestError(
                f"Jira request failed with status {response.status_code}: {text}"
            )

import logging
import os
from typing import List

from jira_backlog_audit.jira.client import JiraClient
from jira_backlog_audit.jira.endpoints import API_SEARCH
from jira_backlog_audit.jira.jql import build_project_jql
from jira_backlog_audit.models.config_models import RuntimeConfig
from jira_backlog_audit.models.jira_models import FetchSummary
from jira_backlog_audit.storage.raw_store import raw_issue_dir, raw_metadata_dir, save_json

logger = logging.getLogger(__name__)


class JiraFetcher:
    def __init__(self, client: JiraClient, config: RuntimeConfig) -> None:
        self.client = client
        self.config = config

    def fetch_all_projects(self, page_size: int = 100) -> List[FetchSummary]:
        summaries: List[FetchSummary] = []

        for project_key in self.config.selection.projects:
            summary = self.fetch_project(project_key=project_key, page_size=page_size)
            summaries.append(summary)

        return summaries

    def fetch_project(self, project_key: str, page_size: int = 100) -> FetchSummary:
        logger.info("Fetching project %s", project_key)

        jql = build_project_jql(
            project_key=project_key,
            start_date=self.config.selection.start_date,
            end_date=self.config.selection.end_date,
            include_issue_types=self.config.selection.include_issue_types,
            exclude_issue_types=self.config.selection.exclude_issue_types,
        )

        start_at = 0
        total = None
        batch_files: List[str] = []

        out_dir = self.config.output.out_dir
        project_raw_dir = raw_issue_dir(out_dir, project_key)

        while total is None or start_at < total:
            params = {
                "jql": jql,
                "startAt": start_at,
                "maxResults": page_size,
                "expand": "changelog",
            }

            payload = self.client.get(API_SEARCH, params=params)

            issues = payload.get("issues", [])
            total = payload.get("total", 0)
            max_results = payload.get("maxResults", page_size)

            file_path = os.path.join(
                project_raw_dir,
                f"issues_{project_key}_start_{start_at}.json",
            )
            save_json(payload, file_path)
            batch_files.append(file_path)

            logger.info(
                "Fetched batch for %s: startAt=%s, returned=%s, total=%s",
                project_key,
                start_at,
                len(issues),
                total,
            )

            if not issues:
                break

            start_at += len(issues)

        metadata_dir = raw_metadata_dir(out_dir)
        metadata_path = os.path.join(metadata_dir, f"{project_key}_fetch_summary.json")
        save_json(
            {
                "project_key": project_key,
                "jql": jql,
                "total_issues": total or 0,
                "batch_files": batch_files,
            },
            metadata_path,
        )

        return FetchSummary(
            project_key=project_key,
            total_issues=total or 0,
            batch_files=batch_files,
            warnings=[],
        )

import os
from typing import Optional, List

import yaml

from jira_backlog_audit.models.config_models import (
    JiraConfig,
    SelectionConfig,
    OutputConfig,
    RuntimeConfig,
)
from jira_backlog_audit.utils.errors import ConfigError


def load_yaml_config(path: Optional[str]) -> dict:
    if not path:
        return {}

    with open(path, "r", encoding="utf-8") as f:
        data = yaml.safe_load(f) or {}
        return data


def build_runtime_config(
    base_url: Optional[str],
    token: Optional[str],
    projects: Optional[List[str]],
    start_date: Optional[str],
    end_date: Optional[str],
    config_path: Optional[str],
    out_dir: Optional[str],
) -> RuntimeConfig:
    file_cfg = load_yaml_config(config_path)

    jira_cfg_file = file_cfg.get("jira", {})
    selection_cfg_file = file_cfg.get("selection", {})
    output_cfg_file = file_cfg.get("output", {})

    resolved_token = token or os.getenv("JIRA_TOKEN")
    resolved_base_url = base_url or jira_cfg_file.get("base_url")
    resolved_projects = projects or selection_cfg_file.get("projects", [])
    resolved_start_date = start_date or selection_cfg_file.get("start_date")
    resolved_end_date = end_date or selection_cfg_file.get("end_date")

    if not resolved_base_url:
        raise ConfigError("base_url is required")

    if not resolved_token:
        raise ConfigError("token is required via --token or JIRA_TOKEN")

    if not resolved_projects:
        raise ConfigError("at least one project key is required")

    if not resolved_start_date or not resolved_end_date:
        raise ConfigError("start_date and end_date are required")

    include_issue_types = selection_cfg_file.get("include_issue_types")
    exclude_issue_types = selection_cfg_file.get("exclude_issue_types")

    jira_cfg = JiraConfig(
        base_url=resolved_base_url,
        token=resolved_token,
    )

    selection_cfg = SelectionConfig(
        projects=resolved_projects,
        start_date=resolved_start_date,
        end_date=resolved_end_date,
        include_issue_types=include_issue_types,
        exclude_issue_types=exclude_issue_types,
    )

    output_cfg = OutputConfig(
        out_dir=out_dir or output_cfg_file.get("out_dir", "./output")
    )

    return RuntimeConfig(
        jira=jira_cfg,
        selection=selection_cfg,
        output=output_cfg,
    )

import typer
from typing import List, Optional

from jira_backlog_audit.config import build_runtime_config
from jira_backlog_audit.jira.client import JiraClient
from jira_backlog_audit.jira.fetcher import JiraFetcher
from jira_backlog_audit.logging_setup import setup_logging
from jira_backlog_audit.utils.errors import ConfigError, JiraAuthError, JiraRequestError

app = typer.Typer()


@app.command()
def fetch(
    base_url: str = typer.Option(...),
    token: Optional[str] = typer.Option(None),
    projects: List[str] = typer.Option(...),
    start_date: str = typer.Option(...),
    end_date: str = typer.Option(...),
    config: Optional[str] = typer.Option(None),
    out_dir: Optional[str] = typer.Option(None),
    page_size: int = typer.Option(100),
    log_level: str = "INFO",
):
    setup_logging(log_level)

    try:
        cfg = build_runtime_config(
            base_url=base_url,
            token=token,
            projects=projects,
            start_date=start_date,
            end_date=end_date,
            config_path=config,
            out_dir=out_dir,
        )

        client = JiraClient(
            base_url=cfg.jira.base_url,
            token=cfg.jira.token or "",
        )
        fetcher = JiraFetcher(client=client, config=cfg)
        summaries = fetcher.fetch_all_projects(page_size=page_size)

        typer.echo("[FETCH] Completed")
        for summary in summaries:
            typer.echo(
                f"  - {summary.project_key}: total_issues={summary.total_issues}, batches={len(summary.batch_files)}"
            )

    except (ConfigError, JiraAuthError, JiraRequestError) as exc:
        typer.echo(f"ERROR: {exc}")
        raise typer.Exit(code=1)


@app.command()
def analyze(
    input_dir: str = typer.Option("./output"),
    out_dir: str = typer.Option("./output"),

def report(

    out_dir: str = typer.Option("./output"),

):

@app.command()
def run(
    base_url: str = typer.Option(...),
    token: Optional[str] = typer.Option(None),
    projects: List[str] = typer.Option(...),
    start_date: str = typer.Option(...),
    end_date: str = typer.Option(...),
    config: Optional[str] = typer.Option(None),
    out_dir: Optional[str] = typer.Option(None),
    page_size: int = typer.Option(100),
    log_level: str = "INFO",

    setup_logging(log_level)

    try:

            end_date=end_date,

        typer.echo("[RUN] Starting full pipeline")

        client = JiraClient(

        typer.echo("[RUN] Step 1/3: Fetch")
        summaries = fetcher.fetch_all_projects(page_size=page_size)

        typer.echo("[RUN] Fetch completed")
        for summary in summaries:
            typer.echo(
                f"  - {summary.project_key}: total_issues={summary.total_issues}, batches={len(summary.batch_files)}"
            )

        typer.echo("[RUN] Step 2/3: Analyze")
        typer.echo("Analyze step is not implemented yet.")

        typer.echo("[RUN] Step 3/3: Report")
        typer.echo("Report step is not implemented yet.")

    except (ConfigError, JiraAuthError, JiraRequestError) as exc:
        typer.echo(f"ERROR: {exc}")
        raise typer.Exit(code=1)

CLI инструмент для анализа качества backlog management в Jira Server / Data Center.

## Установка

```bash
pip install -e .
````

## Аутентификация

Передача токена возможна:

1. через CLI:

```bash
--token "<ТОКЕН>"
```

2. через env var:

```bash
export JIRA_TOKEN="<ТОКЕН>"
```

```bash
python -m jira_backlog_audit fetch \
```
