from jira_backlog_audit.utils.errors import ConfigError


def load_yaml_config(path: Optional[str]) -> dict:
    if not path:
        return {}

    with open(path, "r", encoding="utf-8") as f:
        data = yaml.safe_load(f) or {}
        return data


def build_runtime_config(
    base_url: Optional[str],
    token: Optional[str],
    projects: Optional[List[str]],
    start_date: Optional[str],
    end_date: Optional[str],
    config_path: Optional[str],
    out_dir: Optional[str],
) -> RuntimeConfig:
    file_cfg = load_yaml_config(config_path)

    jira_cfg_file = file_cfg.get("jira", {})
    selection_cfg_file = file_cfg.get("selection", {})
    fields_cfg_file = file_cfg.get("fields", {})
    statuses_cfg_file = file_cfg.get("statuses", {})
    thresholds_cfg_file = file_cfg.get("thresholds", {})
    scoring_cfg_file = file_cfg.get("scoring", {})
    output_cfg_file = file_cfg.get("output", {})

    resolved_token = token or os.getenv("JIRA_TOKEN")
    resolved_base_url = base_url or jira_cfg_file.get("base_url")
    resolved_projects = projects or selection_cfg_file.get("projects", [])
    resolved_start_date = start_date or selection_cfg_file.get("start_date")
    resolved_end_date = end_date or selection_cfg_file.get("end_date")

    if not resolved_base_url:
        raise ConfigError("base_url is required")

    if not resolved_token:
        raise ConfigError("token is required via --token or JIRA_TOKEN")

    if not resolved_projects:
        raise ConfigError("at least one project key is required")

    if not resolved_start_date or not resolved_end_date:
        raise ConfigError("start_date and end_date are required")

    jira_cfg = JiraConfig(
        base_url=resolved_base_url,
        token=resolved_token,
    )

    selection_cfg = SelectionConfig(
        projects=resolved_projects,
        start_date=resolved_start_date,
        end_date=resolved_end_date,
        include_issue_types=selection_cfg_file.get("include_issue_types"),
        exclude_issue_types=selection_cfg_file.get("exclude_issue_types"),
    )

    fields_cfg = FieldMappingConfig(
        sprint_field=fields_cfg_file.get("sprint_field"),
        story_points_field=fields_cfg_file.get("story_points_field"),
        epic_link_field=fields_cfg_file.get("epic_link_field"),
        team_field=fields_cfg_file.get("team_field"),
    )

    statuses_cfg = StatusMappingConfig(
        backlog_statuses=statuses_cfg_file.get("backlog_statuses", ["Backlog", "Open", "Selected for Development"]),
        active_statuses=statuses_cfg_file.get("active_statuses", ["In Progress", "Code Review", "Testing", "QA"]),
        done_statuses=statuses_cfg_file.get("done_statuses", ["Done", "Closed", "Resolved"]),
        cancelled_resolutions=statuses_cfg_file.get("cancelled_resolutions", ["Won't Do", "Duplicate", "Cancelled"]),
    )

    thresholds_cfg = ThresholdConfig(
        quick_start_days=thresholds_cfg_file.get("quick_start_days", 3),
        aging_wip_days=thresholds_cfg_file.get("aging_wip_days", [7, 14, 30]),
        long_in_progress_days=thresholds_cfg_file.get("long_in_progress_days", 14),
    )

    weights = scoring_cfg_file.get("weights", {})
    score_weights_cfg = ScoreWeightsConfig(
        priority_stability=weights.get("priority_stability", 0.25),
        unplanned_work=weights.get("unplanned_work", 0.20),
        overload=weights.get("overload", 0.20),
        readiness=weights.get("readiness", 0.15),
        predictability=weights.get("predictability", 0.20),
    )

    output_cfg = OutputConfig(
        out_dir=out_dir or output_cfg_file.get("out_dir", "./output")
    )

    return RuntimeConfig(
        jira=jira_cfg,
        selection=selection_cfg,
        fields=fields_cfg,
        statuses=statuses_cfg,
        thresholds=thresholds_cfg,
        score_weights=score_weights_cfg,
        output=output_cfg,
    )

from typing import List, Dict, Any

import pandas as pd


def compute_priority_stability_metrics(
    issues_df: pd.DataFrame,
    priority_history_df: pd.DataFrame,
    timeline_facts_df: pd.DataFrame,
) -> tuple[pd.DataFrame, pd.DataFrame]:
    rows: List[Dict[str, Any]] = []
    flagged: List[Dict[str, Any]] = []

    if issues_df.empty:
        return pd.DataFrame(), pd.DataFrame()

    issues_count_by_project = issues_df.groupby("project_key")["issue_key"].nunique().to_dict()

    priority_changes_count = (
        priority_history_df.groupby(["project_key", "issue_key"])
        .size()
        .reset_index(name="priority_change_count")
    )

    changed_issue_counts = (
        priority_changes_count.groupby("project_key")["issue_key"]
        .nunique()
        .to_dict()
    )

    avg_changes_per_issue = (
        priority_changes_count.groupby("project_key")["priority_change_count"]
        .mean()
        .to_dict()
    )

    changed_after_start = (
        timeline_facts_df.groupby("project_key")["priority_changed_after_start"]
        .sum()
        .to_dict()
    )

    for project_key, total_issues in issues_count_by_project.items():
        changed_issues = changed_issue_counts.get(project_key, 0)
        avg_changes = avg_changes_per_issue.get(project_key, 0.0)
        changed_after_start_count = changed_after_start.get(project_key, 0)

        rows.extend(
            [
                {
                    "project_key": project_key,
                    "metric_name": "issues_with_priority_change_ratio",
                    "metric_value": changed_issues / total_issues if total_issues else 0.0,
                },
                {
                    "project_key": project_key,
                    "metric_name": "avg_priority_changes_per_changed_issue",
                    "metric_value": float(avg_changes),
                },
                {
                    "project_key": project_key,
                    "metric_name": "priority_changed_after_start_ratio",
                    "metric_value": changed_after_start_count / total_issues if total_issues else 0.0,
                },
                {
                    "project_key": project_key,
                    "metric_name": "priority_changes_per_100_issues",
                    "metric_value": (priority_changes_count[priority_changes_count["project_key"] == project_key]["priority_change_count"].sum() / total_issues * 100.0)
                    if total_issues else 0.0,
                },
            ]
        )

    if not priority_changes_count.empty:
        top_changed = priority_changes_count.sort_values("priority_change_count", ascending=False).head(50)
        for _, row in top_changed.iterrows():
            flagged.append(
                {
                    "issue_key": row["issue_key"],
                    "project_key": row["project_key"],
                    "reason": "high_priority_churn",
                    "value": str(row["priority_change_count"]),
                }
            )

    return pd.DataFrame(rows), pd.DataFrame(flagged)

from typing import List, Dict, Any

import pandas as pd

from jira_backlog_audit.utils.dates import parse_jira_datetime


def _days_between(start_str, end_str):
    start_dt = parse_jira_datetime(start_str)
    end_dt = parse_jira_datetime(end_str)
    if not start_dt or not end_dt:
        return None
    return (end_dt - start_dt).total_seconds() / 86400.0


def compute_unplanned_work_metrics(
    issues_df: pd.DataFrame,
    timeline_facts_df: pd.DataFrame,
    quick_start_days: int,
) -> tuple[pd.DataFrame, pd.DataFrame]:
    if issues_df.empty or timeline_facts_df.empty:
        return pd.DataFrame(), pd.DataFrame()

    merged = issues_df.merge(
        timeline_facts_df[["issue_key", "first_start_date"]],
        on="issue_key",
        how="left",
    )

    merged["days_to_start"] = merged.apply(
        lambda row: _days_between(row["created"], row["first_start_date"]),
        axis=1,
    )
    merged["quick_start_flag"] = merged["days_to_start"].apply(
        lambda x: x is not None and x <= quick_start_days
    )

    rows: List[Dict[str, Any]] = []
    flagged: List[Dict[str, Any]] = []

    for project_key, group in merged.groupby("project_key"):
        total_issues = group["issue_key"].nunique()
        quick_start_count = int(group["quick_start_flag"].sum())
        started_count = int(group["first_start_date"].notna().sum())

        rows.extend(
            [
                {
                    "project_key": project_key,
                    "metric_name": "quick_start_ratio",
                    "metric_value": quick_start_count / total_issues if total_issues else 0.0,
                },
                {
                    "project_key": project_key,
                    "metric_name": "started_issues_ratio",
                    "metric_value": started_count / total_issues if total_issues else 0.0,
                },
                {
                    "project_key": project_key,
                    "metric_name": "avg_days_to_start",
                    "metric_value": float(group["days_to_start"].dropna().mean()) if group["days_to_start"].notna().any() else 0.0,
                },
            ]
        )

        flagged_group = group[group["quick_start_flag"] == True]
        for _, row in flagged_group.head(50).iterrows():
            flagged.append(
                {
                    "issue_key": row["issue_key"],
                    "project_key": row["project_key"],
                    "reason": "quick_start_issue",
                    "value": str(row["days_to_start"]),
                }
            )

    return pd.DataFrame(rows), pd.DataFrame(flagged)

from typing import List, Dict, Any

import pandas as pd

from jira_backlog_audit.utils.dates import parse_jira_datetime


def _week_start_str(dt_series: pd.Series) -> pd.Series:
    parsed = pd.to_datetime(dt_series, utc=True, errors="coerce")
    return (parsed.dt.to_period("W").dt.start_time.dt.strftime("%Y-%m-%d"))


def compute_overload_metrics(
    timeline_facts_df: pd.DataFrame,
    long_in_progress_days: int,
    aging_wip_days: list[int],
) -> tuple[pd.DataFrame, pd.DataFrame, pd.DataFrame]:
    if timeline_facts_df.empty:
        return pd.DataFrame(), pd.DataFrame(), pd.DataFrame()

    rows: List[Dict[str, Any]] = []
    weekly_rows: List[Dict[str, Any]] = []
    flagged: List[Dict[str, Any]] = []

    active_df = timeline_facts_df[timeline_facts_df["entered_active"] == True].copy()

    active_df["start_week"] = _week_start_str(active_df["first_start_date"])
    active_df["cycle_time_days_num"] = pd.to_numeric(active_df["cycle_time_days"], errors="coerce")

    for project_key, group in active_df.groupby("project_key"):
        total_started = group["issue_key"].nunique()

        avg_cycle = float(group["cycle_time_days_num"].dropna().mean()) if group["cycle_time_days_num"].notna().any() else 0.0
        long_in_progress_count = int((group["cycle_time_days_num"] > long_in_progress_days).fillna(False).sum())

        rows.extend(
            [
                {
                    "project_key": project_key,
                    "metric_name": "avg_cycle_time_days_active_only",
                    "metric_value": avg_cycle,
                },
                {
                    "project_key": project_key,
                    "metric_name": "long_in_progress_ratio",
                    "metric_value": long_in_progress_count / total_started if total_started else 0.0,
                },
            ]
        )

        for threshold in aging_wip_days:
            aging_count = int((group["cycle_time_days_num"] > threshold).fillna(False).sum())
            rows.append(
                {
                    "project_key": project_key,
                    "metric_name": f"aging_wip_over_{threshold}_days_ratio",
                    "metric_value": aging_count / total_started if total_started else 0.0,
                }
            )

        weekly_group = (
            group.groupby("start_week")["issue_key"]
            .nunique()
            .reset_index(name="started_issues_count")
        )

        for _, row in weekly_group.iterrows():
            weekly_rows.append(
                {
                    "project_key": project_key,
                    "week_start": row["start_week"],
                    "metric_name": "started_issues_per_week",
                    "metric_value": float(row["started_issues_count"]),
                }
            )

        flagged_group = group[(group["cycle_time_days_num"] > long_in_progress_days).fillna(False)]
        for _, row in flagged_group.head(50).iterrows():
            flagged.append(
                {
                    "issue_key": row["issue_key"],
                    "project_key": row["project_key"],
                    "reason": "long_in_progress",
                    "value": str(row["cycle_time_days_num"]),
                }
            )

    return pd.DataFrame(rows), pd.DataFrame(weekly_rows), pd.DataFrame(flagged)

from typing import List, Dict, Any

import pandas as pd


def compute_readiness_metrics(
    issues_df: pd.DataFrame,
    timeline_facts_df: pd.DataFrame,
) -> tuple[pd.DataFrame, pd.DataFrame]:
    if issues_df.empty or timeline_facts_df.empty:
        return pd.DataFrame(), pd.DataFrame()

    merged = issues_df.merge(
        timeline_facts_df[["issue_key", "entered_active", "returned_to_backlog"]],
        on="issue_key",
        how="left",
    )

    merged["started_without_story_points"] = (
        (merged["entered_active"] == True) &
        (merged["story_points"].isna())
    )

    rows: List[Dict[str, Any]] = []
    flagged: List[Dict[str, Any]] = []

    for project_key, group in merged.groupby("project_key"):
        total_issues = group["issue_key"].nunique()
        started_count = int((group["entered_active"] == True).sum())
        started_without_estimate = int(group["started_without_story_points"].sum())
        returned_to_backlog_count = int((group["returned_to_backlog"] == True).sum())

        rows.extend(
            [
                {
                    "project_key": project_key,
                    "metric_name": "started_without_estimate_ratio",
                    "metric_value": started_without_estimate / started_count if started_count else 0.0,
                },
                {
                    "project_key": project_key,
                    "metric_name": "returned_to_backlog_ratio",
                    "metric_value": returned_to_backlog_count / total_issues if total_issues else 0.0,
                },
            ]
        )

        flagged_group = group[group["started_without_story_points"] == True]
        for _, row in flagged_group.head(50).iterrows():
            flagged.append(
                {
                    "issue_key": row["issue_key"],
                    "project_key": row["project_key"],
                    "reason": "started_without_estimate",
                    "value": None,
                }
            )

        flagged_group = group[group["returned_to_backlog"] == True]

    return pd.DataFrame(rows), pd.DataFrame(flagged)

from typing import List, Dict, Any

import pandas as pd


def compute_predictability_metrics(
    issues_df: pd.DataFrame,
    timeline_facts_df: pd.DataFrame,
    cancelled_resolutions: list[str],
) -> tuple[pd.DataFrame, pd.DataFrame, pd.DataFrame]:
    if issues_df.empty or timeline_facts_df.empty:
        return pd.DataFrame(), pd.DataFrame(), pd.DataFrame()

    merged = issues_df.merge(
        timeline_facts_df[
            [
                "issue_key",
                "first_done_date",
                "lead_time_days",
                "cycle_time_days",
            ]
        ],
        on="issue_key",
        how="left",
    )

    merged["is_cancelled"] = merged["resolution"].isin(cancelled_resolutions)
    merged["done_flag"] = merged["first_done_date"].notna()

    merged["created_dt"] = pd.to_datetime(merged["created"], utc=True, errors="coerce")
    merged["done_week"] = pd.to_datetime(merged["first_done_date"], utc=True, errors="coerce").dt.to_period("W").dt.start_time.dt.strftime("%Y-%m-%d")

    rows: List[Dict[str, Any]] = []
    weekly_rows: List[Dict[str, Any]] = []
    flagged: List[Dict[str, Any]] = []

    for project_key, group in merged.groupby("project_key"):
        total_issues = group["issue_key"].nunique()
        cancelled_count = int(group["is_cancelled"].sum())
        done_count = int(group["done_flag"].sum())

        avg_lead = float(pd.to_numeric(group["lead_time_days"], errors="coerce").dropna().mean()) if pd.to_numeric(group["lead_time_days"], errors="coerce").notna().any() else 0.0
        avg_cycle = float(pd.to_numeric(group["cycle_time_days"], errors="coerce").dropna().mean()) if pd.to_numeric(group["cycle_time_days"], errors="coerce").notna().any() else 0.0

        rows.extend(
            [
                {
                    "project_key": project_key,
                    "metric_name": "cancelled_ratio",
                    "metric_value": cancelled_count / total_issues if total_issues else 0.0,
                },
                {
                    "project_key": project_key,
                    "metric_name": "done_ratio",
                    "metric_value": done_count / total_issues if total_issues else 0.0,
                },
                {
                    "project_key": project_key,
                    "metric_name": "avg_lead_time_days",
                    "metric_value": avg_lead,
                },
                {
                    "project_key": project_key,
                    "metric_name": "avg_cycle_time_days",
                    "metric_value": avg_cycle,
                },
            ]
        )

        weekly_done = group[group["done_week"].notna()].groupby("done_week")["issue_key"].nunique().reset_index(name="done_count")
        for _, row in weekly_done.iterrows():
            weekly_rows.append(
                {
                    "project_key": project_key,
                    "week_start": row["done_week"],
                    "metric_name": "done_issues_per_week",
                    "metric_value": float(row["done_count"]),
                }
            )

        flagged_group = group[group["is_cancelled"] == True]
        for _, row in flagged_group.head(50).iterrows():
            flagged.append(
                {
                    "issue_key": row["issue_key"],
                    "project_key": row["project_key"],
                    "reason": "cancelled_issue",
                    "value": str(row["resolution"]),
                }
            )

    return pd.DataFrame(rows), pd.DataFrame(weekly_rows), pd.DataFrame(flagged)

import pandas as pd


def _clip_score(value: float) -> float:
    return max(0.0, min(100.0, value))


def _score_from_ratio_bad(ratio: float) -> float:
    return _clip_score(100.0 * (1.0 - ratio))


def _score_from_days_bad(days: float, soft_limit: float) -> float:
    if soft_limit <= 0:
        return 100.0
    return _clip_score(100.0 * max(0.0, 1.0 - (days / (soft_limit * 2.0))))


def compute_scorecard(
    metrics_by_project_df: pd.DataFrame,
    weights,
) -> pd.DataFrame:
    if metrics_by_project_df.empty:
        return pd.DataFrame()

    pivot = metrics_by_project_df.pivot_table(
        index="project_key",
        columns="metric_name",
        values="metric_value",
        aggfunc="first",
    ).reset_index()

    for col in [
        "issues_with_priority_change_ratio",
        "priority_changed_after_start_ratio",
        "quick_start_ratio",
        "long_in_progress_ratio",
        "started_without_estimate_ratio",
        "returned_to_backlog_ratio",
        "cancelled_ratio",
    ]:
        if col not in pivot.columns:
            pivot[col] = 0.0

    if "avg_cycle_time_days" not in pivot.columns:
        pivot["avg_cycle_time_days"] = 0.0

    if "avg_lead_time_days" not in pivot.columns:
        pivot["avg_lead_time_days"] = 0.0

    pivot["priority_stability_score"] = (
        _score_from_ratio_bad(pivot["issues_with_priority_change_ratio"]) * 0.5
        + _score_from_ratio_bad(pivot["priority_changed_after_start_ratio"]) * 0.5
    )

    pivot["unplanned_work_score"] = _score_from_ratio_bad(pivot["quick_start_ratio"])

    pivot["overload_score"] = (
        _score_from_ratio_bad(pivot["long_in_progress_ratio"]) * 0.6
        + pivot["avg_cycle_time_days"].apply(lambda x: _score_from_days_bad(float(x), 14.0)) * 0.4
    )

    pivot["readiness_score"] = (
        _score_from_ratio_bad(pivot["started_without_estimate_ratio"]) * 0.5
        + _score_from_ratio_bad(pivot["returned_to_backlog_ratio"]) * 0.5
    )

    pivot["predictability_score"] = (
        _score_from_ratio_bad(pivot["cancelled_ratio"]) * 0.4
        + pivot["avg_lead_time_days"].apply(lambda x: _score_from_days_bad(float(x), 30.0)) * 0.3
        + pivot["avg_cycle_time_days"].apply(lambda x: _score_from_days_bad(float(x), 14.0)) * 0.3
    )

    pivot["total_health_score"] = (
        pivot["priority_stability_score"] * weights.priority_stability
        + pivot["unplanned_work_score"] * weights.unplanned_work
        + pivot["overload_score"] * weights.overload
        + pivot["readiness_score"] * weights.readiness
        + pivot["predictability_score"] * weights.predictability
    )

    pivot["total_risk_score"] = 100.0 - pivot["total_health_score"]

    return pivot[
        [
            "project_key",
            "priority_stability_score",
            "unplanned_work_score",
            "overload_score",
            "readiness_score",
            "predictability_score",
            "total_health_score",
            "total_risk_score",
        ]
    ].copy()

from datetime import datetime
from typing import Optional


JIRA_DATETIME_FORMATS = [
    "%Y-%m-%dT%H:%M:%S.%f%z",
    "%Y-%m-%dT%H:%M:%S%z",
]


def parse_jira_datetime(value: Optional[str]) -> Optional[datetime]:
    if not value:
        return None

    for fmt in JIRA_DATETIME_FORMATS:
        try:
            return datetime.strptime(value, fmt)
        except ValueError:
            continue

    return None

def _load_normalized_csvs(input_dir: str):
    base = os.path.join(input_dir, "normalized")
    issues_df = pd.read_csv(os.path.join(base, "issues.csv"))
    changelog_df = pd.read_csv(os.path.join(base, "issue_changelog.csv"))

    priority_history_df = pd.read_csv(os.path.join(base, "issue_priority_history.csv"))

    return issues_df, changelog_df, status_history_df, priority_history_df, timeline_facts_df


def _run_metrics(cfg, issues_df, priority_history_df, timeline_facts_df):
    priority_metrics_df, priority_flagged_df = compute_priority_stability_metrics(
        issues_df=issues_df,
        priority_history_df=priority_history_df,
        timeline_facts_df=timeline_facts_df,
    )

    unplanned_metrics_df, unplanned_flagged_df = compute_unplanned_work_metrics(
        issues_df=issues_df,
        timeline_facts_df=timeline_facts_df,
        quick_start_days=cfg.thresholds.quick_start_days,
    )

    overload_metrics_df, overload_weekly_df, overload_flagged_df = compute_overload_metrics(
        timeline_facts_df=timeline_facts_df,
        long_in_progress_days=cfg.thresholds.long_in_progress_days,
        aging_wip_days=cfg.thresholds.aging_wip_days,
    )

    readiness_metrics_df, readiness_flagged_df = compute_readiness_metrics(
        issues_df=issues_df,
        timeline_facts_df=timeline_facts_df,
    )

    predictability_metrics_df, predictability_weekly_df, predictability_flagged_df = compute_predictability_metrics(
        issues_df=issues_df,
        timeline_facts_df=timeline_facts_df,
        cancelled_resolutions=cfg.statuses.cancelled_resolutions,
    )

    metrics_by_project_df = pd.concat(
        [
            priority_metrics_df,
            unplanned_metrics_df,
            overload_metrics_df,
            readiness_metrics_df,
            predictability_metrics_df,
        ],
        ignore_index=True,
    )

    metrics_by_week_df = pd.concat(
        [
            overload_weekly_df,
            predictability_weekly_df,
        ],
        ignore_index=True,
    )

    flagged_issues_df = pd.concat(
        [
            priority_flagged_df,
            unplanned_flagged_df,
            overload_flagged_df,
            readiness_flagged_df,
            predictability_flagged_df,
        ],
        ignore_index=True,
    ).drop_duplicates()

    scorecard_df = compute_scorecard(
        metrics_by_project_df=metrics_by_project_df,
        weights=cfg.score_weights,
    )

    save_metrics_bundle(
        out_dir=cfg.output.out_dir,
        metrics_by_project_df=metrics_by_project_df,
        metrics_by_week_df=metrics_by_week_df,
        scorecard_df=scorecard_df,
        flagged_issues_df=flagged_issues_df,
    )

    return metrics_by_project_df, metrics_by_week_df, scorecard_df, flagged_issues_df
