import os

import yaml

def load_yaml_config(path: Optional[str]) -> dict:
    if not path:
        return {}

    with open(path, "r", encoding="utf-8") as f:
        data = yaml.safe_load(f) or {}
        return data


def build_runtime_config(
    base_url: Optional[str],
    token: Optional[str],
    projects: Optional[List[str]],
    start_date: Optional[str],
    end_date: Optional[str],
    config_path: Optional[str],
    out_dir: Optional[str],
) -> RuntimeConfig:
    file_cfg = load_yaml_config(config_path)

    jira_cfg_file = file_cfg.get("jira", {})
    selection_cfg_file = file_cfg.get("selection", {})
    fields_cfg_file = file_cfg.get("fields", {})
    statuses_cfg_file = file_cfg.get("statuses", {})
    output_cfg_file = file_cfg.get("output", {})

    resolved_token = token or os.getenv("JIRA_TOKEN")
    resolved_base_url = base_url or jira_cfg_file.get("base_url")
    resolved_projects = projects or selection_cfg_file.get("projects", [])
    resolved_start_date = start_date or selection_cfg_file.get("start_date")
    resolved_end_date = end_date or selection_cfg_file.get("end_date")

    if not resolved_base_url:
        raise ConfigError("base_url is required")

    if not resolved_token:
        raise ConfigError("token is required via --token or JIRA_TOKEN")

    if not resolved_projects:
        raise ConfigError("at least one project key is required")

    if not resolved_start_date or not resolved_end_date:
        raise ConfigError("start_date and end_date are required")

    jira_cfg = JiraConfig(
        base_url=resolved_base_url,
        token=resolved_token,
    )

    selection_cfg = SelectionConfig(
        projects=resolved_projects,
        start_date=resolved_start_date,
        end_date=resolved_end_date,
        include_issue_types=selection_cfg_file.get("include_issue_types"),
        exclude_issue_types=selection_cfg_file.get("exclude_issue_types"),
    )

    fields_cfg = FieldMappingConfig(
        sprint_field=fields_cfg_file.get("sprint_field"),
        story_points_field=fields_cfg_file.get("story_points_field"),
        epic_link_field=fields_cfg_file.get("epic_link_field"),
        team_field=fields_cfg_file.get("team_field"),
    )

    statuses_cfg = StatusMappingConfig(
        backlog_statuses=statuses_cfg_file.get("backlog_statuses", ["Backlog", "Open", "Selected for Development"]),
        active_statuses=statuses_cfg_file.get("active_statuses", ["In Progress", "Code Review", "Testing", "QA"]),
        done_statuses=statuses_cfg_file.get("done_statuses", ["Done", "Closed", "Resolved"]),
        cancelled_resolutions=statuses_cfg_file.get("cancelled_resolutions", ["Won't Do", "Duplicate", "Cancelled"]),
    )

    output_cfg = OutputConfig(
        out_dir=out_dir or output_cfg_file.get("out_dir", "./output")
    )

    return RuntimeConfig(
        jira=jira_cfg,
        selection=selection_cfg,
        fields=fields_cfg,
        statuses=statuses_cfg,
        output=output_cfg,
    )

from datetime import datetime
from typing import Optional


JIRA_DATETIME_FORMATS = [
    "%Y-%m-%dT%H:%M:%S.%f%z",
    "%Y-%m-%dT%H:%M:%S%z",
]


def parse_jira_datetime(value: Optional[str]) -> Optional[datetime]:
    if not value:
        return None

    for fmt in JIRA_DATETIME_FORMATS:
        try:
            return datetime.strptime(value, fmt)
        except ValueError:
            continue

    return None

import glob
import json
import logging
import os
from typing import Any, Dict, List, Optional

import pandas as pd

from jira_backlog_audit.models.config_models import RuntimeConfig
from jira_backlog_audit.storage.normalized_store import normalized_dir, save_dataframe

logger = logging.getLogger(__name__)


def _safe_get_name(obj: Optional[Dict[str, Any]]) -> Optional[str]:
    if not obj:
        return None
    return obj.get("name")


def _safe_get_key(obj: Optional[Dict[str, Any]]) -> Optional[str]:
    if not obj:
        return None
    return obj.get("key")


def _safe_get_list_names(items: Optional[List[Dict[str, Any]]]) -> List[str]:
    if not items:
        return []
    return [item.get("name") for item in items if item.get("name")]


def _extract_custom_field(fields: Dict[str, Any], field_name: Optional[str]) -> Any:
    if not field_name:
        return None
    return fields.get(field_name)


def normalize_issues(raw_input_dir: str, config: RuntimeConfig) -> pd.DataFrame:
    pattern = os.path.join(raw_input_dir, "raw", "issues", "*", "*.json")
    files = sorted(glob.glob(pattern))

    rows: List[Dict[str, Any]] = []

    for file_path in files:
        with open(file_path, "r", encoding="utf-8") as f:
            payload = json.load(f)

        for issue in payload.get("issues", []):
            fields = issue.get("fields", {})

            project = fields.get("project") or {}
            issue_type = fields.get("issuetype") or {}
            status = fields.get("status") or {}
            priority = fields.get("priority") or {}
            resolution = fields.get("resolution") or {}
            assignee = fields.get("assignee") or {}
            reporter = fields.get("reporter") or {}

            row = {
                "issue_id": issue.get("id"),
                "issue_key": issue.get("key"),
                "project_key": _safe_get_key(project),
                "project_name": _safe_get_name(project),
                "summary": fields.get("summary"),
                "issue_type": _safe_get_name(issue_type),
                "status": _safe_get_name(status),
                "priority": _safe_get_name(priority),
                "resolution": _safe_get_name(resolution),
                "created": fields.get("created"),
                "updated": fields.get("updated"),
                "assignee": assignee.get("displayName") if assignee else None,
                "assignee_name": assignee.get("name") if assignee else None,
                "reporter": reporter.get("displayName") if reporter else None,
                "reporter_name": reporter.get("name") if reporter else None,
                "labels": "|".join(fields.get("labels", [])),
                "components": "|".join(_safe_get_list_names(fields.get("components"))),
                "fix_versions": "|".join(_safe_get_list_names(fields.get("fixVersions"))),
                "story_points": _extract_custom_field(fields, config.fields.story_points_field),
                "sprint_raw": _extract_custom_field(fields, config.fields.sprint_field),
                "epic_link": _extract_custom_field(fields, config.fields.epic_link_field),
                "team": _extract_custom_field(fields, config.fields.team_field),
            }
            rows.append(row)

    df = pd.DataFrame(rows)
    logger.info("Normalized issues: %s rows", len(df))
    return df


def save_normalized_issues(df: pd.DataFrame, out_dir: str) -> None:
    base_dir = normalized_dir(out_dir)
    save_dataframe(df, os.path.join(base_dir, "issues"), save_parquet=False)

import glob
import json
import logging
import os
from typing import Any, Dict, List, Optional

import pandas as pd

from jira_backlog_audit.storage.normalized_store import normalized_dir, save_dataframe

logger = logging.getLogger(__name__)


TRACKED_FIELDS = {
    "status": "status",
    "priority": "priority",
    "assignee": "assignee",
    "Sprint": "sprint",
    "Story Points": "story_points",
    "Original Estimate": "original_estimate",
}


def _extract_histories(issue: Dict[str, Any]) -> List[Dict[str, Any]]:
    changelog = issue.get("changelog") or {}
    return changelog.get("histories", [])


def parse_changelog(raw_input_dir: str) -> pd.DataFrame:
    pattern = os.path.join(raw_input_dir, "raw", "issues", "*", "*.json")
    files = sorted(glob.glob(pattern))

    rows: List[Dict[str, Any]] = []

    for file_path in files:
        with open(file_path, "r", encoding="utf-8") as f:
            payload = json.load(f)

        for issue in payload.get("issues", []):
            issue_id = issue.get("id")
            issue_key = issue.get("key")
            project_key = ((issue.get("fields") or {}).get("project") or {}).get("key")

            for history in _extract_histories(issue):
                history_id = history.get("id")
                author = history.get("author") or {}
                created = history.get("created")

                for item in history.get("items", []):
                    field_name = item.get("field")
                    normalized_field = TRACKED_FIELDS.get(field_name, field_name)

                    rows.append(
                        {
                            "issue_id": issue_id,
                            "issue_key": issue_key,
                            "project_key": project_key,
                            "history_id": history_id,
                            "history_created": created,
                            "author_display_name": author.get("displayName"),
                            "author_name": author.get("name"),
                            "field": field_name,
                            "normalized_field": normalized_field,
                            "from": item.get("from"),
                            "from_string": item.get("fromString"),
                            "to": item.get("to"),
                            "to_string": item.get("toString"),
                        }
                    )

    df = pd.DataFrame(rows)
    logger.info("Parsed changelog events: %s rows", len(df))
    return df


def build_status_history(changelog_df: pd.DataFrame) -> pd.DataFrame:
    if changelog_df.empty:
        return pd.DataFrame(
            columns=[
                "issue_id",
                "issue_key",
                "project_key",
                "changed_at",
                "from_status",
                "to_status",
            ]
        )

    df = changelog_df[changelog_df["normalized_field"] == "status"].copy()
    df = df.rename(
        columns={
            "history_created": "changed_at",
            "from_string": "from_status",
            "to_string": "to_status",
        }
    )

    return df[
        [
            "issue_id",
            "issue_key",
            "project_key",
            "changed_at",
            "from_status",
            "to_status",
        ]
    ].sort_values(["issue_key", "changed_at"])


def build_priority_history(changelog_df: pd.DataFrame) -> pd.DataFrame:

                "issue_key",

import logging
from typing import Any, Dict, List, Optional

import pandas as pd

from jira_backlog_audit.models.config_models import RuntimeConfig
from jira_backlog_audit.utils.dates import parse_jira_datetime

logger = logging.getLogger(__name__)


def _first_changed_to_status(status_history_df: pd.DataFrame, issue_key: str, target_statuses: List[str]) -> Optional[str]:
    issue_history = status_history_df[status_history_df["issue_key"] == issue_key]
    issue_history = issue_history[issue_history["to_status"].isin(target_statuses)].sort_values("changed_at")

    if issue_history.empty:
        return None

    return issue_history.iloc[0]["changed_at"]


def _has_return_to_backlog(
    status_history_df: pd.DataFrame,
    issue_key: str,
    active_statuses: List[str],
    backlog_statuses: List[str],
) -> bool:
    issue_history = status_history_df[status_history_df["issue_key"] == issue_key].sort_values("changed_at")

    seen_active = False
    for _, row in issue_history.iterrows():
        to_status = row.get("to_status")
        if to_status in active_statuses:
            seen_active = True
        elif seen_active and to_status in backlog_statuses:
            return True

    return False


def _priority_changed_after_start(
    priority_history_df: pd.DataFrame,
    issue_key: str,
    start_dt_str: Optional[str],
) -> bool:
    if not start_dt_str:
        return False

    start_dt = parse_jira_datetime(start_dt_str)
    if not start_dt:
        return False

    issue_priority = priority_history_df[priority_history_df["issue_key"] == issue_key].copy()
    if issue_priority.empty:
        return False

    for _, row in issue_priority.iterrows():
        changed_at = parse_jira_datetime(row.get("changed_at"))
        if changed_at and changed_at > start_dt:
            return True

    return False


def _compute_day_diff(start_str: Optional[str], end_str: Optional[str]) -> Optional[float]:
    start_dt = parse_jira_datetime(start_str)
    end_dt = parse_jira_datetime(end_str)
    if not start_dt or not end_dt:
        return None
    return (end_dt - start_dt).total_seconds() / 86400.0


def build_issue_timeline_facts(
    issues_df: pd.DataFrame,
    status_history_df: pd.DataFrame,
    priority_history_df: pd.DataFrame,
    config: RuntimeConfig,
) -> pd.DataFrame:
    if issues_df.empty:
        return pd.DataFrame(
            columns=[
                "issue_key",
                "project_key",
                "created",
                "current_status",
                "first_start_date",
                "first_done_date",
                "entered_active",
                "returned_to_backlog",
                "priority_changed_after_start",
                "lead_time_days",
                "cycle_time_days",
            ]
        )

    rows: List[Dict[str, Any]] = []

    for _, issue in issues_df.iterrows():
        issue_key = issue["issue_key"]

        first_start_date = _first_changed_to_status(
            status_history_df=status_history_df,
            issue_key=issue_key,
            target_statuses=config.statuses.active_statuses,
        )

        first_done_date = _first_changed_to_status(
            status_history_df=status_history_df,
            issue_key=issue_key,
            target_statuses=config.statuses.done_statuses,
        )

        returned_to_backlog = _has_return_to_backlog(
            status_history_df=status_history_df,
            issue_key=issue_key,
            active_statuses=config.statuses.active_statuses,
            backlog_statuses=config.statuses.backlog_statuses,
        )

        priority_changed_after_start = _priority_changed_after_start(
            priority_history_df=priority_history_df,
            issue_key=issue_key,
            start_dt_str=first_start_date,
        )

        lead_time_days = _compute_day_diff(issue["created"], first_done_date)
        cycle_time_days = _compute_day_diff(first_start_date, first_done_date)

        rows.append(
            {
                "issue_id": issue["issue_id"],
                "issue_key": issue_key,
                "project_key": issue["project_key"],
                "created": issue["created"],
                "current_status": issue["status"],
                "first_start_date": first_start_date,
                "first_done_date": first_done_date,
                "entered_active": first_start_date is not None,
                "returned_to_backlog": returned_to_backlog,
                "priority_changed_after_start": priority_changed_after_start,
                "lead_time_days": lead_time_days,
                "cycle_time_days": cycle_time_days,
            }
        )

    df = pd.DataFrame(rows)
    logger.info("Built issue timeline facts: %s rows", len(df))
    return df
