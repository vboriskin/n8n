import os
import yaml
from typing import Optional, List

from jira_backlog_audit.models.config_models import (
    JiraConfig,
    SelectionConfig,
    OutputConfig,
    RuntimeConfig,
)


def load_yaml_config(path: Optional[str]) -> dict:
    if not path:
        return {}
    with open(path, "r") as f:
        return yaml.safe_load(f)


def build_runtime_config(
    base_url: str,
    token: Optional[str],
    projects: List[str],
    start_date: str,
    end_date: str,
    config_path: Optional[str],
    out_dir: Optional[str],
) -> RuntimeConfig:

    file_cfg = load_yaml_config(config_path)

    token = token or os.getenv("JIRA_TOKEN")

    jira_cfg = JiraConfig(
        base_url=base_url or file_cfg.get("jira", {}).get("base_url"),
        token=token,
    )

    selection_cfg = SelectionConfig(
        projects=projects or file_cfg.get("selection", {}).get("projects", []),
        start_date=start_date or file_cfg.get("selection", {}).get("start_date"),
        end_date=end_date or file_cfg.get("selection", {}).get("end_date"),
    )

    output_cfg = OutputConfig(
        out_dir=out_dir or file_cfg.get("output", {}).get("out_dir", "./output")
    )

    return RuntimeConfig(
        jira=jira_cfg,
        selection=selection_cfg,
        output=output_cfg,
    )

import typer
from typing import List, Optional

from jira_backlog_audit.config import build_runtime_config
from jira_backlog_audit.logging_setup import setup_logging

app = typer.Typer()


@app.command()
def fetch(
    base_url: str = typer.Option(...),
    token: Optional[str] = typer.Option(None),
    projects: List[str] = typer.Option(...),
    start_date: str = typer.Option(...),
    end_date: str = typer.Option(...),
    config: Optional[str] = typer.Option(None),
    out_dir: Optional[str] = typer.Option(None),
    log_level: str = "INFO",
):
    setup_logging(log_level)

    cfg = build_runtime_config(
        base_url, token, projects, start_date, end_date, config, out_dir
    )

    typer.echo(f"[FETCH] Projects: {cfg.selection.projects}")
    typer.echo("Fetch step is not implemented yet.")


@app.command()
def analyze(
    input_dir: str = typer.Option("./output"),
    out_dir: str = typer.Option("./output"),
    log_level: str = "INFO",
):
    setup_logging(log_level)
    typer.echo("[ANALYZE] Not implemented yet.")


@app.command()
def report(
    input_dir: str = typer.Option("./output"),
    out_dir: str = typer.Option("./output"),
    log_level: str = "INFO",
):
    setup_logging(log_level)
    typer.echo("[REPORT] Not implemented yet.")


@app.command()
def run(
    base_url: str = typer.Option(...),
    token: Optional[str] = typer.Option(None),
    projects: List[str] = typer.Option(...),
    start_date: str = typer.Option(...),
    end_date: str = typer.Option(...),
    config: Optional[str] = typer.Option(None),
    out_dir: Optional[str] = typer.Option(None),
    log_level: str = "INFO",
):
    setup_logging(log_level)

    cfg = build_runtime_config(
        base_url, token, projects, start_date, end_date, config, out_dir
    )

    typer.echo("[RUN] Starting full pipeline")
    typer.echo(f"Projects: {cfg.selection.projects}")

    typer.echo("Step 1: Fetch")
    typer.echo("Step 2: Analyze")
    typer.echo("Step 3: Report")

    typer.echo("Pipeline not implemented yet.")
