Да. Ниже даю **вторую пачку n8n Code nodes** — уже для режима `matrix_only` и совместимого mixed-режима.

Они решают 3 задачи:

1. **Build Competency Payloads** — умеет работать без Jira/self-assessment
2. **Build IDP Payload** — строит payload для IDP из матрицы, gaps и learning assets
3. **Build One2One Payloads** — отдельно для сотрудника и руководителя

Я сделаю так, чтобы они работали и в новых данных, и не ломали старый flow.

# 1. `Code / Build Competency Payloads v2`

Mode: `Run Once for All Items`

Эта нода:

* берет `competency_catalog.competencies`
* приклеивает self-assessment
* приклеивает synthetic evidence
* приклеивает internal learning assets
* считает priority score
* умеет работать в `matrix_only`, `matrix_plus_issues`, `full`

```javascript
function safeArray(value) {
  return Array.isArray(value) ? value : [];
}

function normalizeText(value) {
  return String(value || '')
    .trim()
    .toLowerCase()
    .replace(/[ё]/g, 'е');
}

function scorePriority({ type, hasSelf, hasEvidence, coverage, levelCount }) {
  let score = 0;

  if (type === 'hard') score += 3;
  if (type === 'soft') score += 2;

  if (!hasSelf) score += 2;
  if (!hasEvidence) score += 3;

  if (coverage === 'low') score += 3;
  if (coverage === 'medium') score += 1;

  if (levelCount >= 4) score += 1;

  return score;
}

const employee = $json.employee || {};
const workflowConfig = $json.workflow_config || {};
const competencyCatalog = $json.competency_catalog || {};
const competencies = safeArray(competencyCatalog.competencies);

const selfAssessments = safeArray($json.assessment_input?.self_assessments);
const normalizedEvidence = safeArray($json.normalized?.normalized_evidence);
const syntheticEvidence = safeArray($json.normalized?.synthetic_evidence);
const coverageMap = safeArray($json.analysis?.coverage_map);
const learningByCompetency = $json.learning_catalog?.by_competency || {};

const assessmentMode = workflowConfig.assessment_mode || 'full';

const selfMap = {};
for (const sa of selfAssessments) {
  if (!sa?.competency_id) continue;
  selfMap[sa.competency_id] = sa;
}

const evidenceMap = {};
for (const ev of normalizedEvidence) {
  const key = ev?.competency_id;
  if (!key) continue;
  if (!evidenceMap[key]) evidenceMap[key] = [];
  evidenceMap[key].push(ev);
}

const syntheticMap = {};
for (const ev of syntheticEvidence) {
  const key = ev?.competency_id;
  if (!key) continue;
  if (!syntheticMap[key]) syntheticMap[key] = [];
  syntheticMap[key].push(ev);
}

const coverageByCompetency = {};
for (const c of coverageMap) {
  if (!c?.competency_id) continue;
  coverageByCompetency[c.competency_id] = c;
}

const itemsOut = [];

for (const comp of competencies) {
  const competencyId = comp.competency_id;
  const self = selfMap[competencyId] || null;
  const evidence = evidenceMap[competencyId] || [];
  const synthetic = syntheticMap[competencyId] || [];
  const coverage = coverageByCompetency[competencyId] || null;
  const internalLearning = safeArray(learningByCompetency[competencyId]).slice(0, 5);

  const hasSelf = !!self && self.self_level !== null && self.self_level !== undefined;
  const hasEvidence = evidence.length > 0;
  const coverageLevel = coverage?.coverage || (hasEvidence ? 'medium' : 'low');

  const priorityScore = scorePriority({
    type: comp.type,
    hasSelf,
    hasEvidence,
    coverage: coverageLevel,
    levelCount: safeArray(comp.levels).length
  });

  itemsOut.push({
    json: {
      competency_id: competencyId,
      payload: {
        meta: {
          assessment_mode: assessmentMode,
          performance_inference_allowed: !!workflowConfig.performance_inference_allowed,
          development_inference_allowed: !!workflowConfig.development_inference_allowed,
          source: $json.scoring_input?.source || 'unknown'
        },

        employee: {
          employee_id: employee.employee_id,
          full_name: employee.full_name,
          role_id: employee.role_id,
          grade_id: employee.grade_id,
          position: employee.position || null,
          team: employee.team || null
        },

        competency: {
          competency_id: comp.competency_id,
          competency_name: comp.competency_name,
          group_name: comp.group_name,
          type: comp.type || 'unknown',
          role_id: comp.role_id || competencyCatalog.role_id || null,
          levels: safeArray(comp.levels)
        },

        self_assessment: self
          ? {
              self_level: self.self_level ?? null,
              comment: self.comment ?? null,
              source: self.source || 'provided'
            }
          : null,

        evidence: {
          factual: evidence,
          synthetic,
          has_factual_evidence: hasEvidence,
          has_only_synthetic_evidence: !hasEvidence && synthetic.length > 0,
          coverage: coverageLevel
        },

        learning: {
          internal_assets: internalLearning.map(a => ({
            id: a.id || null,
            title: a.title || null,
            matched_competencies_count: a.matched_competencies_count || 0
          }))
        },

        guidance: {
          expected_behavior_from_matrix: safeArray(comp.levels).map(l => ({
            level: l.level,
            description: l.description
          })),
          matrix_only_mode: assessmentMode === 'matrix_only'
        }
      },

      priority_score: priorityScore,
      has_self_assessment: hasSelf,
      has_factual_evidence: hasEvidence,
      coverage: coverageLevel,
      assessment_mode: assessmentMode
    }
  });
}

return itemsOut;
```

# 3. `Code / Build IDP Payload v2`

Mode: `Run Once for All Items`

Эта нода работает после `Code / Normalize Competency Result`
Она собирает:

* competency results
* matrix levels
* learning assets
* employee context
* assessment mode

```javascript
function safeArray(v) {
  return Array.isArray(v) ? v : [];
}

const competencyResults = items.map(i => i.json);
const root = $item(0).$node["Code / Build Unified Root Contract"].json;

const competencies = safeArray(root.competency_catalog?.competencies);
const competencyById = {};
for (const c of competencies) {
  competencyById[c.competency_id] = c;
}

const learningByCompetency = root.learning_catalog?.by_competency || {};
const assessmentMode = root.workflow_config?.assessment_mode || 'full';

const idpInput = competencyResults.map(r => {
  const comp = competencyById[r.competency_id] || {};
  const selfLevel = r.self_level ?? null;
  const estimatedLevel = r.estimated_level ?? null;

  let gap = null;
  if (
    selfLevel !== null &&
    selfLevel !== undefined &&
    estimatedLevel !== null &&
    estimatedLevel !== undefined
  ) {
    gap = estimatedLevel - selfLevel;
  }

  return {
    competency_id: r.competency_id,
    competency_name: comp.competency_name || r.competency_id,
    group_name: comp.group_name || null,
    type: comp.type || 'unknown',

    self_level: selfLevel,
    estimated_level: estimatedLevel,
    confidence: r.confidence || 'low',
    priority: r.priority || 'medium',
    assessment: r.assessment || '',
    development_focus: r.development_focus || '',
    evidence_missing: safeArray(r.evidence_missing),
    risk_flags: safeArray(r.risk_flags),

    matrix_levels: safeArray(comp.levels),
    internal_learning_assets: safeArray(learningByCompetency[r.competency_id]).slice(0, 5),

    inferred_gap: gap
  };
});

return [{
  json: {
    employee: root.employee,
    assessment_mode: assessmentMode,
    performance_inference_allowed: !!root.workflow_config?.performance_inference_allowed,
    development_inference_allowed: !!root.workflow_config?.development_inference_allowed,
    idp_input: idpInput
  }
}];
```

# 5. `Code / Build One2One Employee Payload v2`

Mode: `Run Once for All Items`

function safeArray(v) {
  return Array.isArray(v) ? v : [];
}

const competencyResults = $items("Code / Normalize Competency Result", 0, 0).map(i => i.json);

    development_inference_allowed: !!root.workflow_config?.development_inference_allowed,

    summary: employeeSummary,
    top_competencies: topCompetencies,
    idp: idp.idp || [],

    guidance: {
      audience: 'employee',
      tone: 'reflective_and_actionable'

# 11. Мой практический совет

Чтобы не утонуть, я бы внедрял в таком порядке:

1. `Build Competency Payloads v2`
2. matrix-only summary prompt
3. `Build IDP Payload v2`
4. `Build One2One Employee Payload v2`
5. `Build One2One Manager Payload v2`

Сначала добей, чтобы:

* payloads собирались
* LLM не валился
* результат не делал ложных performance-выводов

А уже потом можно аккуратно добавлять:

* self-assessment ingestion
* employee_issues normalization
* richer evidence scoring

Следующим сообщением я могу собрать тебе **готовую финальную ноду `Build Final Snapshot`**, которая объединит legacy-flow и matrix-flow в один output contract.
