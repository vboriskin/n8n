Да — вот **упрощённый validator для MVP**. Он проверяет только самое важное:

* ответ вообще парсится как JSON
* есть `id`, `name`, `matched_competencies`
* `matched_competencies` — массив
* каждая компетенция:

  * входит в допустимый список
  * не дублируется
  * имеет числовой `score >= 0.7`
  * имеет непустой `reason`

Он мягче production-версии и проще для первого запуска.

## Code node: `Validate_MVP`

```javascript
function safeParseJSON(text) {
  if (typeof text !== 'string') {
    return { ok: false, parsed: null, error: 'llm_response is not a string' };
  }

  try {
    return { ok: true, parsed: JSON.parse(text.trim()), error: null };
  } catch (e) {
    return { ok: false, parsed: null, error: e.message };
  }
}

return items.map(item => {
  const input = item.json;
  const errors = [];
  const warnings = [];

  const parseResult = safeParseJSON(input.llm_response);

  if (!parseResult.ok) {
    return {
      json: {
        ...input,
        parsed_result: null,
        validation_status: 'invalid',
        validation_errors: [`JSON parse error: ${parseResult.error}`],
        validation_warnings: [],
        retry_recommended: true,
      },
    };
  }

  const parsed = parseResult.parsed;

  if (!parsed || typeof parsed !== 'object' || Array.isArray(parsed)) {
    errors.push('Root JSON must be an object');
  }

  if (!parsed?.hasOwnProperty('id')) {
    errors.push('Missing field: id');
  }

  if (!parsed?.hasOwnProperty('name')) {
    errors.push('Missing field: name');
  }

  if (!parsed?.hasOwnProperty('matched_competencies')) {
    errors.push('Missing field: matched_competencies');
  }

  if (!Array.isArray(parsed?.matched_competencies)) {
    errors.push('matched_competencies must be an array');
  }

  const allowedCompetencies = new Set(
    Array.isArray(input.competencies)
      ? input.competencies.map(x => String(x).trim())
      : []
  );

  const seen = new Set();

  if (Array.isArray(parsed?.matched_competencies)) {
    for (let i = 0; i < parsed.matched_competencies.length; i++) {
      const mc = parsed.matched_competencies[i];

      if (!mc || typeof mc !== 'object' || Array.isArray(mc)) {
        errors.push(`matched_competencies[${i}] must be an object`);
        continue;
      }

      const name = String(mc.name ?? '').trim();
      const reason = String(mc.reason ?? '').trim();
      const score = mc.score;

      if (!name) {
        errors.push(`matched_competencies[${i}].name is empty`);
      } else {
        if (!allowedCompetencies.has(name)) {
          errors.push(`matched_competencies[${i}].name is not in competencies list: ${name}`);
        }

        if (seen.has(name)) {
          errors.push(`Duplicate competency: ${name}`);
        }

        seen.add(name);
      }

      if (typeof score !== 'number' || Number.isNaN(score)) {
        errors.push(`matched_competencies[${i}].score must be a number`);
      } else if (score < 0.7) {
        errors.push(`matched_competencies[${i}].score must be >= 0.7`);
      }

      if (!reason) {
        errors.push(`matched_competencies[${i}].reason is empty`);
      }
    }
  }

  if (String(parsed?.id ?? '') !== String(input.id ?? '')) {
    warnings.push('Returned id differs from input id');
  }

  if (typeof parsed?.name === 'string' && typeof input.name_clean === 'string') {
    if (parsed.name !== input.name_clean) {
      warnings.push('Returned name differs from input name');
    }
  }

  return {
    json: {
      ...input,
      parsed_result: parsed,
      validation_status: errors.length ? 'invalid' : 'valid',
      validation_errors: errors,
      validation_warnings: warnings,
      retry_recommended: errors.length > 0,
    },
  };
});
```

## Что делать на ветке `invalid`

Пока для MVP есть 2 простых варианта:

### Вариант 1

Сразу в manual review

### Вариант 2

Сразу на один retry через тот же OpenAI prompt с короткой инструкцией:

```text
Исправь предыдущий ответ.
Верни только валидный JSON.
Используй только компетенции из входного списка.
Не добавляй дубликаты.
Если не уверен — убери компетенцию.
```

---

## Чем этот MVP validator отличается от полного

Он **не проверяет**:

* максимум 5 компетенций
* длину `reason`
* whitelist reason
* сортировку по `score`
* suspicious cases
* слишком похожие score
* общие названия курсов

Это нормально для первого запуска.
Задача MVP — быстро понять, что пайплайн живой и JSON стабильно проходит.
