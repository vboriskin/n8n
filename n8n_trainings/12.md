Ниже — вариант именно под **n8n + OpenAI node**, без HTTP Request и без отдельной ноды GigaChat.

Сделаю в формате “копируй-вставляй”:

* какие ноды создать
* что в них вставить
* какие поля использовать
* как связать первый и второй проход

Буду исходить из такой логики:

* на входе есть курс и массив `competencies`
* OpenAI node возвращает текстовый ответ модели
* дальше мы валидируем JSON

```javascript
function cleanText(value, fallback = '') {
  let text = String(value ?? fallback);

  text = text.replace(/<[^>]*>/g, ' ');
  text = text.replace(/&nbsp;/g, ' ');
  text = text.replace(/\s+/g, ' ').trim();

  return text || fallback;
}

return items.map(item => {
  const j = item.json;

  const id = j.id ?? '';
  const name = cleanText(j.name, 'Без названия');
  const description = cleanText(j.description, 'Нет описания');
  const type = cleanText(j.type, 'Не указан');
  const duration = cleanText(j.duration, 'Не указана');

  const competencies = Array.isArray(j.competencies)
    ? j.competencies.map(x => String(x ?? '').trim()).filter(Boolean)
    : [];

  return {
    json: {
      ...j,
      id,
      name,
      description,
      type,
      duration,
      competencies,
      name_clean: name,
      description_clean: description,
      type_clean: type,
      duration_clean: duration,
      prompt_version: 'competency_matcher_v1',
    },
  };
});
```

# 04_Extract_Response_Pass1

Тип: `Code`

OpenAI node в n8n в зависимости от версии может вернуть ответ по-разному. Ниже extractor, который пытается вытащить текст из самых частых мест.

```javascript
function extractOpenAIText(j) {
  return (
    j.output_text ??
    j.text ??
    j.content ??
    j.message ??
    j.response ??
    j?.choices?.[0]?.message?.content ??
    j?.data?.choices?.[0]?.message?.content ??
    j?.messages?.[0]?.content ??
    ''
  );
}

return items.map(item => {
  const j = item.json;

  let llm_response = extractOpenAIText(j);

  if (Array.isArray(llm_response)) {
    llm_response = llm_response.map(x => {
      if (typeof x === 'string') return x;
      if (x?.text) return x.text;
      return JSON.stringify(x);
    }).join('\n');
  }

  llm_response = String(llm_response ?? '').trim();

  return {
    json: {
      ...j,
      llm_response,
    },
  };
});
```

# 05_Validate_Pass1

Тип: `Code`

Вставь этот валидатор целиком:

```javascript
function safeParseJSON(text) {
  if (typeof text !== 'string') {
    return { ok: false, error: 'llm_response is not a string', parsed: null };
  }

  const trimmed = text.trim();

  try {
    return { ok: true, error: null, parsed: JSON.parse(trimmed) };
  } catch (e) {
    return { ok: false, error: `JSON parse error: ${e.message}`, parsed: null };
  }
}

function countWords(text) {
  if (typeof text !== 'string') return 0;
  return text.trim().split(/\s+/).filter(Boolean).length;
}

function isSortedDescByScore(arr) {
  for (let i = 1; i < arr.length; i++) {
    if ((arr[i - 1].score ?? -Infinity) < (arr[i].score ?? -Infinity)) {
      return false;
    }
  }
  return true;
}

function normalizeCompetencyName(value) {
  return String(value ?? '').trim();
}

function validateResult(input, parsed) {
  const errors = [];
  const warnings = [];

  const allowedCompetencies = Array.isArray(input.competencies)
    ? input.competencies.map(normalizeCompetencyName)
    : [];

  const allowedSet = new Set(allowedCompetencies);

  const allowedReasons = new Set([
    'указана в названии',
    'указана в описании',
    'основная тема курса',
    'значимая часть программы',
  ]);

  if (!parsed || typeof parsed !== 'object' || Array.isArray(parsed)) {
    errors.push('Root must be an object');
    return { valid: false, errors, warnings };
  }

  const rootKeys = Object.keys(parsed);
  const allowedRootKeys = ['id', 'name', 'matched_competencies'];

  for (const key of rootKeys) {
    if (!allowedRootKeys.includes(key)) {
      errors.push(`Unexpected root field: ${key}`);
    }
  }

  if (!('id' in parsed)) errors.push('Missing field: id');
  if (!('name' in parsed)) errors.push('Missing field: name');
  if (!('matched_competencies' in parsed)) errors.push('Missing field: matched_competencies');

  if (typeof parsed.name !== 'string') {
    errors.push('Field "name" must be a string');
  }

  if (!Array.isArray(parsed.matched_competencies)) {
    errors.push('Field "matched_competencies" must be an array');
    return { valid: false, errors, warnings };
  }

  if (parsed.matched_competencies.length > 5) {
    errors.push('matched_competencies contains more than 5 items');
  }

  const seenNames = new Set();

  parsed.matched_competencies.forEach((item, index) => {
    if (!item || typeof item !== 'object' || Array.isArray(item)) {
      errors.push(`matched_competencies[${index}] must be an object`);
      return;
    }

    const itemKeys = Object.keys(item);
    const allowedItemKeys = ['name', 'score', 'reason'];

    for (const key of itemKeys) {
      if (!allowedItemKeys.includes(key)) {
        errors.push(`Unexpected field in matched_competencies[${index}]: ${key}`);
      }
    }

    if (!('name' in item)) errors.push(`Missing name in matched_competencies[${index}]`);
    if (!('score' in item)) errors.push(`Missing score in matched_competencies[${index}]`);
    if (!('reason' in item)) errors.push(`Missing reason in matched_competencies[${index}]`);

    const competencyName = normalizeCompetencyName(item.name);

    if (typeof item.name !== 'string' || !competencyName) {
      errors.push(`Invalid competency name at index ${index}`);
    } else {
      if (!allowedSet.has(competencyName)) {
        errors.push(`Competency not in allowed list at index ${index}: ${competencyName}`);
      }
      if (seenNames.has(competencyName)) {
        errors.push(`Duplicate competency at index ${index}: ${competencyName}`);
      }
      seenNames.add(competencyName);
    }

    if (typeof item.score !== 'number' || Number.isNaN(item.score)) {
      errors.push(`Score must be a number at index ${index}`);
    } else {
      if (item.score < 0.7) {
        errors.push(`Score below 0.70 at index ${index}`);
      }
      if (item.score > 1.0) {
        errors.push(`Score above 1.00 at index ${index}`);
      }
    }

    if (typeof item.reason !== 'string' || !item.reason.trim()) {
      errors.push(`Invalid reason at index ${index}`);
    } else {
      const reason = item.reason.trim();
      const words = countWords(reason);
      if (words > 5) {
        errors.push(`Reason has more than 5 words at index ${index}`);
      }
      if (!allowedReasons.has(reason)) {
        errors.push(`Reason is not allowed at index ${index}: ${reason}`);
      }
    }
  });

  if (Array.isArray(parsed.matched_competencies) && !isSortedDescByScore(parsed.matched_competencies)) {
    warnings.push('matched_competencies is not sorted by descending score');
  }

  if (String(parsed.id) !== String(input.id)) {
    warnings.push('Returned id differs from input id');
  }

  if (typeof parsed.name === 'string' && parsed.name !== input.name_clean) {
    warnings.push('Returned name differs from input name');
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings,
  };
}

function detectSuspicious(input, parsed, validation) {
  const reasons = [];

  const description = String(input.description_clean ?? '').trim();
  const courseName = String(input.name_clean ?? '').trim();
  const matches = Array.isArray(parsed?.matched_competencies) ? parsed.matched_competencies : [];

  if (!description || description === 'Нет описания') {
    reasons.push('empty_description');
  }

  if (description.length > 0 && description.length < 40) {
    reasons.push('very_short_description');
  }

  if (/^(введение|основы|базовый курс|обзор)$/i.test(courseName)) {
    reasons.push('generic_course_name');
  }

  if (matches.length >= 5) {
    reasons.push('too_many_matches');
  }

  if (matches.length > 0) {
    const lowConfidenceCount = matches.filter(x => typeof x.score === 'number' && x.score <= 0.74).length;
    if (lowConfidenceCount === matches.length) {
      reasons.push('all_scores_near_threshold');
    }

    const uniqueReasons = new Set(matches.map(x => String(x.reason ?? '').trim()));
    if (uniqueReasons.size === 1 && matches.length >= 3) {
      reasons.push('reasons_are_too_uniform');
    }

    const scores = matches.map(x => x.score).filter(x => typeof x === 'number');
    if (scores.length >= 3) {
      const max = Math.max(...scores);
      const min = Math.min(...scores);
      if ((max - min) <= 0.03) {
        reasons.push('scores_too_similar');
      }
    }
  }

  if (validation.warnings.includes('matched_competencies is not sorted by descending score')) {
    reasons.push('not_sorted');
  }

  return reasons;
}

return items.map(item => {
  const input = item.json;

  const parseResult = safeParseJSON(input.llm_response);

  let parsed = null;
  let validation = { valid: false, errors: [], warnings: [] };
  let suspiciousReasons = [];
  let status = 'invalid';

  if (parseResult.ok) {
    parsed = parseResult.parsed;
    validation = validateResult(input, parsed);

    if (validation.valid) {
      suspiciousReasons = detectSuspicious(input, parsed, validation);
      status = suspiciousReasons.length > 0 ? 'retry_needed' : 'valid';
    } else {
      status = 'invalid';
    }
  } else {
    validation.errors.push(parseResult.error);
    status = 'invalid';
  }

  return {
    json: {
      ...input,
      parsed_result: parsed,
      validation_status: status,
      validation_errors: validation.errors,
      validation_warnings: validation.warnings,
      suspicious_reasons: suspiciousReasons,
      retry_recommended: status !== 'valid',
      retry_prompt_context: {
        id: input.id,
        name: input.name_clean,
        description: input.description_clean,
        type: input.type_clean,
        duration: input.duration_clean,
        competencies: input.competencies ?? [],
        previous_llm_response: input.llm_response ?? '',
        parsed_result: parsed,
        validation_errors: validation.errors,
        validation_warnings: validation.warnings,
        suspicious_reasons: suspiciousReasons,
      },
    },
  };
});
```

```javascript
  prompt_version: $json.prompt_version || 'competency_matcher_v1'
}
}}
```

# Полезная альтернатива: Structured Output в OpenAI node

Если в твоей версии n8n/OpenAI node есть режим структурированного ответа, используй его. Тогда можно дополнительно задать схему:

```json
{
  "type": "object",
  "additionalProperties": false,
  "required": ["id", "name", "matched_competencies"],
  "properties": {
    "id": {
      "type": ["string", "number"]
    },
    "name": {
      "type": "string"
    },
    "matched_competencies": {
      "type": "array",
      "maxItems": 5,
      "items": {
        "type": "object",
        "additionalProperties": false,
        "required": ["name", "score", "reason"],
        "properties": {
          "name": { "type": "string" },
          "score": { "type": "number", "minimum": 0.7, "maximum": 1.0 },
          "reason": {
            "type": "string",
            "enum": [
              "указана в названии",
              "указана в описании",
              "основная тема курса",
              "значимая часть программы"
            ]
          }
        }
      }
    }
  }
}
```

Но даже при structured output я бы всё равно оставил `Validate` node, потому что надо проверять:

* входит ли компетенция в реальный входной список
* нет ли дубликатов
* не слишком ли подозрителен ответ

# На что обратить внимание в OpenAI node

У разных версий n8n поле ответа может называться по-разному. Если после `03_OpenAI_Pass1` extractor не находит текст, открой output node и посмотри, где лежит ответ. Чаще всего это одно из:

* `content`
* `text`
* `message`
* `output_text`
* `choices[0].message.content`

Если покажешь пример JSON-выхода именно твоей OpenAI node, я под него сразу дам точный extractor без универсальной “простыни”.
