Ниже — готовый набор для n8n:

1. JSON Schema для проверки ответа модели
2. `Code` node для n8n, который:

   * парсит ответ
   * валидирует
   * ставит статусы `valid / invalid / retry_needed`
   * готовит полезные поля для второго прохода

Я дам сначала **рекомендуемую схему**, потом **готовый код**.

---

# 1. JSON Schema

Это базовая schema для структуры ответа.

```json
{
  "type": "object",
  "additionalProperties": false,
  "required": ["id", "name", "matched_competencies"],
  "properties": {
    "id": {
      "type": ["string", "number"]
    },
    "name": {
      "type": "string"
    },
    "matched_competencies": {
      "type": "array",
      "maxItems": 5,
      "items": {
        "type": "object",
        "additionalProperties": false,
        "required": ["name", "score", "reason"],
        "properties": {
          "name": {
            "type": "string"
          },
          "score": {
            "type": "number",
            "minimum": 0.7,
            "maximum": 1
          },
          "reason": {
            "type": "string"
          }
        }
      }
    }
  }
}
```

Но одной schema мало.
Нужна ещё **бизнес-валидация**, потому что JSON Schema не проверит удобно:

* входит ли `name` в исходный список компетенций
* есть ли дубликаты
* не больше ли 5 слов в `reason`
* отсортирован ли массив по убыванию `score`

Поэтому ниже — нормальный production-код.

# 2. Code node для n8n

Предположим, во входе каждого item есть:

* `id`
* `name`
* `description`
* `competencies` — массив допустимых компетенций
* `llm_response` — сырой ответ модели строкой

Если у тебя поле называется иначе, просто поменяй имя в коде.

## Готовый код для `Code` node

```javascript
const itemsOut = [];

function safeParseJSON(text) {
  if (typeof text !== 'string') {
    return { ok: false, error: 'llm_response is not a string', parsed: null };
  }

  const trimmed = text.trim();

  try {
    return { ok: true, error: null, parsed: JSON.parse(trimmed) };
  } catch (e) {
    return { ok: false, error: `JSON parse error: ${e.message}`, parsed: null };
  }
}

function countWords(text) {
  if (typeof text !== 'string') return 0;
  return text.trim().split(/\s+/).filter(Boolean).length;
}

function isSortedDescByScore(arr) {
  for (let i = 1; i < arr.length; i++) {
    if ((arr[i - 1].score ?? -Infinity) < (arr[i].score ?? -Infinity)) {
      return false;
    }
  }
  return true;
}

function normalizeCompetencyName(value) {
  return String(value ?? '').trim();
}

function validateResult(input, parsed) {
  const errors = [];
  const warnings = [];

  const allowedCompetencies = Array.isArray(input.competencies)
    ? input.competencies.map(normalizeCompetencyName)
    : [];

  const allowedSet = new Set(allowedCompetencies);

  if (!parsed || typeof parsed !== 'object' || Array.isArray(parsed)) {
    errors.push('Root must be an object');
    return { valid: false, errors, warnings };
  }

  const rootKeys = Object.keys(parsed);
  const allowedRootKeys = ['id', 'name', 'matched_competencies'];

  for (const key of rootKeys) {
    if (!allowedRootKeys.includes(key)) {
      errors.push(`Unexpected root field: ${key}`);
    }
  }

  if (!('id' in parsed)) errors.push('Missing field: id');
  if (!('name' in parsed)) errors.push('Missing field: name');
  if (!('matched_competencies' in parsed)) errors.push('Missing field: matched_competencies');

  if (typeof parsed.name !== 'string') {
    errors.push('Field "name" must be a string');
  }

  if (!Array.isArray(parsed.matched_competencies)) {
    errors.push('Field "matched_competencies" must be an array');
    return { valid: false, errors, warnings };
  }

  if (parsed.matched_competencies.length > 5) {
    errors.push('matched_competencies contains more than 5 items');
  }

  const seenNames = new Set();

  parsed.matched_competencies.forEach((item, index) => {
    if (!item || typeof item !== 'object' || Array.isArray(item)) {
      errors.push(`matched_competencies[${index}] must be an object`);
      return;
    }

    const itemKeys = Object.keys(item);
    const allowedItemKeys = ['name', 'score', 'reason'];

    for (const key of itemKeys) {
      if (!allowedItemKeys.includes(key)) {
        errors.push(`Unexpected field in matched_competencies[${index}]: ${key}`);
      }
    }

    if (!('name' in item)) errors.push(`Missing name in matched_competencies[${index}]`);
    if (!('score' in item)) errors.push(`Missing score in matched_competencies[${index}]`);
    if (!('reason' in item)) errors.push(`Missing reason in matched_competencies[${index}]`);

    const competencyName = normalizeCompetencyName(item.name);

    if (typeof item.name !== 'string' || !competencyName) {
      errors.push(`Invalid competency name at index ${index}`);
    } else {
      if (!allowedSet.has(competencyName)) {
        errors.push(`Competency not in allowed list at index ${index}: ${competencyName}`);
      }
      if (seenNames.has(competencyName)) {
        errors.push(`Duplicate competency at index ${index}: ${competencyName}`);
      }
      seenNames.add(competencyName);
    }

    if (typeof item.score !== 'number' || Number.isNaN(item.score)) {
      errors.push(`Score must be a number at index ${index}`);
    } else {
      if (item.score < 0.7) {
        errors.push(`Score below 0.70 at index ${index}`);
      }
      if (item.score > 1.0) {
        errors.push(`Score above 1.00 at index ${index}`);
      }
    }

    if (typeof item.reason !== 'string' || !item.reason.trim()) {
      errors.push(`Invalid reason at index ${index}`);
    } else {
      const words = countWords(item.reason);
      if (words > 5) {
        errors.push(`Reason has more than 5 words at index ${index}`);
      }
    }
  });

  if (Array.isArray(parsed.matched_competencies) && !isSortedDescByScore(parsed.matched_competencies)) {
    warnings.push('matched_competencies is not sorted by descending score');
  }

  if (
    parsed.id !== input.id &&
    String(parsed.id) !== String(input.id)
  ) {
    warnings.push('Returned id differs from input id');
  }

  if (
    typeof parsed.name === 'string' &&
    typeof input.name === 'string' &&
    parsed.name !== input.name
  ) {
    warnings.push('Returned name differs from input name');
  }

  return {
    valid: errors.length === 0,
    errors,
    warnings,
  };
}

function detectSuspicious(input, parsed, validation) {
  const reasons = [];

  const description = String(input.description ?? '').trim();
  const courseName = String(input.name ?? '').trim();
  const matches = Array.isArray(parsed?.matched_competencies) ? parsed.matched_competencies : [];

  if (!description || description === 'Нет описания') {
    reasons.push('empty_description');
  }

  if (description.length > 0 && description.length < 40) {
    reasons.push('very_short_description');
  }

  if (/^(введение|основы|базовый курс|обзор)$/i.test(courseName)) {
    reasons.push('generic_course_name');
  }

  if (matches.length >= 5) {
    reasons.push('too_many_matches');
  }

  if (matches.length > 0) {
    const lowConfidenceCount = matches.filter(x => typeof x.score === 'number' && x.score <= 0.74).length;
    if (lowConfidenceCount === matches.length) {
      reasons.push('all_scores_near_threshold');
    }

    const uniqueReasons = new Set(matches.map(x => String(x.reason ?? '').trim()));
    if (uniqueReasons.size === 1 && matches.length >= 3) {
      reasons.push('reasons_are_too_uniform');
    }

    const scores = matches
      .map(x => x.score)
      .filter(x => typeof x === 'number');

    if (scores.length >= 3) {
      const max = Math.max(...scores);
      const min = Math.min(...scores);
      if ((max - min) <= 0.03) {
        reasons.push('scores_too_similar');
      }
    }
  }

  if (validation.warnings.includes('matched_competencies is not sorted by descending score')) {
    reasons.push('not_sorted');
  }

  return reasons;
}

for (const item of items) {
  const input = item.json;

  const llmRaw = input.llm_response;
  const parseResult = safeParseJSON(llmRaw);

  let parsed = null;
  let validation = { valid: false, errors: [], warnings: [] };
  let suspiciousReasons = [];
  let status = 'invalid';

  if (parseResult.ok) {
    parsed = parseResult.parsed;
    validation = validateResult(input, parsed);

    if (validation.valid) {
      suspiciousReasons = detectSuspicious(input, parsed, validation);
      status = suspiciousReasons.length > 0 ? 'retry_needed' : 'valid';
    } else {
      status = 'invalid';
    }
  } else {
    validation.errors.push(parseResult.error);
    status = 'invalid';
  }

  itemsOut.push({
    json: {
      ...input,

      validation_status: status,
      parsed_result: parsed,
      validation_errors: validation.errors,
      validation_warnings: validation.warnings,
      suspicious_reasons: suspiciousReasons,

      retry_recommended: status !== 'valid',

      retry_prompt_context: {
        id: input.id,
        name: input.name,
        description: input.description ?? 'Нет описания',
        type: input.type ?? 'Не указан',
        duration: input.duration ?? 'Не указана',
        competencies: input.competencies ?? [],
        previous_llm_response: input.llm_response ?? '',
        parsed_result: parsed,
        validation_errors: validation.errors,
        validation_warnings: validation.warnings,
        suspicious_reasons: suspiciousReasons,
      },
    },
  });
}

return itemsOut;
```

# 3. Что делает этот код

Он возвращает для каждого item:

* `validation_status`

  * `valid`
  * `invalid`
  * `retry_needed`

* `parsed_result` — распарсенный JSON модели

* `validation_errors` — критические ошибки

* `validation_warnings` — некритичные замечания

* `suspicious_reasons` — причины отправить на второй проход

* `retry_recommended` — булево поле

* `retry_prompt_context` — готовый контекст для retry

---

# 4. Как интерпретировать статусы

### `valid`

Можно сохранять результат как финальный.

### `invalid`

Ответ плохой:

* не JSON
* сломанная структура
* компетенции вне списка
* дубликаты
* score вне диапазона
* и так далее

Такой ответ лучше отправлять в retry.

### `retry_needed`

Формально ответ валиден, но подозрителен.
Например:

* пустое описание
* слишком общее название
* все score около порога
* слишком много совпадений
* однотипные reason

Это хороший кандидат на второй проход через self-check prompt.

# 5. Готовый retry prompt для второго прохода в n8n

Можно собирать из `retry_prompt_context`.

```text
Проверь классификацию курса заново и верни исправленный результат.

Допустимые компетенции:
{{ JSON.stringify($json.retry_prompt_context.competencies) }}

Данные курса:
{
  "id": {{ JSON.stringify($json.retry_prompt_context.id) }},
  "name": {{ JSON.stringify($json.retry_prompt_context.name) }},
  "description": {{ JSON.stringify($json.retry_prompt_context.description) }},
  "type": {{ JSON.stringify($json.retry_prompt_context.type) }},
  "duration": {{ JSON.stringify($json.retry_prompt_context.duration) }}
}

Предыдущий ответ модели:
{{ JSON.stringify($json.retry_prompt_context.previous_llm_response) }}

Ошибки валидации:
{{ JSON.stringify($json.retry_prompt_context.validation_errors) }}

Предупреждения:
{{ JSON.stringify($json.retry_prompt_context.validation_warnings) }}

Причины повторной проверки:
{{ JSON.stringify($json.retry_prompt_context.suspicious_reasons) }}

Требования:
- Используй только компетенции из допустимого списка
- Удаляй всё, что не подтверждается напрямую названием или описанием
- Удаляй всё, что относится не к основной теме курса
- Если есть сомнение, исключай компетенцию
- Максимум 5 компетенций
- Не добавляй дубликаты
- Отсортируй matched_competencies по убыванию score
- Все score должны быть числами от 0.70 до 1.00
- reason максимум 5 слов

Верни только валидный JSON:
{
  "id": "...",
  "name": "...",
  "matched_competencies": [
    {
      "name": "...",
      "score": 0.85,
      "reason": "..."
    }
  ]
}
```

# 6. Если хочешь совсем жёстко: автоисправление сортировки

Сейчас код только предупреждает, если массив не отсортирован.
Можно сделать автопочинку.

Внутри `if (validation.valid)` добавь:

```javascript
if (parsed && Array.isArray(parsed.matched_competencies)) {
  parsed.matched_competencies.sort((a, b) => b.score - a.score);
}
```

Тогда результат будет автоматически выровнен.
Но я бы оставил warning, чтобы видеть качество модели.

# 7. Упрощённая схема потока в n8n

Практически:

### Node 1 — Prepare Input

Готовишь:

* `id`
* `name`
* `description`
* `type`
* `duration`
* `competencies`

### Node 2 — LLM

Получаешь `llm_response`

### Node 3 — Code

Вставляешь код выше

### Node 4 — IF

Проверяешь `validation_status`

* `valid` → сохранить
* `invalid` или `retry_needed` → второй LLM проход

### Node 5 — Retry LLM

Используешь retry prompt

### Node 6 — Code validate again

Снова тот же валидатор

### Node 7

* если `valid` → сохранить
* иначе → manual review

# 9. Маленькое улучшение: canonical reason

Чтобы уменьшить вариативность, можно вообще разрешить только фиксированный словарь `reason`.

Например только:

* `указана в названии`
* `указана в описании`
* `основная тема курса`
* `значимая часть программы`

Тогда в validator можно добавить проверку по whitelist. Это сильно упрощает downstream.

Пример проверки:

```javascript
const allowedReasons = new Set([
  'указана в названии',
  'указана в описании',
  'основная тема курса',
  'значимая часть программы',
]);
```

И затем:

```javascript
if (!allowedReasons.has(item.reason.trim())) {
  errors.push(`Reason is not allowed at index ${index}: ${item.reason}`);
}
```

Для production это часто лучше, чем свободный текст.
